16  package org.atmosphere.interceptor;
17  
18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.AtmosphereInterceptorAdapter;
20  import org.atmosphere.cpr.AtmosphereRequest;
21  import org.atmosphere.cpr.AtmosphereResource;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  
25  import static org.atmosphere.util.IOUtils.isBodyEmpty;
26  import static org.atmosphere.util.IOUtils.readEntirely;
27  
28  /**
29   * This read the request's body and invoke the associated {@link org.atmosphere.cpr.Broadcaster} of an {@link AtmosphereResource}.
30   * The broadcast always happens AFTER the request has been delivered to an {@link org.atmosphere.cpr.AtmosphereHandler}.
31   *
32   * @author Jeanfrancois Arcand
33   */
34  public class BroadcastOnPostAtmosphereInterceptor extends AtmosphereInterceptorAdapter {
35  
36      private final static Logger logger = LoggerFactory.getLogger(BroadcastOnPostAtmosphereInterceptor.class);
37  
38      @Override
39      public Action inspect(AtmosphereResource r) {
40          return Action.CONTINUE;
41      }
42  
43  
44      @Override
45      public void postInspect(AtmosphereResource r) {
46          if (r.getRequest().getMethod().equalsIgnoreCase("POST")) {
47              AtmosphereRequest request = r.getRequest();
48              Object o = readEntirely(r);
49              if (isBodyEmpty(o)) {
50                  logger.warn("{} received an empty body", request);
51                  return;
52              }
53              r.getBroadcaster().broadcast(o);
54          }
55      }
56  }
