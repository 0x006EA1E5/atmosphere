16  package org.atmosphere.interceptor;
17  
18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.ApplicationConfig;
20  import org.atmosphere.cpr.AtmosphereConfig;
21  import org.atmosphere.cpr.AtmosphereInterceptorAdapter;
22  import org.atmosphere.cpr.AtmosphereRequest;
23  import org.atmosphere.cpr.AtmosphereResource;
24  import org.atmosphere.cpr.AtmosphereResponse;
25  import org.atmosphere.cpr.HeaderConfig;
26  import org.atmosphere.util.Utils;
27  
28  /**
29   * CORS support.
30   *
31   * @author Janusz Sobolewski
32   */
33  public class CorsInterceptor extends AtmosphereInterceptorAdapter {
34  
35      private final String EXPOSE_HEADERS = "X-Atmosphere-tracking-id, " + HeaderConfig.X_HEARTBEAT_SERVER;
36  
37      private boolean enableAccessControl = true;
38  
39      @Override
40      public void configure(AtmosphereConfig config) {
41          String ac = config.getInitParameter(ApplicationConfig.DROP_ACCESS_CONTROL_ALLOW_ORIGIN_HEADER);
42          if (ac != null) {
43              enableAccessControl = !Boolean.parseBoolean(ac);
44          }
45      }
46  
47      @Override
48      public Action inspect(AtmosphereResource r) {
49  
50          if (Utils.webSocketMessage(r)) return Action.CONTINUE;
51  
52          if (!enableAccessControl) return Action.CONTINUE;
53  
54          AtmosphereRequest req = r.getRequest();
55          AtmosphereResponse res = r.getResponse();
56  
57          if (req.getHeader("Origin") != null && res.getHeader("Access-Control-Allow-Origin") == null) {
58              res.addHeader("Access-Control-Allow-Origin", req.getHeader("Origin"));
59              res.addHeader("Access-Control-Expose-Headers", EXPOSE_HEADERS);
60              res.setHeader("Access-Control-Allow-Credentials", "true");
61          }
62  
63          if ("OPTIONS".equals(req.getMethod())) {
64              res.setHeader("Access-Control-Allow-Methods", "OPTIONS, GET, POST");
65              res.setHeader("Access-Control-Allow-Headers",
66                      "Origin, Content-Type, AuthToken, X-Atmosphere-Framework, X-Requested-With, "
67                              + EXPOSE_HEADERS
68                              + ", X-Atmosphere-Transport, X-Atmosphere-TrackMessageSize, X-atmo-protocol");
69              res.setHeader("Access-Control-Max-Age", "-1");
70  
71              return Action.SKIP_ATMOSPHEREHANDLER;
72          }
73  
74          return Action.CONTINUE;
75      }
76  
77      public boolean enableAccessControl() {
78          return enableAccessControl;
79      }
80  
81      public CorsInterceptor enableAccessControl(boolean enableAccessControl) {
82          this.enableAccessControl = enableAccessControl;
83          return this;
84      }
85  
86      @Override
87      public PRIORITY priority() {
88          return InvokationOrder.FIRST_BEFORE_DEFAULT;
89      }
90  
91      @Override
92      public String toString() {
93          return "CORS Interceptor Support";
94      }
95  }
