16  package org.atmosphere.interceptor;

18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.AsyncIOInterceptorAdapter;
20  import org.atmosphere.cpr.AsyncIOWriter;
21  import org.atmosphere.cpr.AtmosphereInterceptorAdapter;
22  import org.atmosphere.cpr.AtmosphereInterceptorWriter;

24  import org.atmosphere.cpr.AtmosphereResource;
25  import org.atmosphere.cpr.AtmosphereResponse;
26  import org.atmosphere.cpr.HeaderConfig;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;

30  import java.io.IOException;

32  /**
33   * JSONP Transport Support.
34   *
35   * @author Jeanfrancois Arcand
36   */
37  public class JSONPAtmosphereInterceptor extends AtmosphereInterceptorAdapter {
38  
39      private static final Logger logger = LoggerFactory.getLogger(JSONPAtmosphereInterceptor.class);
40      private static final String END_CHUNK = "\"});";
41      private static final Object START_CHUNK = "({\"message\" : \"";
42  
43      @Override
44      public Action inspect(AtmosphereResource r) {
45          final AtmosphereRequest request = r.getRequest();
46          final AtmosphereResponse response = r.getResponse();
47          if (r.transport().equals(AtmosphereResource.TRANSPORT.JSONP)) {
48              super.inspect(r);
49  
50              AsyncIOWriter writer = response.getAsyncIOWriter();
51              if (AtmosphereInterceptorWriter.class.isAssignableFrom(writer.getClass())) {
52                  AtmosphereInterceptorWriter.class.cast(writer).interceptor(new AsyncIOInterceptorAdapter() {
53  
54                      String callbackName() {
55                          return request.getParameter(HeaderConfig.JSONP_CALLBACK_NAME);
56                      }
57  
58                      @Override
59                      public void prePayload(AtmosphereResponse response, byte[] data, int offset, int length) {
60                          String callbackName = callbackName();
61                          response.write(callbackName + START_CHUNK);
62                      }
63  
64                      @Override
65                      public byte[] transformPayload(AtmosphereResponse response, byte[] responseDraft, byte[] data) throws IOException {
66                          String charEncoding = response.getCharacterEncoding() == null ? "UTF-8" : response.getCharacterEncoding();
67                          // TODO: TOTALLY INEFFICIENT. We MUST uses binary replacement instead.
68                          String s = new String(responseDraft, charEncoding);
69                          return s.replaceAll("(['\"\\/])", "\\\\$1")
70                                  .replaceAll("\b", "\\\\b").replaceAll("\n", "\\\\n")
71                                  .replaceAll("\t", "\\\\t").replaceAll("\f", "\\\\f")
72                                  .replaceAll("\r", "\\\\r").getBytes(charEncoding);
73                      }
74  
75                      @Override
76                      public void postPayload(AtmosphereResponse response, byte[] data, int offset, int length) {
77                          response.write(END_CHUNK, true);
78                      }
79                  });
80              } else {
81                  logger.warn("Unable to apply {}. Your AsyncIOWriter must implement {}", getClass().getName(), AtmosphereInterceptorWriter.class.getName());
82              }
83          }
84          return Action.CONTINUE;
85      }
86  
87      @Override
88      public String toString() {
89          return "JSONP Interceptor Support";
90      }
91  }
