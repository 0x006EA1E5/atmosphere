53  /**
54   * <p>
55   * An interceptor that send whitespace every in 60 seconds by default. Another value could be specified with the
56   * {@link org.atmosphere.cpr.ApplicationConfig#HEARTBEAT_INTERVAL_IN_SECONDS} in the atmosphere config. The heartbeat will be scheduled as soon as the
57   * request is suspended.
58   * </p>
59   * <p/>
60   * <p>
61   * Moreover, any client can ask for a particular value with the {@link HeaderConfig#X_HEARTBEAT_SERVER} header set in
62   * request. This value will be taken in consideration if it is greater than the configured value. Client can also
63   * specify the value "0" to disable heartbeat.
64   * </p>
65   * <p/>
66   * <p>
67   * Finally the server notifies thanks to the {@link JavaScriptProtocol} the desired heartbeat interval that the client
68   * should applies. This interceptor just manage the configured value and the {@link JavaScriptProtocol protocol} sends
69   * the value to the client.
70   * </p>
71   *
72   * @author Jeanfrancois Arcand
73   */
74  public class HeartbeatInterceptor extends AtmosphereInterceptorAdapter {
75      public final static String INTERCEPTOR_ADDED = HeartbeatInterceptor.class.getName();
76      public final static String HEARTBEAT_FUTURE = "heartbeat.future";
77

142     @Override
143     public void configure(final AtmosphereConfig config) {
144         // Server
145         String s = config.getInitParameter(HEARTBEAT_INTERVAL_IN_SECONDS);
146         if (s != null) {
147             heartbeatFrequencyInSeconds = Integer.valueOf(s);
148         }
149 
150         // Server
151         s = config.getInitParameter(HEARTBEAT_PADDING_CHAR);
152         if (s != null) {
153             try {
154                 paddingBytes = s.getBytes("UTF-8");
155             } catch (UnsupportedEncodingException e) {
156                 logger.error("", e);
157             }
158         }
159 
160         // Client
161         s = config.getInitParameter(CLIENT_HEARTBEAT_INTERVAL_IN_SECONDS);
162         if (s != null) {
163             clientHeartbeatFrequencyInSeconds = Integer.valueOf(s);
164         }
165 
166         heartBeat = ExecutorsFactory.getScheduler(config);
167 
168         resumeOnHeartbeat = config.getInitParameter(RESUME_ON_HEARTBEAT, true);
169         logger.info("HeartbeatInterceptor configured with padding value '{}', client frequency {} seconds and server frequency {} seconds", new String[]
170                 {new String(paddingBytes), String.valueOf(heartbeatFrequencyInSeconds), String.valueOf(clientHeartbeatFrequencyInSeconds)});
171 
172         this.config = config;
173     }

175     private static class Clock extends AtmosphereResourceEventListenerAdapter implements AllowInterceptor {
176         public Clock() {
177         }
178     }
179 
180     @Override
181     public Action inspect(final AtmosphereResource r) {
182         final AtmosphereResourceImpl impl = AtmosphereResourceImpl.class.cast(r);
183         final AtmosphereRequest request = impl.getRequest(false);
184         final AtmosphereResponse response = impl.getResponse(false);
185 
186         // Check heartbeat
187         if (clientHeartbeatFrequencyInSeconds > 0) {
188             byte[] body = new byte[0];
189             try {
190                 if (!request.getMethod().equalsIgnoreCase("GET")) {
191                     body = IOUtils.readEntirelyAsByte(r);
192                 }
193             } catch (IOException e) {
194                 logger.warn("", e);
195                 cancelF(request);
196                 return Action.CONTINUE;
197             }
198 
199             if (Arrays.equals(paddingBytes, body)) {
200                 // Dispatch an event to notify that a heartbeat has been intercepted
201                 // TODO: see https://github.com/Atmosphere/atmosphere/issues/1561
202                 final AtmosphereResourceEvent event = new HeartbeatAtmosphereResourceEvent(AtmosphereResourceImpl.class.cast(r));
203 
204                 if (AtmosphereResourceHeartbeatEventListener.class.isAssignableFrom(r.getAtmosphereHandler().getClass())) {
205                     r.addEventListener(new AtmosphereResourceEventListenerAdapter.OnHeartbeat() {
206                         @Override
207                         public void onHeartbeat(AtmosphereResourceEvent event) {
208                             AtmosphereResourceHeartbeatEventListener.class.cast(r.getAtmosphereHandler()).onHeartbeat(event);
209                         }
210                     });
211                 }
212 
213                 // Fire event
214                 r.notifyListeners(event);
215 
216                 return Action.CANCELLED;
217             }
218 
219             request.body(body);
220         }
221 
222         if (Utils.webSocketMessage(r)) return Action.CONTINUE;
223 
224         final int interval = extractHeartbeatInterval(impl);
225 
226         if (interval != 0) {
227             if (!Utils.pollableTransport(r.transport())) {
228                 super.inspect(r);
229                 final boolean wasSuspended = r.isSuspended();
230 
231                 // Suspended? Ok can clock now
232                 // Otherwise, the listener will do the job
233                 if (wasSuspended) {
234                     clock(interval, r, request, response);
235                 }
236 
237                 r.addEventListener(new Clock() {
238                     @Override
239                     public void onSuspend(AtmosphereResourceEvent event) {
240 
241                         // We did not clocked when this listener was added because connection was not already suspended
242                         if (!wasSuspended) {
243                             clock(interval, r, request, response);
244                         }
245                     }
246 
247                     @Override
248                     public void onResume(AtmosphereResourceEvent event) {
249                         cancelF(request);
250                     }
251 
252                     @Override
253                     public void onDisconnect(AtmosphereResourceEvent event) {
254                         cancelF(request);
255                     }
256 
257                     @Override
258                     public void onClose(AtmosphereResourceEvent event) {
259                         cancelF(request);
260                     }
261                 });
262             } else {
263                 return Action.CONTINUE;
264             }
265 
266             final AsyncIOWriter writer = response.getAsyncIOWriter();
267 
268             if (!Utils.resumableTransport(r.transport())
269                     && AtmosphereInterceptorWriter.class.isAssignableFrom(writer.getClass())
270                     && request.getAttribute(INTERCEPTOR_ADDED) == null) {
271                 AtmosphereInterceptorWriter.class.cast(writer).interceptor(new AsyncIOInterceptorAdapter() {
272 
273                     @Override
274                     public byte[] transformPayload(AtmosphereResponse response, byte[] responseDraft, byte[] data) throws IOException {
275                         cancelF(request);
276                         return responseDraft;
277                     }
278 
279                     @Override
280                     public void postPayload(final AtmosphereResponse response, byte[] data, int offset, int length) {
281                         logger.trace("Scheduling heartbeat for {}", r.uuid());
282                         clock(interval, r, request, response);
283                     }
284                 });
285                 request.setAttribute(INTERCEPTOR_ADDED, Boolean.TRUE);
286             }
287 
288         }
289 
290         return Action.CONTINUE;
291     }

293     /**
294      * <p>
295      * Extracts the heartbeat interval as explained in class description. This method could be overridden to change the
296      * the configuration points.
297      * </p>
298      *
299      * @param resource the resource
300      * @return the interval, 0 won't trigger the heartbeat
301      */
302     protected int extractHeartbeatInterval(final AtmosphereResourceImpl resource) {
303         // Extract the desired heartbeat interval
304         // Won't be applied if lower config value
305         int interval = heartbeatFrequencyInSeconds;
306         final String s = resource.getRequest(false).getHeader(HeaderConfig.X_HEARTBEAT_SERVER);
307 
308         if (s != null) {
309             try {
310                 interval = Integer.parseInt(s);
311 
312                 if (interval != 0 && interval < heartbeatFrequencyInSeconds) {
313                     interval = heartbeatFrequencyInSeconds;
314                 }
315             } catch (NumberFormatException nfe) {
316                 logger.warn("{} header is not an integer", HeaderConfig.X_HEARTBEAT_SERVER, nfe);
317             }
318         }
319 
320         return interval;
321     }
322 
323     void cancelF(AtmosphereRequest request) {
324         try {
325             Future<?> f = (Future<?>) request.getAttribute(HEARTBEAT_FUTURE);
326             if (f != null) f.cancel(false);
327             request.removeAttribute(HEARTBEAT_FUTURE);
328         } catch (Exception ex) {
329             // https://github.com/Atmosphere/atmosphere/issues/1503
330             logger.trace("", ex);
331         }
332     }

334     /**
335      * <p>
336      * Configures the heartbeat sent by the server in an interval in seconds specified in parameter for the given
337      * resource.
338      * </p>
339      *
340      * @param interval the interval in seconds
341      * @param r        the resource
342      * @param request  the request response
343      * @param response the resource response
344      * @return this
345      */
346     public HeartbeatInterceptor clock(final int interval,
347                                       final AtmosphereResource r,
348                                       final AtmosphereRequest request,
349                                       final AtmosphereResponse response) {
350 
351         try {
352             request.setAttribute(HEARTBEAT_FUTURE, heartBeat.schedule(new Callable<Object>() {
353                 @Override
354                 public Object call() throws Exception {
355                     synchronized (r) {
356                         if (AtmosphereResourceImpl.class.cast(r).isInScope() && r.isSuspended()) {
357                             try {
358                                 logger.trace("Heartbeat for Resource {}", r);
359                                 response.write(paddingBytes, false);
360                                 if (Utils.resumableTransport(r.transport()) && resumeOnHeartbeat) {
361                                     r.resume();
362                                 } else {
363                                     response.flushBuffer();
364                                 }
365                             } catch (Throwable t) {
366                                 logger.trace("{}", r.uuid(), t);
367                                 cancelF(request);
368                             }
369                         } else {
370                             cancelF(request);
371                         }
372                     }
373                     return null;
374                 }
375             }, interval, TimeUnit.SECONDS));
376         } catch (Throwable t) {
377             logger.warn("", t);
378         }
379 
380         return this;
381     }

383     @Override
384     public String toString() {
385         return "Heartbeat Interceptor Support";
386     }

388     @Override
389     public void destroy() {
390         if (destroyed.getAndSet(true)) return;
391 
392         for (AtmosphereResource r : config.resourcesFactory().findAll()) {
393             cancelF(r.getRequest());
394         }
