50  /**
51   * <p>
52   * An interceptor that send whitespace every in 60 seconds by default. Another value could be specified with the
53   * {@link org.atmosphere.cpr.ApplicationConfig#HEARTBEAT_INTERVAL_IN_SECONDS} in the atmosphere config. The heartbeat will be scheduled as soon as the
54   * request is suspended.
55   * </p>
56   * <p/>
57   * <p>
58   * Moreover, any client can ask for a particular value with the {@link HeaderConfig#X_HEARTBEAT_SERVER} header set in
59   * request. This value will be taken in consideration if it is greater than the configured value. Client can also
60   * specify the value "0" to disable heartbeat.
61   * </p>
62   * <p/>
63   * <p>
64   * Finally the server notifies thanks to the {@link JavaScriptProtocol} the desired heartbeat interval that the client
65   * should applies. This interceptor just manage the configured value and the {@link JavaScriptProtocol protocol} sends
66   * the value to the client.
67   * </p>
68   *
69   * @author Jeanfrancois Arcand
70   */
71  public class HeartbeatInterceptor extends AtmosphereInterceptorAdapter {

137     @Override
138     public void configure(final AtmosphereConfig config) {
139         // Server
140         String s = config.getInitParameter(HEARTBEAT_INTERVAL_IN_SECONDS);
141         if (s != null) {
142             heartbeatFrequencyInSeconds = Integer.valueOf(s);
143         }
144 
145         // Client
146         s = config.getInitParameter(CLIENT_HEARTBEAT_INTERVAL_IN_SECONDS);
147         if (s != null) {
148             clientHeartbeatFrequencyInSeconds = Integer.valueOf(s);
149         }
150 
151         heartBeat = ExecutorsFactory.getScheduler(config);
152 
153         resumeOnHeartbeat = config.getInitParameter(RESUME_ON_HEARTBEAT, true);
154     }

156     private static class Clock extends AtmosphereResourceEventListenerAdapter implements AllowInterceptor {
157         public Clock() {
158         }
159     }

161     @Override
162     public Action inspect(final AtmosphereResource r) {
163         final AtmosphereRequest request = r.getRequest();
164         final AtmosphereResponse response = r.getResponse();
165 
166         // Check heartbeat
167         if (clientHeartbeatFrequencyInSeconds > 0) {
168             byte[] body = IOUtils.readEntirelyAsByte(r);
169 
170             if (Arrays.equals(paddingBytes, body)) {
171                 // Dispatch an event to notify that a heartbeat has been intercepted
172                 // TODO: see https://github.com/Atmosphere/atmosphere/issues/1561
173                 final AtmosphereResourceEvent event = new HeartbeatAtmosphereResourceEvent(AtmosphereResourceImpl.class.cast(r));
174 
175                 if (AtmosphereResourceHeartbeatEventListener.class.isAssignableFrom(r.getAtmosphereHandler().getClass())) {
176                     r.addEventListener(new AtmosphereResourceEventListenerAdapter.OnHeartbeat() {
177                         @Override
178                         public void onHeartbeat(AtmosphereResourceEvent event) {
179                             AtmosphereResourceHeartbeatEventListener.class.cast(r.getAtmosphereHandler()).onHeartbeat(event);
180                         }
181                     });
182                 }
183 
184                 // Fire event
185                 r.notifyListeners(event);
186 
187                 return Action.CANCELLED;
188             }
189 
190             request.body(body);
191         }
192 
193         if (Utils.webSocketMessage(r)) return Action.CONTINUE;
194 
195         final int interval = extractHeartbeatInterval(r);
196 
197         if (interval != 0) {
198             if (!Utils.pollableTransport(r.transport())) {
199                 super.inspect(r);
200                 final boolean wasSuspended = r.isSuspended();
201 
202                 // Suspended? Ok can clock now
203                 // Otherwise, the listener will do the job
204                 if (wasSuspended) {
205                     clock(interval, r, request, response);
206                 }
207 
208                 r.addEventListener(new Clock() {
209                     @Override
210                     public void onSuspend(AtmosphereResourceEvent event) {
211 
212                         // We did not clocked when this listener was added because connection was not already suspended
213                         if (!wasSuspended) {
214                             clock(interval, r, request, response);
215                         }
216                     }
217 
218                     @Override
219                     public void onResume(AtmosphereResourceEvent event) {
220                         cancelF(request);
221                     }
222 
223                     @Override
224                     public void onDisconnect(AtmosphereResourceEvent event) {
225                         cancelF(request);
226                     }
227 
228                     @Override
229                     public void onClose(AtmosphereResourceEvent event) {
230                         cancelF(request);
231                     }
232                 });
233             } else {
234                 return Action.CONTINUE;
235             }
236 
237             final AsyncIOWriter writer = response.getAsyncIOWriter();
238 
239             if (!Utils.resumableTransport(r.transport())
240                     && AtmosphereInterceptorWriter.class.isAssignableFrom(writer.getClass())
241                     && r.getRequest().getAttribute(INTERCEPTOR_ADDED) == null) {
242                 AtmosphereInterceptorWriter.class.cast(writer).interceptor(new AsyncIOInterceptorAdapter() {
243 
244                     @Override
245                     public byte[] transformPayload(AtmosphereResponse response, byte[] responseDraft, byte[] data) throws IOException {
246                         cancelF(request);
247                         return responseDraft;
248                     }
249 
250                     @Override
251                     public void postPayload(final AtmosphereResponse response, byte[] data, int offset, int length) {
252                         logger.trace("Scheduling heartbeat for {}", r.uuid());
253                         clock(interval, r, request, response);
254                     }
255                 });
256                 r.getRequest().setAttribute(INTERCEPTOR_ADDED, Boolean.TRUE);
257             }
258 
259         }
260 
261         return Action.CONTINUE;
262     }

264     /**
265      * <p>
266      * Extracts the heartbeat interval as explained in class description. This method could be overridden to change the
267      * the configuration points.
268      * </p>
269      *
270      * @param resource the resource
271      * @return the interval, 0 won't trigger the heartbeat
272      */
273     protected int extractHeartbeatInterval(final AtmosphereResource resource) {
274         // Extract the desired heartbeat interval
275         // Won't be applied if lower config value
276         int interval = heartbeatFrequencyInSeconds;
277         final String s = resource.getRequest().getHeader(HeaderConfig.X_HEARTBEAT_SERVER);
278 
279         if (s != null) {
280             try {
281                 interval = Integer.parseInt(s);
282 
283                 if (interval != 0 && interval < heartbeatFrequencyInSeconds) {
284                     interval = heartbeatFrequencyInSeconds;
285                 }
286             } catch (NumberFormatException nfe) {
287                 logger.warn("{} header is not an integer", HeaderConfig.X_HEARTBEAT_SERVER, nfe);
288             }
289         }
290 
291         return interval;
292     }
293 
294     void cancelF(AtmosphereRequest request) {
295         try {
296             Future<?> f = (Future<?>) request.getAttribute(HEARTBEAT_FUTURE);
297             if (f != null) f.cancel(false);
298             request.removeAttribute(HEARTBEAT_FUTURE);
299         } catch (Exception ex) {
300             // https://github.com/Atmosphere/atmosphere/issues/1503
301             logger.trace("", ex);
302         }
303     }

305     /**
306      * <p>
307      * Configures the heartbeat sent by the server in an interval in seconds specified in parameter for the given
308      * resource.
309      * </p>
310      *
311      * @param interval the interval in seconds
312      * @param r        the resource
313      * @param request  the request response
314      * @param response the resource response
315      * @return this
316      */
317     public HeartbeatInterceptor clock(final int interval,
318                                       final AtmosphereResource r,
319                                       final AtmosphereRequest request,
320                                       final AtmosphereResponse response) {
321         request.setAttribute(HEARTBEAT_FUTURE, heartBeat.schedule(new Callable<Object>() {
322             @Override
323             public Object call() throws Exception {
324                 synchronized (r) {
325                     if (AtmosphereResourceImpl.class.cast(r).isInScope() && r.isSuspended()) {
326                         try {
327                             logger.trace("Heartbeat for Resource {}", r);
328                             response.write(paddingBytes, false);
329                             if (Utils.resumableTransport(r.transport()) && resumeOnHeartbeat) {
330                                 r.resume();
331                             } else {
332                                 response.flushBuffer();
333                             }
334                         } catch (Throwable t) {
335                             logger.trace("{}", r.uuid(), t);
336                             cancelF(request);
337                         }
338                     } else {
339                         cancelF(request);
340                     }
341                 }
342                 return null;
343             }
344         }, interval, TimeUnit.SECONDS));
345 
346         return this;
347     }
