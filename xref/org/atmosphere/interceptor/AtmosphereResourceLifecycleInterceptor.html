17  
18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.AtmosphereConfig;

21  import org.atmosphere.cpr.AtmosphereResource;
22  import org.atmosphere.cpr.AtmosphereResourceEvent;
23  import org.atmosphere.cpr.AtmosphereResourceImpl;
24  import org.atmosphere.util.Utils;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import java.io.IOException;
29  import java.util.concurrent.TimeUnit;
30  
31  import static org.atmosphere.cpr.ApplicationConfig.ATMOSPHERERESOURCE_INTERCEPTOR_METHOD;
32  import static org.atmosphere.cpr.ApplicationConfig.ATMOSPHERERESOURCE_INTERCEPTOR_TIMEOUT;
33  import static org.atmosphere.cpr.AtmosphereResource.TRANSPORT.POLLING;
34  import static org.atmosphere.cpr.AtmosphereResource.TRANSPORT.UNDEFINED;
35  import static org.atmosphere.cpr.AtmosphereResourceEventListenerAdapter.OnBroadcast;
36  
37  /**
38   * <p>This {@link AtmosphereInterceptor} implementation automatically suspends the intercepted
39   * {@link AtmosphereResource} and takes care of managing the response's state (flushing, resuming,
40   * etc.) when a {@link org.atmosphere.cpr.Broadcaster#broadcast} is invoked. When used, {@link org.atmosphere.cpr.AtmosphereHandler} implementations no longer need to make calls to
41   * {@link AtmosphereResource#suspend}.
42   * </p>
43   * If your application doesn't use {@link org.atmosphere.cpr.Broadcaster}, this interceptor will not work and you need to programmatically
44   * resume, flush, etc.
45   * <p/>
46   * <p>By default, intercepted {@link AtmosphereResource} instances are suspended when a GET
47   * request is received. You can change the triggering http method by configuring
48   * {@link org.atmosphere.cpr.ApplicationConfig#ATMOSPHERERESOURCE_INTERCEPTOR_METHOD}
49   * <p/>
50   * <p/>
51   * <p>Use this class when you don't want to manage the suspend/resume operation from your
52   * particular Atmosphere framework implementation classes ({@link org.atmosphere.cpr.AtmosphereHandler},
53   * {@link org.atmosphere.websocket.WebSocketHandler}, or
54   * {@link org.atmosphere.cpr.Meteor} instances) or extensions (GWT, Jersey, Wicket, etc...)
55   * </p>
56   * <strong>For this mechanism to work properly, each client must set the
57   * {@link org.atmosphere.cpr.HeaderConfig#X_ATMOSPHERE_TRANSPORT} header. Your AtmosphereHandler must also extends the
58   * {@link org.atmosphere.handler.AbstractReflectorAtmosphereHandler} or implements the logic defined inside
59   * {@link org.atmosphere.handler.AbstractReflectorAtmosphereHandler#postStateChange(org.atmosphere.cpr.AtmosphereResourceEvent)} </strong>
60   *
61   * @author Jeanfrancois Arcand
62   */
63  public class AtmosphereResourceLifecycleInterceptor implements AtmosphereInterceptor {
64  
65      private String method = "GET";
66      // For backward compat.
67      private Integer timeoutInSeconds = -1;
68      private static final Logger logger = LoggerFactory.getLogger(AtmosphereResourceLifecycleInterceptor.class);
69      private final boolean force;
70      private long timeoutInMilli = -1;
71  
72      public AtmosphereResourceLifecycleInterceptor(){
73          this(false);
74      }
75  
76      public AtmosphereResourceLifecycleInterceptor(boolean force){
77          this.force = force;
78      }
79  
80      @Override
81      public void configure(AtmosphereConfig config) {
82          String s = config.getInitParameter(ATMOSPHERERESOURCE_INTERCEPTOR_METHOD);
83          if (s != null) {
84              method = s;
85          }
86  
87          s = config.getInitParameter(ATMOSPHERERESOURCE_INTERCEPTOR_TIMEOUT);
88          if (s != null) {
89              timeoutInSeconds = Integer.valueOf(s);
90              timeoutInMilli = TimeUnit.MILLISECONDS.convert(timeoutInSeconds, TimeUnit.SECONDS);
91          }
92      }
93  
94      public String method() {
95          return method;
96      }
97  
98      public int timeoutInSeconds(){
99          return timeoutInSeconds;
100     }
101 
102     public AtmosphereResourceLifecycleInterceptor method(String method){
103         this.method = method;
104         return this;
105     }
106 
107     public AtmosphereResourceLifecycleInterceptor timeoutInSeconds(int timeoutInSeconds) {
108         this.timeoutInSeconds = timeoutInSeconds;
109         timeoutInMilli = TimeUnit.MILLISECONDS.convert(timeoutInSeconds, TimeUnit.SECONDS);
110         return this;
111     }
112 
113     /**
114      * Automatically suspend the {@link AtmosphereResource} based on {@link AtmosphereResource.TRANSPORT} value.
115      *
116      * @param r a {@link AtmosphereResource}
117      * @return
118      */
119     @Override
120     public Action inspect(AtmosphereResource r) {
121 
122         switch (r.transport()) {
123             case JSONP:
124             case AJAX:
125             case LONG_POLLING:
126                 r.resumeOnBroadcast(true);
127                 break;
128             default:
129                 break;
130         }
131         return Action.CONTINUE;
132     }
133 
134     @Override
135     public void postInspect(final AtmosphereResource r) {
136 
137         if (r.transport().equals(UNDEFINED) || Utils.webSocketMessage(r) || r.transport().equals(POLLING)) return;
138 
139         AtmosphereResourceImpl impl = AtmosphereResourceImpl.class.cast(r);
140         if ( (force || impl.getRequest(false).getMethod().equalsIgnoreCase(method))
141             && !impl.action().equals(Action.CANCELLED)
142             && impl.isInScope()) {
143 
144             logger.trace("Marking AtmosphereResource {} for suspend operation", r.uuid());
145             r.addEventListener(new OnBroadcast() {
146                 @Override
147                 public void onBroadcast(AtmosphereResourceEvent event) {
148                     switch (r.transport()) {
149                         case JSONP:
150                         case AJAX:
151                         case LONG_POLLING:
152                             break;
153                         default:
154                             try {
155                                 r.getResponse().flushBuffer();
156                             } catch (IOException e) {
157                                 logger.trace("", e);
158                             }
159                             break;
160                     }
161                 }
162             }).suspend(timeoutInMilli);
163         }
164     }
165 
166     public String toString() {
167         return "Atmosphere LifeCycle";
168     }
169 }
