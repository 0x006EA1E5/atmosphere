17  /**
18   * Shiro Interceptor, it creates a request attribute (subject) that contains the true Subject.
19   * For more information about why don't use directly SecurityUtils.getSubject
20   * http://jfarcand.wordpress.com/2011/07/13/quick-tip-using-apache-shiro-with-your-atmospheres-websocketcomet-app/
21   */
22  public class ShiroInterceptor extends AtmosphereInterceptorAdapter {
23  
24      private static final Logger logger = LoggerFactory.getLogger(ShiroInterceptor.class);
25  
26      @Override
27      public Action inspect(AtmosphereResource r) {
28          if (r.getRequest().attributes().containsKey(FrameworkConfig.SECURITY_SUBJECT) == false) {
29              try {
30                  Subject currentUser = null;
31                  if (r.transport().equals(TRANSPORT.WEBSOCKET)) {
32                      WebEnvironment env = WebUtils.getRequiredWebEnvironment(r.getAtmosphereConfig().getServletContext());
33                      currentUser = new WebSubject.Builder(env.getSecurityManager(), r.getRequest(), r.getResponse()).buildWebSubject();
34                  } else {
35                      currentUser = SecurityUtils.getSubject();
36                  }
37                  if (currentUser != null) {
38                      r.getRequest().setAttribute(FrameworkConfig.SECURITY_SUBJECT, currentUser);
39                  }
40              } catch (UnavailableSecurityManagerException ex) {
41                  logger.info("Shiro Web Security : {}", ex.getMessage());
42              } catch (java.lang.IllegalStateException ex) {
43                  logger.info("Shiro Web Environment : {}", ex.getMessage());
44              }
45          }
46  
47          return Action.CONTINUE;
48      }
49  }
