53  /**
54   * An Atmosphere interceptor to enable a simple rest binding protocol.
55   * This protocol is a simplified version of SwaggerSocket with some new features
56   * https://github.com/swagger-api/swaggersocket
57   *
58   * This interceptor currently handles both Websocket and SSE protocols.
59   * It was originally developed for enabling atmosphere for kafka-rest at
60   * https://github.com/elakito/kafka-rest-atmosphere
61   *
62   * @author elakito
63   */
64  public class SimpleRestInterceptor extends AtmosphereInterceptorAdapter {
65      private static final Logger LOG = LoggerFactory.getLogger(SimpleRestInterceptor.class);
66      /**
67       * Servlet init property to enable the detached mode in the response
68       */
69      public final static String PROTOCOL_DETACHED_KEY = "atmosphere.simple-rest.protocol.detached";
70  
71      /**
72       * Connection request property to enable the detached mode in the response
73       */
74      public final static String X_ATMOSPHERE_SIMPLE_REST_PROTOCOL_DETACHED = "X-Atmosphere-SimpleRestProtocolDetached";

76      protected final static String REQUEST_DISPATCHED = "request.dispatched";
77      protected final static String REQUEST_ID = "request.id";

79      private final static byte[] RESPONSE_TEMPLATE_HEAD = "{\"id\": \"".getBytes();
80      private final static byte[] RESPONSE_TEMPLATE_BELLY = "\", \"data\": ".getBytes();
81      private final static byte[] RESPONSE_TEMPLATE_BELLY_CONTINUE = "\", \"continue\":true, \"data\": ".getBytes();
82      private final static byte[] RESPONSE_TEMPLATE_BELLY_DETACHED = "\", \"detached\": true".getBytes();
83      private final static byte[] RESPONSE_TEMPLATE_BELLY_CONTINUE_DETACHED = "\", \"continue\":true, \"detached\": true".getBytes();
84      private final static byte[] RESPONSE_TEMPLATE_TAIL = "}".getBytes();
85      private final static byte[] RESPONSE_TEMPLATE_NEWLINE = "\n".getBytes();

88      private final static String HEARTBEAT_SCHEDULED = "heatbeat.scheduled";
89      private final static String HEARTBEAT_TEMPLATE = "{\"heartbeat\": \"%s\", \"time\": %d}";
90      private final static long DEFAULT_HEARTBEAT_INTERVAL = 60;

105     @Override
106     public void configure(AtmosphereConfig config) {
107         super.configure(config);
108         detached = Boolean.parseBoolean(config.getInitParameter(PROTOCOL_DETACHED_KEY));
109         //TODO make the heartbeat configurable
110         heartbeat = config.getBroadcasterFactory().lookup(DefaultBroadcaster.class, getHeartbeatBroadcasterName());
111         if (heartbeat == null) {
112             heartbeat = config.getBroadcasterFactory().get(DefaultBroadcaster.class, getHeartbeatBroadcasterName());
113         }

116     @Override
117     public Action inspect(final AtmosphereResource r) {
118         if (AtmosphereResource.TRANSPORT.WEBSOCKET != r.transport()
119                 && AtmosphereResource.TRANSPORT.SSE != r.transport()
120                 && AtmosphereResource.TRANSPORT.POLLING != r.transport()) {
121             LOG.debug("Skipping for non websocket request");
122             return Action.CONTINUE;
123         }
124         if (AtmosphereResource.TRANSPORT.POLLING == r.transport()) {
125             final String saruuid = (String)r.getRequest().getAttribute(ApplicationConfig.SUSPENDED_ATMOSPHERE_RESOURCE_UUID);
126             final AtmosphereResponse suspendedResponse = suspendedResponses.get(saruuid);
127             LOG.debug("Attaching a proxy writer to suspended response");
128             r.getResponse().asyncIOWriter(new AtmosphereInterceptorWriter() {
129                 @Override
130                 public AsyncIOWriter write(AtmosphereResponse r, String data) throws IOException {
131                     suspendedResponse.write(data);
132                     suspendedResponse.flushBuffer();
133                     return this;
134                 }
135 
136                 @Override
137                 public AsyncIOWriter write(AtmosphereResponse r, byte[] data) throws IOException {
138                     suspendedResponse.write(data);
139                     suspendedResponse.flushBuffer();
140                     return this;
141                 }
142 
143                 @Override
144                 public AsyncIOWriter write(AtmosphereResponse r, byte[] data, int offset, int length) throws IOException {
145                     suspendedResponse.write(data, offset, length);
146                     suspendedResponse.flushBuffer();
147                     return this;
148                 }
149 
150                 @Override
151                 public void close(AtmosphereResponse response) throws IOException {
152                 }
153             });
154             // REVISIT we need to keep this response's asyncwriter alive so that data can be written to the
155             //   suspended response, but investigate if there is a better alternative.
156             r.getResponse().destroyable(false);
157             return Action.CONTINUE;
158         }
159 
160         r.addEventListener(new AtmosphereResourceEventListenerAdapter() {
161             @Override
162             public void onSuspend(AtmosphereResourceEvent event) {
163                 final String srid = (String)event.getResource().getRequest().getAttribute(ApplicationConfig.SUSPENDED_ATMOSPHERE_RESOURCE_UUID);
164                 LOG.debug("Registrering suspended resource: {}", srid);
165                 suspendedResponses.put(srid, event.getResource().getResponse());
166 
167                 AsyncIOWriter writer = event.getResource().getResponse().getAsyncIOWriter();
168                 if (writer == null) {
169                     writer = new AtmosphereInterceptorWriter();
170                     r.getResponse().asyncIOWriter(writer);
171                 }
172                 if (writer instanceof AtmosphereInterceptorWriter) {
173                     ((AtmosphereInterceptorWriter)writer).interceptor(interceptor);
174                 }
175             }
176 
177             @Override
178             public void onDisconnect(AtmosphereResourceEvent event) {
179                 super.onDisconnect(event);
180                 final String srid = (String)event.getResource().getRequest().getAttribute(ApplicationConfig.SUSPENDED_ATMOSPHERE_RESOURCE_UUID);
181                 LOG.debug("Unregistrering suspended resource: {}", srid);
182                 suspendedResponses.remove(srid);
183             }
184 
185         });
186 
187         AtmosphereRequest request = r.getRequest();
188         if (request.getAttribute(REQUEST_DISPATCHED) == null) {
189             try {
190                 //REVISIT use a more efficient approach for the detached mode (i.e.,avoid reading the message into a string)
191                 // read the message entity and dispatch a service call
192                 String body = IOUtils.readEntirelyAsString(r).toString();
193                 LOG.debug("Request message: '{}'", body);
194                 if (body.length() == 0) {
195                     //TODO we might want to move this heartbeat scheduling after the handshake phase (if that is added)
196                     if ((AtmosphereResource.TRANSPORT.WEBSOCKET == r.transport() ||
197                             AtmosphereResource.TRANSPORT.SSE == r.transport())
198                             && request.getAttribute(HEARTBEAT_SCHEDULED) == null) {
199                         r.suspend();
200                         scheduleHeartbeat(r);
201                         request.setAttribute(HEARTBEAT_SCHEDULED, "true");
202                         return Action.SUSPEND;
203                     }
204                     return Action.CANCELLED;
205                 }
206 
207                 AtmosphereRequest ar = createAtmosphereRequest(request, body);
208                 if (ar == null) {
209                     return Action.CANCELLED;
210                 }
211                 AtmosphereResponse response = r.getResponse();
212                 ar.localAttributes().put(REQUEST_DISPATCHED, "true");
213 
214                 request.removeAttribute(FrameworkConfig.INJECTED_ATMOSPHERE_RESOURCE);
215                 response.request(ar);
216 
217                 attachWriter(r);
218 
219                 Action action = r.getAtmosphereConfig().framework().doCometSupport(ar, response);
220                 if (action.type() == Action.TYPE.SUSPEND) {
221                     ar.destroyable(false);
222                     response.destroyable(false);
223                 }
224                 return Action.CANCELLED;
225             } catch (IOException | ServletException e) {
226                 LOG.error("Failed to process", e);
227             }
228         }
229 
230         return Action.CONTINUE;
231     }

245     protected AtmosphereRequest createAtmosphereRequest(AtmosphereRequest request, String body) throws IOException {
246         //REVISIT find a more efficient way to read and extract the message data
247         JSONEnvelopeReader jer = new JSONEnvelopeReader(new StringReader(body));
248 
249         final String id = jer.getHeader("id");
250         if (id != null) {
251             request.localAttributes().put(REQUEST_ID, id);
252         }
253 
254         boolean skip = false;
255         final boolean continued = Boolean.valueOf(jer.getHeader("continue"));
256         Reader reader = readerPool.getReader(id, false);
257         if (reader != null) {
258             skip = true;
259         } else if (continued) {
260             reader = readerPool.getReader(id, true);
261         }
262 
263         if (skip) {
264             // add the request data to the prevously dispatched request and skip dispatching a new one
265             final Reader data = jer.getReader();
266             if (data != null) {
267                 readerPool.addChunk(id, data, continued);
268             }
269             return null;
270         } else {
271             // prepare a new request for dispatching
272             final String method = jer.getHeader("method");
273             String path = jer.getHeader("path");
274             final String type = jer.getHeader("type");
275             final String accept = jer.getHeader("accept");
276 
277             AtmosphereRequest.Builder b = new AtmosphereRequestImpl.Builder();
278             b.method(method != null ? method : "GET").pathInfo(path != null ? path: "/");
279             if (accept != null || type != null) {
280                 Map<String, String> headers = new TreeMap<String, String>(String.CASE_INSENSITIVE_ORDER);
281                 if (accept != null) {
282                     headers.put("Accept", accept);
283                 }
284                 if (type != null) {
285                     b.contentType(type);
286                 }
287                 b.headers(headers);
288             }
289             final int qpos = path.indexOf('?');
290             if (qpos > 0) {
291                 b.queryString(path.substring(qpos + 1));
292                 path = path.substring(0, qpos);
293             }
294             final Reader data = jer.getReader();
295             if (data != null) {
296                 if (reader != null) {
297                     b.reader(reader);
298                     readerPool.addChunk(id, data, true);
299                 } else {
300                     b.reader(data);
301                 }
302             }
303             String requestURL = request.getRequestURL() + path.substring(request.getRequestURI().length());
304             b.requestURI(path).requestURL(requestURL).request(request);
305 
306             return b.build();
307         }
308     }

310     protected byte[] createResponse(AtmosphereResponse response, byte[] payload) {
311         if (LOG.isDebugEnabled()) {
312             LOG.debug("createResponse for payload {}", new String(payload));
313         }
314         AtmosphereRequest request = response.request();
315         String id = (String)request.getAttribute(REQUEST_ID);
316         if (id == null) {
317             // control response such as heartbeat or plain responses
318             return payload;
319         }
320         //TODO find a nicer way to build the response entity
321         ByteArrayOutputStream baos = new ByteArrayOutputStream();
322         if (id != null) {
323             try {
324                 baos.write(RESPONSE_TEMPLATE_HEAD);
325                 baos.write(id.getBytes());
326                 if (isDetached(request)) {
327                     // {"id":...,  "detached": true}\n
328                     // <payload>
329                     if (isLastResponse(request, response)) {
330                         baos.write(RESPONSE_TEMPLATE_BELLY_DETACHED);
331                     } else {
332                         baos.write(RESPONSE_TEMPLATE_BELLY_CONTINUE_DETACHED);
333                     }
334                     baos.write(RESPONSE_TEMPLATE_TAIL);
335                     baos.write(RESPONSE_TEMPLATE_NEWLINE);
336                     baos.write(payload);
337                 } else {
338                     // {"id":..., "data": <payload>}
339                     boolean isobj = isJSONObject(payload);
340                     if (isLastResponse(request, response)) {
341                         baos.write(RESPONSE_TEMPLATE_BELLY);
342                     } else {
343                         baos.write(RESPONSE_TEMPLATE_BELLY_CONTINUE);
344                     }
345                     if (!isobj) {
346                         baos.write(quote(payload));
347                     } else {
348                         baos.write(payload);
349                     }
350                     baos.write(RESPONSE_TEMPLATE_TAIL);
351                 }
352 
353             } catch (IOException e) {
354                 //ignore as it can't happen
355             }
356         }
357         return baos.toByteArray();
358     }

360     private void scheduleHeartbeat(AtmosphereResource r) {
361         //REVISIT make the schedule configurable
362         heartbeat.addAtmosphereResource(r);
363         if (!heartbeatScheduled) {
364             heartbeat.scheduleFixedBroadcast(String.format(getHeartbeatTemplate(), getHeartbeatTemplateArguments()),
365                     DEFAULT_HEARTBEAT_INTERVAL, DEFAULT_HEARTBEAT_INTERVAL, TimeUnit.SECONDS);
366             heartbeatScheduled = true;
367         }
368     }

370     protected static boolean isLastResponse(AtmosphereRequest request, AtmosphereResponse response) {
371         return (response instanceof CompletionAware && ((CompletionAware)response).completed())
372                 || Boolean.TRUE != request.getAttribute(ApplicationConfig.RESPONSE_COMPLETION_AWARE);
373     }

375     protected boolean isDetached(AtmosphereRequest request) {
376         // the default detached setting configured by the init property can be overrriden by the connection property
377         final String prop = request.getHeader(X_ATMOSPHERE_SIMPLE_REST_PROTOCOL_DETACHED);
378         return (detached && prop == null) || Boolean.valueOf(prop);
379     }

381     private void attachWriter(final AtmosphereResource r) {
382         AtmosphereResponse res = r.getResponse();
383         AsyncIOWriter writer = res.getAsyncIOWriter();
384 
385         if (writer instanceof AtmosphereInterceptorWriter) {
386             ((AtmosphereInterceptorWriter)writer).interceptor(interceptor, 0);
387         }
388     }

390     private final class Interceptor extends AsyncIOInterceptorAdapter {

392         public byte[] transformPayload(AtmosphereResponse response, byte[] responseDraft, byte[] data) throws IOException {

397     protected static boolean isJSONObject(byte[] b) {
398         return b.length > 0 && (b[0] == (byte)'[' || b[0] == (byte)'{');
399     }
400     protected static byte[] quote(byte[] b) {
401         ByteArrayOutputStream baos = new ByteArrayOutputStream();
402         baos.write('"');
403         for (byte c : b) {
404             if (c == '"') {
405                 baos.write('\\');
406             }
407             baos.write(c);
408         }
409         baos.write('"');
410         return baos.toByteArray();
411     }

413     /*
414      * A custom json envelope reader to parse a character sequence by extracting the key value pairs
415      * but leaves the data value unparsed so that it can be subsequently consumed directly as a chracter seqeunce.
416      */
417     static class JSONEnvelopeReader {
418         private Reader reader;
419         private Map<String, String> headers;
420         private boolean datap;
421         private boolean detachedp;
422         private int peek = -1;
423 
424         public JSONEnvelopeReader(Reader reader) throws IOException {
425             this.reader = reader;
426             this.headers = new HashMap<String, String>();
427 
428             prepare();
429         }
430 
431         public String getHeader(String name) {
432             return headers.get(name);
433         }
434 
435         public Map<String, String> getHeaders() {
436             return headers;
437         }
438 
439         public Reader getReader() {
440             if (!datap && !detachedp) {
441                 return null;
442             }
443 
444             return new Reader() {
445                 private int b;
446                 @Override
447                 public int read(char[] cbuf, int off, int len) throws IOException {
448                     int n = reader.read(cbuf, off, len);
449                     if (n > 0) {
450                         boolean escaping = false;
451                         char quot = 0;
452                         for (int i = off; i < n; i++) {
453                             char c = cbuf[i];
454                             if (c == '{' && !escaping) {
455                                 b++;
456                             } else if (c == '}' && !escaping) {
457                                 b--;
458                                 if (b < 0) {
459                                     // past the logical eof
460                                     n--;
461                                 }
462                             } else if ((c == '"' || c == '\'') && !escaping) {
463                                 if (c == quot) {
464                                     quot = 0;
465                                 } else {
466                                     quot = c;
467                                 }
468                             } else if (c == '\\' && quot != 0 && !escaping) {
469                                 escaping = true;
470                             } else if (escaping) {
471                                 escaping = false;
472                             }
473                         }
474                     }
475                     return n;
476                 }
477 
478                 @Override
479                 public void close() throws IOException {
480                     reader.close();
481                 }
482 
483                 @Override
484                 public boolean ready() throws IOException {
485                     return reader.ready();
486                 }
487             };
488         }
489 
490 
491         private void prepare() throws IOException {
492             int c = next(true);
493             if (c == '{') {
494                 for (;;) {
495                     String name = nextName();
496                     c = next(true);
497                     if (c == ':') {
498                         if ("data".equals(name)) {
499                             datap = true;
500                             break;
501                         } else if ("detached".equals(name)) {
502                             if (Boolean.valueOf(nextValue())) {
503                                 detachedp = true;
504                             }
505                         } else {
506                             headers.put(name, nextValue());
507                         }
508                     } else {
509                         throw new IOException("invalid value: missing name-separator ':'");
510                     }
511                     c = next(true);
512                     if (c != ',') {
513                         if (c == '}' && detachedp) {
514                             while (c != -1) {
515                                 c = next(false);
516                                 if (c == '\n') {
517                                     break;
518                                 }
519                             }
520                         } else {
521                             unread(c);
522                         }
523                         break;
524                     }
525                 }
526             } else {
527                 throw new IOException("invalid object: missing being-object '{'");
528             }
529         }
530 
531         private String nextName() throws IOException {
532             int c = next(true);
533             if (c == '"' || c == '\'') {
534                 return nextQuoted(c);
535             }
536             throw new IOException("invalid name: missing quote '\"'");
537         }
538 
539         private String nextValue() throws IOException {
540             int c = next(true);
541             if (c == '"' || c == '\'') {
542                 // quoted string
543                 return nextQuoted(c);
544             } else if (c == 't' || c == 'f' || ('0' <= c && c <= '9')) {
545                 // true, false, or number
546                 unread(c);
547                 return nextNonQuoted();
548             }
549             throw new IOException("invalid value: unquoted non literals");
550         }
551 
552         private String nextQuoted(int quot) throws IOException {
553             StringBuilder sb = new StringBuilder();
554             boolean escaping = false;
555             int c;
556             while ((c = next(false)) != -1) {
557                 if (c == '\\' && !escaping) {
558                     escaping = true;
559                 } else if (c == quot && !escaping) {
560                     break;
561                 } else {
562                     sb.append((char) c);
563                     if (escaping) {
564                         escaping = false;
565                     }
566                 }
567             }
568             if (c != -1) {
569                 return sb.toString();
570             }
571             throw new IOException("invalid quoted string: missing quotation");
572         }
573 
574         private String nextNonQuoted() throws IOException {
575             StringBuilder sb = new StringBuilder();
576             int c;
577             while ((c = next(false)) != -1) {
578                 if (c == '}' || c == ',' || isWS(c)) {
579                     unread(c);
580                     break;
581                 } else {
582                     sb.append((char) c);
583                 }
584             }
585             if (c != -1) {
586                 return sb.toString();
587             }
588             throw new IOException("invalid value: non-terminated");
589         }
590 
591         private int next(boolean skipws) throws IOException {
592             int c;
593             if (peek != -1) {
594                 c = peek;
595                 peek = -1;
596             } else {
597                 while ((c = reader.read()) != -1 && skipws && isWS(c));
598             }
599             return c;
600         }
601 
602         private void unread(int c) {
603             peek = c;
604         }
605 
606         private boolean isWS(int c) {
607             return c == 0x20 || c == 0x09 || c == 0x0a || c == 0x0d;
608         }
609     }
