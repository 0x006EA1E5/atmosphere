17  
18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.ApplicationConfig;
20  import org.atmosphere.cpr.AsyncIOInterceptorAdapter;
21  import org.atmosphere.cpr.AsyncIOWriter;
22  import org.atmosphere.cpr.AtmosphereConfig;
23  import org.atmosphere.cpr.AtmosphereInterceptorAdapter;
24  import org.atmosphere.cpr.AtmosphereInterceptorWriter;
25  import org.atmosphere.cpr.AtmosphereResource;
26  import org.atmosphere.cpr.AtmosphereResponse;
27  import org.atmosphere.util.Utils;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import javax.xml.bind.DatatypeConverter;
32  import java.io.IOException;
33  import java.util.Arrays;
34  import java.util.HashSet;
35  
36  import static org.atmosphere.cpr.ApplicationConfig.EXCLUDED_CONTENT_TYPES;
37  
38  /**
39   * An {@link org.atmosphere.cpr.AtmosphereInterceptor} that adds message size and delimiter, and encodes the message in Base64.
40   * This allows for broadcasting of messages containing the delimiter character.
41   * <p/>
42   * You can configure this class to exclude some response's content-type by using the {@link ApplicationConfig#EXCLUDED_CONTENT_TYPES}
43   *
44   * @author Jeanfrancois Arcand
45   * @author Martin Maƒçura
46   */
47  public class TrackMessageSizeB64Interceptor extends AtmosphereInterceptorAdapter {
48  
49      private static final Logger logger = LoggerFactory.getLogger(TrackMessageSizeB64Interceptor.class);
50      private static final String DELIMITER = "|";
51      private final static String OUT_ENCODING = "UTF-8";
52      public final static String SKIP_INTERCEPTOR = TrackMessageSizeB64Interceptor.class.getName() + ".skip";
53  
54      private final HashSet<String> excludedContentTypes = new HashSet<String>();
55  
56      private final Interceptor interceptor = new Interceptor();
57  
58      @Override
59      public void configure(AtmosphereConfig config) {
60          String s = config.getInitParameter(EXCLUDED_CONTENT_TYPES);
61          if (s != null) {
62              excludedContentTypes.addAll(Arrays.asList(s.split(",")));
63          }
64      }
65  
66      /**
67       * Excluse response's content-type from being processed by this class.
68       *
69       * @param excludedContentType the value of {@link org.atmosphere.cpr.AtmosphereResponse#getContentType()}
70       * @return this
71       */
72      public TrackMessageSizeB64Interceptor excludedContentType(String excludedContentType) {
73          excludedContentTypes.add(excludedContentType.toLowerCase());
74          return this;
75      }
76  
77      @Override
78      public Action inspect(final AtmosphereResource r) {
79  
80          if (Utils.webSocketMessage(r)) return Action.CONTINUE;
81  
82          final AtmosphereResponse response = r.getResponse();
83  
84          super.inspect(r);
85  
86          AsyncIOWriter writer = response.getAsyncIOWriter();
87          if (AtmosphereInterceptorWriter.class.isAssignableFrom(writer.getClass())) {
88              AtmosphereInterceptorWriter.class.cast(writer).interceptor(interceptor);
89          } else {
90              logger.warn("Unable to apply {}. Your AsyncIOWriter must implement {}", getClass().getName(), AtmosphereInterceptorWriter.class.getName());
91          }
92          return Action.CONTINUE;
93      }
94  
95      @Override
96      public String toString() {
97          return " Track Message Size Base64 Interceptor using " + DELIMITER;
98      }
99  
100     private final class Interceptor extends AsyncIOInterceptorAdapter {
101         @Override
102         public byte[] transformPayload(AtmosphereResponse response, byte[] responseDraft, byte[] data) throws IOException {
103 
104             if (response.request().getAttribute(SKIP_INTERCEPTOR) == null
105                     && (response.getContentType() == null
106                     || !excludedContentTypes.contains(response.getContentType().toLowerCase()))) {
107                 response.setCharacterEncoding(OUT_ENCODING);
108                 String s = DatatypeConverter.printBase64Binary(responseDraft);
109                 StringBuilder sb = new StringBuilder();
110                 sb.append(s.length()).append(DELIMITER).append(s);
111                 return sb.toString().getBytes(OUT_ENCODING);
112             } else {
113                 return responseDraft;
114             }
115 
116         }
117     }
118 }
