16  package org.atmosphere.inject;
17  
18  import org.atmosphere.cpr.AtmosphereConfig;
19  import org.atmosphere.cpr.AtmosphereFramework;
20  import org.atmosphere.cpr.AtmosphereObjectFactory;
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import javax.annotation.PostConstruct;
25  import javax.inject.Inject;
26  import java.lang.reflect.Field;
27  import java.lang.reflect.InvocationTargetException;
28  import java.lang.reflect.Method;
29  import java.util.ArrayList;
30  import java.util.List;
31  
32  /**
33   * Support injection of Atmosphere's Internal object using {@link }
34   * {@link org.atmosphere.cpr.AtmosphereConfig},{@link AtmosphereFramework,{@link AtmosphereFramework,{@link org.atmosphere.cpr.BroadcasterFactory,
35   * {@link org.atmosphere.cpr.AtmosphereResourceFactory } ,{@link org.atmosphere.cpr.DefaultMetaBroadcaster } and
36   * {@link org.atmosphere.cpr.AtmosphereResourceSessionFactory }
37   *
38   * @author Jeanfrancois Arcand
39   */
40  public class InjectableObjectFactory implements AtmosphereObjectFactory<Injectable<?>> {
41  
42      protected static final Logger logger = LoggerFactory.getLogger(AtmosphereFramework.class);
43  
44      public final static Class<Injectable<?>>[] DEFAULT_ATMOSPHERE_INJECTABLE = new Class[]{
45              AtmosphereConfigInjectable.class,
46              AtmosphereFrameworkInjectable.class,
47              AtmosphereResourceFactoryInjectable.class,
48              AtmosphereResourceSessionFactoryInjectable.class,
49              BroadcasterFactoryInjectable.class,
50              MetaBroadcasterInjectable.class
51      };
52  
53      private final List<Injectable<?>> injectables = new ArrayList<Injectable<?>>();
54      private AtmosphereConfig config;
55  
56      @Override
57      public void configure(AtmosphereConfig config) {
58          this.config = config;
59          for (Class<Injectable<?>> i : DEFAULT_ATMOSPHERE_INJECTABLE) {
60              try {
61                  injectables.add(i.newInstance());
62              } catch (Exception e) {
63                  logger.error("", e);
64              }
65          }
66      }
67  
68      @Override
69      public <T, U extends T> U newClassInstance(Class<T> classType,
70                                                 Class<U> defaultType) throws InstantiationException, IllegalAccessException {
71  
72          U instance = defaultType.newInstance();
73  
74          injectAtmosphereInternalObject(instance, defaultType, config.framework());
75          postConstructExecution(instance, defaultType);
76  
77          return instance;
78      }
79  
80      /**
81       * Execute {@PostConstruct} method.
82       *
83       * @param instance    the requested object.
84       * @param defaultType the type of the requested object
85       * @param <U>
86       * @throws IllegalAccessException
87       */
88      public <U> void postConstructExecution(U instance, Class<U> defaultType) throws IllegalAccessException {
89          injectMethods(defaultType.getDeclaredMethods(), instance);
90          injectMethods(defaultType.getMethods(), instance);
91  
92      }
93  
94      private <U> void injectMethods(Method[] methods, U instance) throws IllegalAccessException {
95          for (Method m : methods) {
96              if (m.isAnnotationPresent(PostConstruct.class)) {
97                  try {
98                      m.setAccessible(true);
99                      m.invoke(instance);
100                     break;
101                 } catch (InvocationTargetException e) {
102                     throw new RuntimeException(e);
103                 }
104             }
105         }
106     }
107 
108     /**
109      * @param instance    the requested object.
110      * @param defaultType the type of the requested object
111      * @param framework   the {@link org.atmosphere.cpr.AtmosphereFramework}
112      * @param <U>
113      * @throws IllegalAccessException
114      */
115     public <U> void injectAtmosphereInternalObject(U instance, Class<U> defaultType, AtmosphereFramework framework) throws IllegalAccessException {
116         injectFields(defaultType.getDeclaredFields(), instance, framework);
117         injectFields(defaultType.getFields(), instance, framework);
118     }
119 
120     private <U>  void injectFields(Field[] fields, U instance, AtmosphereFramework framework) throws IllegalAccessException {
121         for (Field field : fields) {
122             if (field.isAnnotationPresent(Inject.class)) {
123                 for (Injectable c : injectables) {
124                     if (c.supportedType(field.getType())) {
125                         field.setAccessible(true);
126                         field.set(instance, c.injectable(framework.getAtmosphereConfig()));
127                         break;
128                     }
129                 }
130             }
131         }
132     }
133 
134     public AtmosphereObjectFactory allowInjectionOf(Injectable<?> injectable) {
135         injectables.add(injectable);
136         return this;
137     }
138 
139     @Override
140     public String toString() {
141         return InjectableObjectFactory.class.getName();
142     }
143 
144 }
