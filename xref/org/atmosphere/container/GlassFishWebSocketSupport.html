60  /**
61   * Websocket Portable Runtime implementation on top of GlassFish 3.0.1 and up.
62   *
63   * @author Jeanfrancois Arcand
64   */
65  public class GlassFishWebSocketSupport extends GrizzlyCometSupport {
66  
67      private static final Logger logger = LoggerFactory.getLogger(GlassFishWebSocketSupport.class);
68  
69      public GlassFishWebSocketSupport(AtmosphereConfig config) {
70          super(config);
71      }
72  
73      @Override
74      public void init(ServletConfig sc) throws ServletException {
75          super.init(sc);
76          WebSocketEngine.getEngine().register(sc.getServletContext().getContextPath(), new GrizzlyApplication());
77      }
78  
79      /**
80       * {@inheritDoc}
81       */
82      @Override
83      public Action service(HttpServletRequest request, HttpServletResponse response)
84              throws IOException, ServletException {
85  
86          boolean webSocketEnabled = false;
87          if (request.getHeaders("Connection") != null && request.getHeaders("Connection").hasMoreElements()) {
88              String[] e = request.getHeaders("Connection").nextElement().split(",");
89              for (String upgrade : e) {
90                  if (upgrade.equalsIgnoreCase("Upgrade")) {
91                      webSocketEnabled = true;
92                      break;
93                  }
94              }
95          }
96          if (!webSocketEnabled) {
97              return super.service(request, response);
98          } else {
99              Action action = suspended(request, response);
100             if (action.type == Action.TYPE.SUSPEND) {
101                 logger.debug("Suspending response: {}", response);
102             } else if (action.type == Action.TYPE.RESUME) {
103                 logger.debug("Resuming response: {}", response);
104             }
105             return action;
106         }
107     }
108 
109     /**
110      * Return the container's name.
111      */
112     public String getContainerName() {
113         return config.getServletConfig().getServletContext().getServerInfo() + " with WebSocket enabled.";
114     }

116     /**
117      * Grizzly support for WebSocket.
118      */
119     public class GrizzlyApplication extends WebSocketApplication {
120 
121         private WebSocketProcessor webSocketProcessor;
122 
123         public void onConnect(com.sun.grizzly.websockets.WebSocket w) {
124 
125             if (!BaseServerWebSocket.class.isAssignableFrom(w.getClass())) {
126                 throw new IllegalStateException();
127             }
128 
129             BaseServerWebSocket webSocket = BaseServerWebSocket.class.cast(w);
130             try {
131 
132                 webSocketProcessor = (WebSocketProcessor) GrizzlyWebSocket.class.getClassLoader()
133                         .loadClass(config.getServlet().getWebSocketProcessorClassName())
134                         .getDeclaredConstructor(new Class[]{AtmosphereServlet.class, WebSocket.class})
135                         .newInstance(new Object[]{config.getServlet(), new GrizzlyWebSocket(webSocket)});
136 
137                 webSocketProcessor.connect(new HttpServletRequestWrapper(webSocket.getRequest()));
138             } catch (Exception e) {
139                 logger.warn("failed to connect to web socket", e);
140             }
141         }
142 
143         @Override
144         public boolean isApplicationRequest(Request request) {
145             return true;
146         }
147 
148         public void onMessage(com.sun.grizzly.websockets.WebSocket webSocket, DataFrame dataFrame) {
149             webSocketProcessor.broadcast(dataFrame.getTextPayload());
150         }
151 
152         public void onClose(com.sun.grizzly.websockets.WebSocket webSocket) {
153             webSocketProcessor.close();
154         }
155 
156     }
157 
158     @Override
159     public boolean supportWebSocket() {
160         return true;
161     }

175         }
