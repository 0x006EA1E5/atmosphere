41  public class JSR356Endpoint extends Endpoint {
42  
43      private static final Logger logger = LoggerFactory.getLogger(JSR356Endpoint.class);
44  
45      private final WebSocketProcessor webSocketProcessor;
46      private final Integer maxBinaryBufferSize;
47      private final Integer maxTextBufferSize;
48      private AtmosphereRequest request;
49      private final AtmosphereFramework framework;

53      public JSR356Endpoint(AtmosphereFramework framework, WebSocketProcessor webSocketProcessor) {
54          this.framework = framework;
55          this.webSocketProcessor = webSocketProcessor;
56  
57          if (framework.isUseNativeImplementation()) {
58              throw new IllegalStateException("You cannot use WebSocket native implementation with JSR356. Please set " + ApplicationConfig.PROPERTY_NATIVE_COMETSUPPORT + " to false");
59          }
60  
61          String s = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.WEBSOCKET_IDLETIME);
62          if (s != null) {
63              webSocketWriteTimeout = Integer.valueOf(s);
64          } else {
65              webSocketWriteTimeout = -1;
66          }
67  
68          s = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.WEBSOCKET_MAXBINARYSIZE);
69          if (s != null) {
70              maxBinaryBufferSize = Integer.valueOf(s);
71          } else {
72              maxBinaryBufferSize = -1;
73          }
74  
75          s = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.WEBSOCKET_MAXTEXTSIZE);
76          if (s != null) {
77              maxTextBufferSize = Integer.valueOf(s);
78          } else {
79              maxTextBufferSize = -1;
80          }
81      }

83      @Override
84      public void onOpen(Session session, EndpointConfig endpointConfig) {
85  
86          if (!webSocketProcessor.handshake(request)) {
87              try {
88                  session.close(new CloseReason(CloseReason.CloseCodes.CANNOT_ACCEPT, "Handshake not accepted."));
89              } catch (IOException e) {
90                  logger.trace("", e);
91              }
92              return;
93          }
94  
95          if (maxBinaryBufferSize != -1) session.setMaxBinaryMessageBufferSize(maxBinaryBufferSize);
96          if (webSocketWriteTimeout != -1) session.setMaxIdleTimeout(webSocketWriteTimeout);
97          if (maxTextBufferSize != -1) session.setMaxTextMessageBufferSize(maxTextBufferSize);
98  
99          webSocket = new JSR356WebSocket(session, framework.getAtmosphereConfig());
100 
101         // TODO: This is quite bogus!
102         Map<String, String> headers = new HashMap<String, String>();
103         headers.put("Sec-WebSocket-Version", "13");
104         headers.put("Connection", "Upgrade");
105         headers.put("Upgrade", "websocket");
106 
107         StringBuffer pathInfo = new StringBuffer("/");
108         for (Map.Entry<String, String> e : session.getPathParameters().entrySet()) {
109             pathInfo.append(e.getValue());
110         }
111 
112         try {
113 
114             request = new AtmosphereRequest.Builder()
115                     .requestURI(session.getRequestURI().toASCIIString())
116                     .requestURL(session.getRequestURI().toASCIIString())
117                     .headers(headers)
118                     .contextPath(framework.getServletContext().getContextPath())
119                     .pathInfo(pathInfo.toString())
120                     .build()
121                     .queryString(session.getQueryString());
122 
123 
124             // TODO: Fix this crazy code.
125             framework.addInitParameter(ALLOW_QUERYSTRING_AS_REQUEST, "false");
126 
127             webSocketProcessor.open(webSocket, request, AtmosphereResponse.newInstance(framework.getAtmosphereConfig(), request, webSocket));
128 
129             framework.addInitParameter(ALLOW_QUERYSTRING_AS_REQUEST, "true");
130 
131         } catch (Throwable e) {
132             try {
133                 session.close(new CloseReason(CloseReason.CloseCodes.UNEXPECTED_CONDITION, e.getMessage()));
134             } catch (IOException e1) {
135                 logger.trace("", e);
136             }
137             logger.error("", e);
138             return;
139         }
140 
141         session.addMessageHandler(new MessageHandler.Whole<String>() {
142             @Override
143             public void onMessage(String s) {
144                 webSocketProcessor.invokeWebSocketProtocol(webSocket, s);
145             }
146         });
147 
148         session.addMessageHandler(new MessageHandler.Whole<ByteBuffer>() {
149             @Override
150             public void onMessage(ByteBuffer bb) {
151                 byte[] b = new byte[bb.limit()];
152                 bb.get(b);
153                 webSocketProcessor.invokeWebSocketProtocol(webSocket, b, 0, b.length);
154             }
155         });
156     }

158     @Override
159     public void onClose(javax.websocket.Session session, javax.websocket.CloseReason closeCode) {
160         logger.trace("{} closed {}", session, closeCode);
161         request.destroy();
162         webSocketProcessor.close(webSocket, closeCode.getCloseCode().getCode());
163     }
164 
165     @Override
166     public void onError(javax.websocket.Session session, java.lang.Throwable t) {
167         logger.error("", t);
168         webSocketProcessor.notifyListener(webSocket,
169                 new WebSocketEventListener.WebSocketEvent<Throwable>(t, WebSocketEventListener.WebSocketEvent.TYPE.EXCEPTION, webSocket));
170     }
