54  public class JSR356Endpoint extends Endpoint {
55  
56      private static final Logger logger = LoggerFactory.getLogger(JSR356Endpoint.class);
57  
58      private final static String JAVAX_WEBSOCKET_ENDPOINT_LOCAL_ADDRESS = "javax.websocket.endpoint.localAddress";
59      private final static String JAVAX_WEBSOCKET_ENDPOINT_REMOTE_ADDRESS = "javax.websocket.endpoint.remoteAddress";
60  
61      private final WebSocketProcessor webSocketProcessor;
62      private final Integer maxBinaryBufferSize;
63      private final Integer maxTextBufferSize;
64      private AtmosphereRequest request;
65      private final AtmosphereFramework framework;
66      private WebSocket webSocket;
67      private final int webSocketWriteTimeout;
68      private HandshakeRequest handshakeRequest;
69  
70      public JSR356Endpoint(AtmosphereFramework framework, WebSocketProcessor webSocketProcessor) {
71          this.framework = framework;
72          this.webSocketProcessor = webSocketProcessor;
73  
74          if (framework.isUseNativeImplementation()) {
75              throw new IllegalStateException("You cannot use WebSocket native implementation with JSR356. Please set " + ApplicationConfig.PROPERTY_NATIVE_COMETSUPPORT + " to false");
76          }
77  
78          String s = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.WEBSOCKET_IDLETIME);
79          if (s != null) {
80              webSocketWriteTimeout = Integer.valueOf(s);
81          } else {
82              webSocketWriteTimeout = -1;
83          }
84  
85          s = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.WEBSOCKET_MAXBINARYSIZE);
86          if (s != null) {
87              maxBinaryBufferSize = Integer.valueOf(s);
88          } else {
89              maxBinaryBufferSize = -1;
90          }
91  
92          s = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.WEBSOCKET_MAXTEXTSIZE);
93          if (s != null) {
94              maxTextBufferSize = Integer.valueOf(s);
95          } else {
96              maxTextBufferSize = -1;
97          }
98      }

100     public JSR356Endpoint handshakeRequest(HandshakeRequest handshakeRequest) {
101         this.handshakeRequest = handshakeRequest;
102         return this;
103     }

105     @Override
106     public void onOpen(Session session, final EndpointConfig endpointConfig) {
107 
108         if (framework.isDestroyed()) return;
109 
110         if (!session.isOpen()) {
111             logger.trace("Session Closed {}", session);
112             return;
113         }
114 
115         if (maxBinaryBufferSize != -1) session.setMaxBinaryMessageBufferSize(maxBinaryBufferSize);
116         if (webSocketWriteTimeout != -1) session.setMaxIdleTimeout(webSocketWriteTimeout);
117         if (maxTextBufferSize != -1) session.setMaxTextMessageBufferSize(maxTextBufferSize);
118 
119         webSocket = new JSR356WebSocket(session, framework.getAtmosphereConfig());
120 
121         Map<String, String> headers = new HashMap<String, String>();
122         // TODO: We don't support multi map header, which cause => https://github.com/Atmosphere/atmosphere/issues/1945
123         for (Map.Entry<String, List<String>> e : handshakeRequest.getHeaders().entrySet()) {
124             headers.put(e.getKey(), !e.getValue().isEmpty() ? e.getValue().get(0) : "");
125         }
126 
127         // Force WebSocket. Hack for https://github.com/Atmosphere/atmosphere/issues/1944
128         headers.put("Connection", "Upgrade");
129 
130         String servletPath = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.JSR356_MAPPING_PATH);
131         if (servletPath == null) {
132             servletPath = IOUtils.guestServletPath(framework.getAtmosphereConfig());
133         }
134 
135         boolean recomputeForBackwardCompat = false;
136         URI uri = session.getRequestURI();
137         String rawPath = uri.getPath();
138         String contextPath = framework.getAtmosphereConfig().getServletContext().getContextPath();
139         int pathInfoStartAt = rawPath.indexOf(servletPath) + servletPath.length();
140 
141         String pathInfo = null;
142         if (rawPath.length() >= pathInfoStartAt) {
143             pathInfo = rawPath.substring(pathInfoStartAt);
144         } else {
145             recomputeForBackwardCompat = true;
146         }
147 
148         if (recomputeForBackwardCompat) {
149             // DON"T SCREAM this code is for broken/backward compatible
150             String[] paths = uri.getPath() != null ? uri.getPath().split("/") : new String[]{};
151 
152             int pathInfoStartIndex = 3;
153             if ("".equals(contextPath) || "".equals(servletPath)) {
154                 pathInfoStartIndex = 2;
155             }
156             ///contextPath / servletPath / pathInfo or / servletPath / pathInfo
157             StringBuffer b = new StringBuffer("/");
158             for (int i = 0; i < paths.length; i++) {
159                 if (i >= pathInfoStartIndex) {
160                     b.append(paths[i]).append("/");
161                 }
162             }
163 
164             if (b.length() > 1) {
165                 b.deleteCharAt(b.length() - 1);
166             }
167 
168             pathInfo = b.toString();
169         }
170 
171         if (pathInfo.equals("/")) {
172             pathInfo = null;
173         }
174 
175         try {
176             String requestURL = uri.toASCIIString();
177             if (requestURL.contains("?")) {
178                 requestURL = requestURL.substring(0, requestURL.indexOf("?"));
179             }
180 
181             // https://issues.apache.org/bugzilla/show_bug.cgi?id=56573
182             // https://java.net/jira/browse/WEBSOCKET_SPEC-228
183             if ((!requestURL.startsWith("http://")) || (!requestURL.startsWith("https://"))) {
184                 if (requestURL.startsWith("/")) {
185                     List<String> l = handshakeRequest.getHeaders().get("origin");
186                     if (l == null) {
187                         // https://issues.jboss.org/browse/UNDERTOW-252
188                         l = handshakeRequest.getHeaders().get("Origin");
189                     }
190                     String origin;
191                     if (l != null && !l.isEmpty()) {
192                         origin = l.get(0);
193                     } else {
194                         // Broken WebSocket Spec
195                         logger.trace("Unable to retrieve the `origin` header for websocket {}", session);
196                         origin = new StringBuilder("http").append(session.isSecure() ? "s" : "").append("://0.0.0.0:80").toString();
197                     }
198                     requestURL = new StringBuilder(origin).append(requestURL).toString();
199                 } else if (requestURL.startsWith("ws://")) {
200                     requestURL = requestURL.replace("ws://", "http://");
201                 } else if (requestURL.startsWith("wss://")) {
202                     requestURL = requestURL.replace("wss://", "https://");
203                 }
204             }
205 
206             List<String> cookieHeaders = handshakeRequest.getHeaders().get("Cookie");
207             Set<Cookie> cookies = null;
208             if (cookieHeaders != null) {
209                 cookies = new HashSet<Cookie>();
210                 for (String cookieHeader : cookieHeaders)
211                 cookies.addAll(CookieUtil.ServerCookieDecoder.STRICT.decode(cookieHeader));
212             }
213             
214             request = new AtmosphereRequestImpl.Builder()
215                     .requestURI(uri.getPath())
216                     .requestURL(requestURL)
217                     .headers(headers)
218                     .cookies(cookies)
219                     .session((HttpSession) handshakeRequest.getHttpSession())
220                     .servletPath(servletPath)
221                     .contextPath(framework.getServletContext().getContextPath())
222                     .pathInfo(pathInfo)
223                     .destroyable(false)
224                     .userPrincipal(session.getUserPrincipal())
225                     .remoteInetSocketAddress(new Callable<InetSocketAddress>() {
226                         @Override
227                         public InetSocketAddress call() throws Exception {
228                             return (InetSocketAddress) endpointConfig.getUserProperties().get(JAVAX_WEBSOCKET_ENDPOINT_REMOTE_ADDRESS);
229                         }
230                     })
231                     .localInetSocketAddress(new Callable<InetSocketAddress>() {
232                         @Override
233                         public InetSocketAddress call() throws Exception {
234                             return (InetSocketAddress) endpointConfig.getUserProperties().get(JAVAX_WEBSOCKET_ENDPOINT_LOCAL_ADDRESS);
235                         }
236                     })
237                     .build()
238                     .queryString(session.getQueryString());
239 
240             if (!webSocketProcessor.handshake(request)) {
241                 try {
242                     session.close(new CloseReason(CloseReason.CloseCodes.CANNOT_ACCEPT, "Handshake not accepted."));
243                 } catch (IOException e) {
244                     logger.trace("", e);
245                 }
246                 return;
247             }
248 
249             // TODO: Fix this crazy code.
250             framework.addInitParameter(ALLOW_QUERYSTRING_AS_REQUEST, "false");
251 
252             webSocketProcessor.open(webSocket, request, AtmosphereResponseImpl.newInstance(framework.getAtmosphereConfig(), request, webSocket));
253 
254             framework.addInitParameter(ALLOW_QUERYSTRING_AS_REQUEST, "true");
255 
256             if (session.isOpen()) {
257                 session.addMessageHandler(new MessageHandler.Whole<String>() {
258                     @Override
259                     public void onMessage(String s) {
260                         webSocketProcessor.invokeWebSocketProtocol(webSocket, s);
261                     }
262                 });
263 
264                 session.addMessageHandler(new MessageHandler.Whole<ByteBuffer>() {
265                     @Override
266                     public void onMessage(ByteBuffer bb) {
267                         byte[] b = bb.hasArray() ? bb.array() : new byte[bb.limit()];
268                         bb.get(b);
269                         webSocketProcessor.invokeWebSocketProtocol(webSocket, b, 0, b.length);
270                     }
271                 });
272             } else {
273                 logger.trace("Session closed during onOpen {}", session);
274                 onClose(session, new CloseReason(CloseReason.CloseCodes.GOING_AWAY, "Session closed already"));
275             }
276         } catch (Throwable e) {
277             if (session.isOpen()){
278                 logger.error("", e);
279             } else {
280                 logger.trace("Session closed during onOpen", e);
281             }
282 
283             try {
284                 session.close(new CloseReason(CloseReason.CloseCodes.UNEXPECTED_CONDITION, e.getMessage()));
285             } catch (IOException e1) {
286                 logger.trace("", e);
287             }
288             return;
289         }
290 
291     }
292 
293     @Override
294     public void onClose(javax.websocket.Session session, javax.websocket.CloseReason closeCode) {
295         logger.trace("{} closed {}", session, closeCode);
296         if (request != null) {
297             request.destroy();
298             webSocketProcessor.close(webSocket, closeCode.getCloseCode().getCode());
299         }
300     }
301 
302     @Override
303     public void onError(javax.websocket.Session session, java.lang.Throwable t) {
304         try {
305             logger.error("", t);
306             webSocketProcessor.notifyListener(webSocket,
307                     new WebSocketEventListener.WebSocketEvent<Throwable>(t, WebSocketEventListener.WebSocketEvent.TYPE.EXCEPTION, webSocket));
308         } catch (Exception ex) {
309             // Ignore completely
310             // https://github.com/Atmosphere/atmosphere/issues/1758
311         }
312     }
