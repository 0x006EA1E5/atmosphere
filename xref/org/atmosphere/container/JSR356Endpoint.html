44  import static org.atmosphere.cpr.ApplicationConfig.ALLOW_QUERYSTRING_AS_REQUEST;
45  
46  public class JSR356Endpoint extends Endpoint {
47  
48      private static final Logger logger = LoggerFactory.getLogger(JSR356Endpoint.class);
49  
50      private final WebSocketProcessor webSocketProcessor;
51      private final Integer maxBinaryBufferSize;
52      private final Integer maxTextBufferSize;
53      private AtmosphereRequest request;
54      private final AtmosphereFramework framework;
55      private WebSocket webSocket;
56      private final int webSocketWriteTimeout;
57      private HandshakeRequest handshakeRequest;
58  
59      public JSR356Endpoint(AtmosphereFramework framework, WebSocketProcessor webSocketProcessor) {
60          this.framework = framework;
61          this.webSocketProcessor = webSocketProcessor;
62  
63          if (framework.isUseNativeImplementation()) {
64              throw new IllegalStateException("You cannot use WebSocket native implementation with JSR356. Please set " + ApplicationConfig.PROPERTY_NATIVE_COMETSUPPORT + " to false");
65          }
66  
67          String s = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.WEBSOCKET_IDLETIME);
68          if (s != null) {
69              webSocketWriteTimeout = Integer.valueOf(s);
70          } else {
71              webSocketWriteTimeout = -1;
72          }
73  
74          s = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.WEBSOCKET_MAXBINARYSIZE);
75          if (s != null) {
76              maxBinaryBufferSize = Integer.valueOf(s);
77          } else {
78              maxBinaryBufferSize = -1;
79          }
80  
81          s = framework.getAtmosphereConfig().getInitParameter(ApplicationConfig.WEBSOCKET_MAXTEXTSIZE);
82          if (s != null) {
83              maxTextBufferSize = Integer.valueOf(s);
84          } else {
85              maxTextBufferSize = -1;
86          }
87      }
88  
89      public JSR356Endpoint handshakeRequest(HandshakeRequest handshakeRequest) {
90          this.handshakeRequest = handshakeRequest;
91          return this;
92      }
93  
94      @Override
95      public void onOpen(Session session, EndpointConfig endpointConfig) {
96  
97          if (framework.isDestroyed()) return;
98  
99          if (!webSocketProcessor.handshake(request)) {
100             try {
101                 session.close(new CloseReason(CloseReason.CloseCodes.CANNOT_ACCEPT, "Handshake not accepted."));
102             } catch (IOException e) {
103                 logger.trace("", e);
104             }
105             return;
106         }
107 
108         if (!session.isOpen()) {
109             logger.trace("Session Closed {}", session);
110             return;
111         }
112 
113         if (maxBinaryBufferSize != -1) session.setMaxBinaryMessageBufferSize(maxBinaryBufferSize);
114         if (webSocketWriteTimeout != -1) session.setMaxIdleTimeout(webSocketWriteTimeout);
115         if (maxTextBufferSize != -1) session.setMaxTextMessageBufferSize(maxTextBufferSize);
116 
117         webSocket = new JSR356WebSocket(session, framework.getAtmosphereConfig());
118 
119         Map<String, String> headers = new HashMap<String, String>();
120         for (Map.Entry<String, List<String>> e : handshakeRequest.getHeaders().entrySet()) {
121             headers.put(e.getKey(), e.getValue().size() > 0 ? e.getValue().get(0) : "");
122         }
123 
124         String servletPath = IOUtils.guestServletPath(framework.getAtmosphereConfig());
125 
126         URI uri = session.getRequestURI();
127         String[] paths = uri.getPath() != null ? uri.getPath().split("/") : new String[]{};
128 
129         int pathInfoStartIndex = 3;
130         String contextPath = framework.getAtmosphereConfig().getServletContext().getContextPath();
131         if ("".equals(contextPath) || "".equals(servletPath)) {
132             pathInfoStartIndex = 2;
133         }
134         
135         // /contextPath/servletPath/pathInfo or /servletPath/pathInfo
136         StringBuffer b = new StringBuffer("/");
137         for (int i = 0; i < paths.length; i++) {
138             if (i >= pathInfoStartIndex) {
139                 b.append(paths[i]).append("/");
140             }
141         }
142 
143         if (b.length() > 1) {
144             b.deleteCharAt(b.length() - 1);
145         }
146 
147         String pathInfo = b.toString();
148         if (pathInfo.equals("/")) {
149             pathInfo = null;
150         }
151 
152         try {
153             String requestURL = uri.toASCIIString();
154             if (requestURL.contains("?")) {
155                 requestURL = requestURL.substring(0, requestURL.indexOf("?"));
156             }
157 
158             // https://issues.apache.org/bugzilla/show_bug.cgi?id=56573
159             // https://java.net/jira/browse/WEBSOCKET_SPEC-228
160             if ((!requestURL.startsWith("http://")) || (!requestURL.startsWith("https://"))) {
161                 if (requestURL.startsWith("/")) {
162                     List<String> l = handshakeRequest.getHeaders().get("origin");
163                     if (l == null) {
164                         // https://issues.jboss.org/browse/UNDERTOW-252
165                         l = handshakeRequest.getHeaders().get("Origin");
166                     }
167                     String origin;
168                     if (l != null && l.size() > 0) {
169                         origin = l.get(0);
170                     } else {
171                         // Broken WebSocket Spec
172                         logger.trace("Unable to retrieve the `origin` header for websocket {}", session);
173                         origin = new StringBuilder("http").append(session.isSecure() ? "s" : "").append("://0.0.0.0:80").toString();
174                     }
175                     requestURL = new StringBuilder(origin).append(requestURL).toString();
176                 } else if (requestURL.startsWith("ws://")) {
177                     requestURL = requestURL.replace("ws://", "http://");
178                 } else if (requestURL.startsWith("wss://")) {
179                     requestURL = requestURL.replace("wss://", "https://");
180                 }
181             }
182 
183             request = new AtmosphereRequest.Builder()
184                     .requestURI(uri.getPath())
185                     .requestURL(requestURL)
186                     .headers(headers)
187                     .session((HttpSession) handshakeRequest.getHttpSession())
188                     .servletPath(servletPath)
189                     .contextPath(framework.getServletContext().getContextPath())
190                     .pathInfo(pathInfo)
191                     .userPrincipal(session.getUserPrincipal())
192                     .build()
193                     .queryString(session.getQueryString());
194 
195             // TODO: Fix this crazy code.
196             framework.addInitParameter(ALLOW_QUERYSTRING_AS_REQUEST, "false");
197 
198             webSocketProcessor.open(webSocket, request, AtmosphereResponse.newInstance(framework.getAtmosphereConfig(), request, webSocket));
199 
200             framework.addInitParameter(ALLOW_QUERYSTRING_AS_REQUEST, "true");
201 
202             if (session.isOpen()) {
203                 session.addMessageHandler(new MessageHandler.Whole<String>() {
204                     @Override
205                     public void onMessage(String s) {
206                         webSocketProcessor.invokeWebSocketProtocol(webSocket, s);
207                     }
208                 });
209 
210                 session.addMessageHandler(new MessageHandler.Whole<ByteBuffer>() {
211                     @Override
212                     public void onMessage(ByteBuffer bb) {
213                         byte[] b = bb.hasArray() ? bb.array() : new byte[bb.limit()];
214                         bb.get(b);
215                         webSocketProcessor.invokeWebSocketProtocol(webSocket, b, 0, b.length);
216                     }
217                 });
218             } else {
219                 logger.trace("Session closed during onOpen {}", session);
220                 onClose(session, new CloseReason(CloseReason.CloseCodes.GOING_AWAY, "Session closed already"));
221             }
222         } catch (Throwable e) {
223             if (session.isOpen()){
224                 logger.error("", e);
225             } else {
226                 logger.warn("Session closed during onOpen", e);
227             }
228 
229             try {
230                 session.close(new CloseReason(CloseReason.CloseCodes.UNEXPECTED_CONDITION, e.getMessage()));
231             } catch (IOException e1) {
232                 logger.debug("", e);
233             }
234             return;
235         }
236 
237     }
238 
239     @Override
240     public void onClose(javax.websocket.Session session, javax.websocket.CloseReason closeCode) {
241         logger.trace("{} closed {}", session, closeCode);
242         if (request != null) {
243             request.destroy();
244             webSocketProcessor.close(webSocket, closeCode.getCloseCode().getCode());
245         }
246     }
247 
248     @Override
249     public void onError(javax.websocket.Session session, java.lang.Throwable t) {
250         logger.error("", t);
251         webSocketProcessor.notifyListener(webSocket,
252                 new WebSocketEventListener.WebSocketEvent<Throwable>(t, WebSocketEventListener.WebSocketEvent.TYPE.EXCEPTION, webSocket));
253     }
