37  public class JettyWebSocketUtil {
38  
39      private static final Logger logger = LoggerFactory.getLogger(JettyWebSocketUtil.class);
40  
41      public final static Action doService(AsynchronousProcessor cometSupport,
42                                           AtmosphereRequest req,
43                                           AtmosphereResponse res,
44                                           WebSocketFactory webSocketFactory) throws IOException, ServletException {
45  
46          Boolean b = (Boolean) req.getAttribute(WebSocket.WEBSOCKET_INITIATED);
47          if (b == null) b = Boolean.FALSE;
48  
49          if (!Utils.webSocketEnabled(req) && req.getAttribute(WebSocket.WEBSOCKET_ACCEPT_DONE) == null) {
50              if (req.resource() != null && req.resource().transport() == AtmosphereResource.TRANSPORT.WEBSOCKET) {
51                  WebSocket.notSupported(req, res);
52                  return Action.CANCELLED;
53              } else {
54                  return null;
55              }
56          } else {
57              if (webSocketFactory != null && !b) {
58                  req.setAttribute(WebSocket.WEBSOCKET_INITIATED, true);
59                  try {
60                      webSocketFactory.acceptWebSocket(req, res);
61                  } catch (IllegalStateException ex) {
62                      logger.trace("", ex);
63                      WebSocket.notSupported(req, res);
64                      return Action.CANCELLED;
65                  }
66                  req.setAttribute(WebSocket.WEBSOCKET_ACCEPT_DONE, true);
67                  return new Action();
68              }
69  
70              Action action = cometSupport.suspended(req, res);
71              if (action.type() == Action.TYPE.SUSPEND) {
72              } else if (action.type() == Action.TYPE.RESUME) {
73                  req.setAttribute(WebSocket.WEBSOCKET_RESUME, true);
74              }
75  
76              return action;
77          }
78      }
79  
80      public final static WebSocketFactory getFactory(final AtmosphereConfig config, final WebSocketProcessor webSocketProcessor) {
81  
82          final AtomicBoolean useBuildInSession = new AtomicBoolean(config.isSupportSession());
83          // Override the value.
84          String s = config.getInitParameter(ApplicationConfig.BUILT_IN_SESSION);
85          if (s != null) {
86              useBuildInSession.set(Boolean.valueOf(s));
87          }
88  
89          WebSocketFactory webSocketFactory = new WebSocketFactory(new WebSocketFactory.Acceptor() {
90              public boolean checkOrigin(HttpServletRequest request, String origin) {
91                  // Allow all origins
92                  logger.trace("WebSocket-checkOrigin request {} with origin {}", request.getRequestURI(), origin);
93                  return true;
94              }
95  
96              public org.eclipse.jetty.websocket.WebSocket doWebSocketConnect(HttpServletRequest request, String protocol) {
97                  logger.trace("WebSocket-connect request {} with protocol {}", request.getRequestURI(), protocol);
98  
99                  boolean isDestroyable = false;
100                 String s = config.getInitParameter(ApplicationConfig.RECYCLE_ATMOSPHERE_REQUEST_RESPONSE);
101                 if (s != null && Boolean.valueOf(s)) {
102                     isDestroyable = true;
103                 }
104 
105                 if (!webSocketProcessor.handshake(request)) {
106                     // res.sendError(HttpServletResponse.SC_FORBIDDEN, "WebSocket requests rejected.");
107                     throw new IllegalStateException();
108                 }
109 
110                 return new JettyWebSocketHandler(AtmosphereRequest.cloneRequest(request, false, useBuildInSession.get(), isDestroyable),
111                         config.framework(), webSocketProcessor);
112             }
113         });
114 
115         int bufferSize = 8192;
116         if (config.getInitParameter(ApplicationConfig.WEBSOCKET_BUFFER_SIZE) != null) {
117             bufferSize = Integer.valueOf(config.getInitParameter(ApplicationConfig.WEBSOCKET_BUFFER_SIZE));
118         }
119         logger.debug("WebSocket Buffer size {}", bufferSize);
120         webSocketFactory.setBufferSize(bufferSize);
121 
122         int timeOut = 5 * 60000;
123         if (config.getInitParameter(ApplicationConfig.WEBSOCKET_IDLETIME) != null) {
124             timeOut = Integer.valueOf(config.getInitParameter(ApplicationConfig.WEBSOCKET_IDLETIME));
125         }
126         logger.debug("WebSocket idle timeout {}", timeOut);
127         webSocketFactory.setMaxIdleTime(timeOut);
128 
129         int maxTextBufferSize = 8192;
130         if (config.getInitParameter(ApplicationConfig.WEBSOCKET_MAXTEXTSIZE) != null) {
131             maxTextBufferSize = Integer.valueOf(config.getInitParameter(ApplicationConfig.WEBSOCKET_MAXTEXTSIZE));
132         }
133         logger.debug("WebSocket maxTextBufferSize {}", maxTextBufferSize);
134         webSocketFactory.setMaxTextMessageSize(maxTextBufferSize);
135 
136         int maxBinaryBufferSize = 8192;
137         if (config.getInitParameter(ApplicationConfig.WEBSOCKET_MAXBINARYSIZE) != null) {
138             maxBinaryBufferSize = Integer.valueOf(config.getInitParameter(ApplicationConfig.WEBSOCKET_MAXBINARYSIZE));
139         }
140         logger.debug("WebSocket maxBinaryBufferSize {}", maxBinaryBufferSize);
141         webSocketFactory.setMaxBinaryMessageSize(maxBinaryBufferSize);
142 
143         if (config.getInitParameter(ApplicationConfig.JETTY_WEBSOCKET_MIN_VERSION) != null) {
144             int minVersion = Integer.valueOf(config.getInitParameter(ApplicationConfig.JETTY_WEBSOCKET_MIN_VERSION));
145             webSocketFactory.setMinVersion(minVersion);
146             logger.debug("WebSocket Jetty minVersion {}", minVersion);
147         }
148 
149         return webSocketFactory;
150     }
