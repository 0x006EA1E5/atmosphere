66  public class TomcatWebSocketUtil {
67      private static final Logger logger = LoggerFactory.getLogger(TomcatWebSocketUtil.class);
68      private static final Queue<MessageDigest> sha1Helpers = new ConcurrentLinkedQueue<MessageDigest>();
69      private static byte[] WS_ACCEPT;

71      static {
72          try {
73              WS_ACCEPT = "258EAFA5-E914-47DA-95CA-C5AB0DC85B11".getBytes("ISO_8859_1");
74          } catch (Exception e) {
75              logger.error("", e);
76          }
77      }

80      public static Action doService(AtmosphereRequest req, AtmosphereResponse res,
81                                     Delegate delegate,
82                                     AtmosphereConfig config,
83                                     WebSocketProcessor webSocketProcessor) throws IOException, ServletException {
84          // First, handshake
85          if (req.getAttribute(WebSocket.WEBSOCKET_SUSPEND) == null) {
86  
87              boolean useBuildInSession = config.isSupportSession();
88              // Override the value.
89              String s = config.getInitParameter(ApplicationConfig.BUILT_IN_SESSION);
90              if (s != null) {
91                  useBuildInSession= Boolean.valueOf(s);
92              }
93  
94              // Information required to send the server handshake message
95              String key =  null;
96              String subProtocol = null;
97  
98              boolean allowWebSocketWithoutHeaders = req.getHeader(HeaderConfig.X_ATMO_WEBSOCKET_PROXY) != null ? true : false;
99              if (!allowWebSocketWithoutHeaders)   {
100                 if (!headerContainsToken(req, "Upgrade", "websocket")) {
101                     return delegate.doService(req, res);
102                 }
103 
104                 if (!headerContainsToken(req, "Connection", "upgrade")) {
105                     return delegate.doService(req, res);
106                 }
107 
108                 if (!headerContainsToken(req, "sec-websocket-version", "13")) {
109                     WebSocket.notSupported(req, res);
110                     return new Action(Action.TYPE.CANCELLED);
111                 }
112             }
113 
114             key = req.getHeader("Sec-WebSocket-Key");
115             if (key == null) {
116                 return delegate.doService(req, res);
117             }
118 
119             String requireSameOrigin = config.getInitParameter(ApplicationConfig.WEBSOCKET_REQUIRE_SAME_ORIGIN);
120 
121             if (!webSocketProcessor.handshake(req)) {
122                 res.sendError(HttpServletResponse.SC_FORBIDDEN, "WebSocket requests rejected.");
123                 return new Action(Action.TYPE.CANCELLED);
124             }
125 
126             if (Boolean.valueOf(requireSameOrigin) && !verifyOrigin(req)) {
127                 res.sendError(HttpServletResponse.SC_FORBIDDEN, "Origin header does not match expected value");
128                 return new Action(Action.TYPE.CANCELLED);
129             }
130 
131             // If we got this far, all is good. Accept the connection.
132             res.setHeader("Upgrade", "websocket");
133             res.setHeader("Connection", "upgrade");
134             res.setHeader("Sec-WebSocket-Accept", getWebSocketAccept(key));
135 
136             if (subProtocol != null) {
137                 res.setHeader("Sec-WebSocket-Protocol", subProtocol);
138             }
139 
140             HttpServletRequest hsr = req.wrappedRequest();
141             while (hsr instanceof HttpServletRequestWrapper)
142                 hsr = (HttpServletRequest) ((HttpServletRequestWrapper) hsr).getRequest();
143 
144             RequestFacade facade = (RequestFacade) hsr;
145             boolean isDestroyable = false;
146             s = config.getInitParameter(ApplicationConfig.RECYCLE_ATMOSPHERE_REQUEST_RESPONSE);
147             if (s != null && Boolean.valueOf(s)) {
148                 isDestroyable = true;
149             }
150             StreamInbound inbound = new TomcatWebSocketHandler(
151                     AtmosphereRequestImpl.cloneRequest(req, true, useBuildInSession, isDestroyable, config.getInitParameter(PROPERTY_SESSION_CREATE, true)),
152                     config.framework(), webSocketProcessor);
153             facade.doUpgrade(inbound);
154             return new Action(Action.TYPE.CREATED);
155         }
156 
157         try {
158             Action action = delegate.suspended(req, res);
159             if (action.type() == Action.TYPE.RESUME) {
160                 req.setAttribute(WebSocket.WEBSOCKET_RESUME, true);
161             }
162             return action;
163         } catch (Exception ex) {
164             logger.warn("", ex);
165         }
166         return Action.CANCELLED;
167     }

169     private static boolean verifyOrigin(HttpServletRequest req) {
170         String origin = req.getHeader("Origin");
171         if (origin == null) {
172             return false;
173         }
174 
175         try {
176             URI requestUri = new URI(req.getRequestURL().toString());
177             URI originUri = new URI(origin);
178 
179             // First check that the protocol (http, https, etc) is not null and matches
180             if (requestUri.getScheme() != null && originUri.getScheme() != null &&
181                     requestUri.getScheme().equals(originUri.getScheme())) {
182 
183                 // If the authority (userinfo@hostname:port) is not null and matches, then the origin is verified
184                 if (requestUri.getAuthority() != null && originUri.getAuthority() != null &&
185                         requestUri.getAuthority().equals(originUri.getAuthority())) {
186                     return true;
187                 }
188             }
189         } catch (URISyntaxException e) {
190             logger.error("Unable to verify request origin", e);
191         }
192 
193         return false;
194     }

196     /*
197     * This only works for tokens. Quoted strings need more sophisticated
198     * parsing.
199     */
200     private static boolean headerContainsToken(HttpServletRequest req,
201                                                String headerName, String target) {
202         Enumeration<String> headers = req.getHeaders(headerName);
203         while (headers.hasMoreElements()) {
204             String header = headers.nextElement();
205             String[] tokens = header.split(",");
206             for (String token : tokens) {
207                 if (target.equalsIgnoreCase(token.trim())) {
208                     return true;
209                 }
210             }
211         }
212         return false;
213     }

216     /*
217      * This only works for tokens. Quoted strings need more sophisticated
218      * parsing.
219      */
220     private static List<String> getTokensFromHeader(HttpServletRequest req,
221                                                     String headerName) {
222         List<String> result = new ArrayList<String>();
223 
224         Enumeration<String> headers = req.getHeaders(headerName);
225         while (headers.hasMoreElements()) {
226             String header = headers.nextElement();
227             String[] tokens = header.split(",");
228             for (String token : tokens) {
229                 result.add(token.trim());
230             }
231         }
232         return result;
233     }

236     private static String getWebSocketAccept(String key) throws ServletException {
237 
238         MessageDigest sha1Helper = sha1Helpers.poll();
239         if (sha1Helper == null) {
240             try {
241                 sha1Helper = MessageDigest.getInstance("SHA1");
242             } catch (NoSuchAlgorithmException e) {
243                 throw new ServletException(e);
244             }
245         }
246 
247         sha1Helper.reset();
248         try {
249             sha1Helper.update(key.getBytes("ISO_8859_1"));
250         } catch (UnsupportedEncodingException e) {
251             throw new ServletException(e);
252         }
253         String result = org.apache.catalina.util.Base64.encode(sha1Helper.digest(WS_ACCEPT));
254 
255         sha1Helpers.add(sha1Helper);
256 
257         return result;
258     }
