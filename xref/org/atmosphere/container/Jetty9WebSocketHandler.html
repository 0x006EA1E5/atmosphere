16  package org.atmosphere.container;
17  
18  import org.atmosphere.container.version.Jetty9WebSocket;
19  import org.atmosphere.cpr.AtmosphereFramework;
20  import org.atmosphere.cpr.AtmosphereRequest;
21  import org.atmosphere.cpr.AtmosphereResponse;
22  import org.atmosphere.websocket.WebSocket;
23  import org.atmosphere.websocket.WebSocketProcessor;
24  import org.eclipse.jetty.websocket.api.Session;
25  import org.eclipse.jetty.websocket.api.WebSocketListener;
26  import org.eclipse.jetty.websocket.servlet.ServletUpgradeRequest;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import javax.servlet.http.HttpServletRequest;
31  import java.lang.reflect.Field;
32  
33  import static org.atmosphere.cpr.ApplicationConfig.PROPERTY_SESSION_CREATE;
34  
35  public class Jetty9WebSocketHandler implements WebSocketListener {
36  
37      private static final Logger logger = LoggerFactory.getLogger(Jetty9WebSocketHandler.class);
38  
39      private AtmosphereRequest request;
40      private final AtmosphereFramework framework;
41      private final WebSocketProcessor webSocketProcessor;
42      private WebSocket webSocket;
43      private static final boolean jetty93Up;
44  
45      static {
46          Exception ex = null;
47          try {
48              Class.forName("org.eclipse.jetty.websocket.api.WebSocketFrameListener");
49          } catch (ClassNotFoundException e) {
50              ex = e;
51          } finally {
52              jetty93Up = ex == null ? true : false;
53          }
54      }
55  
56      public Jetty9WebSocketHandler(HttpServletRequest request, AtmosphereFramework framework, WebSocketProcessor webSocketProcessor) {
57          this.framework = framework;
58          this.request = cloneRequest(request);
59          this.webSocketProcessor = webSocketProcessor;
60      }
61  
62      private AtmosphereRequest cloneRequest(final HttpServletRequest request) {
63          try {
64              AtmosphereRequest r = AtmosphereRequest.wrap(request);
65              return AtmosphereRequest.cloneRequest(r, false, false, false, framework.getAtmosphereConfig().getInitParameter(PROPERTY_SESSION_CREATE, true));
66          } catch (Exception ex) {
67              logger.error("", ex);
68              throw new RuntimeException("Invalid WebSocket Request");
69          }
70      }
71  
72      @Override
73      public void onWebSocketBinary(byte[] data, int offset, int length) {
74          logger.trace("WebSocket.onMessage (bytes)");
75          webSocketProcessor.invokeWebSocketProtocol(webSocket, data, offset, length);
76      }
77  
78      @Override
79      public void onWebSocketClose(int closeCode, String s) {
80          logger.trace("onClose {}:{}", closeCode, s);
81          try {
82              webSocketProcessor.close(webSocket, closeCode);
83          } finally {
84              request.destroy();
85          }
86      }
87  
88      @Override
89      public void onWebSocketConnect(Session session) {
90          logger.trace("WebSocket.onOpen.");
91          webSocket = new Jetty9WebSocket(session, framework.getAtmosphereConfig());
92  
93          /**
94           * https://github.com/Atmosphere/atmosphere/issues/1998
95           * The Original Jetty Request will be recycled, hence we must loads its content in memory. We can't do that before
96           * as it break Jetty 9.3.0 upgrade process.
97           *
98           * This is a performance regression from 9.2 as we need to clone again the request. 9.3.0+ should use jsr 356!
99           */
100         if (jetty93Up) {
101             HttpServletRequest r = originalRequest(session);
102             if (r != null) {
103                 // We close except the session which we can still reach.
104                 request = AtmosphereRequest.cloneRequest(r, true, false, false, framework.getAtmosphereConfig().getInitParameter(PROPERTY_SESSION_CREATE, true));
105             } else {
106                 // Bad Bad Bad
107                 request = AtmosphereRequest.cloneRequest(r, true, true, false, framework.getAtmosphereConfig().getInitParameter(PROPERTY_SESSION_CREATE, true));
108             }
109         }
110 
111         try {
112             webSocketProcessor.open(webSocket, request, AtmosphereResponse.newInstance(framework.getAtmosphereConfig(), request, webSocket));
113         } catch (Exception e) {
114             logger.warn("Failed to connect to WebSocket", e);
115         }
116     }
117 
118     @Override
119     public void onWebSocketError(Throwable e) {
120         logger.error("{}", e);
121         onWebSocketClose(1006, "Unexpected error");
122     }
123 
124     @Override
125     public void onWebSocketText(String s) {
126         logger.trace("WebSocket.onMessage (bytes)");
127         webSocketProcessor.invokeWebSocketProtocol(webSocket, s);
128     }
129 
130     private HttpServletRequest originalRequest(Session session) {
131         try {
132             // Oh boy...
133             ServletUpgradeRequest request = (ServletUpgradeRequest) session.getUpgradeRequest();
134             Field[] fields = ServletUpgradeRequest.class.getDeclaredFields();
135             for (Field f : fields) {
136                 f.setAccessible(true);
137                 Object o = f.get(request);
138                 if (o instanceof HttpServletRequest) {
139                     return HttpServletRequest.class.cast(o);
140                 }
141             }
142         } catch (Exception ex) {
143             logger.error("", ex);
144         }
145         return null;
146     }
147
