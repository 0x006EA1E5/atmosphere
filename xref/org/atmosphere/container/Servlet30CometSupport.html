38  /**
39   * This class is used when the {@link org.atmosphere.cpr.AtmosphereFramework} detect the container
40   * detect Servlet 3.0 Asynch API.
41   *
42   * @author Jeanfrancois Arcand
43   */
44  public class Servlet30CometSupport extends AsynchronousProcessor {
45  
46      private static final Logger logger = LoggerFactory.getLogger(Servlet30CometSupport.class);
47  
48      public Servlet30CometSupport(AtmosphereConfig config) {
49          super(config);
50      }
51  
52      /**
53       * Return "javax.servlet".
54       *
55       * @return "javax.servlet"
56       */
57      @Override
58      public String getContainerName() {
59          return super.getContainerName() + " using javax.servlet/3.0";
60      }
61  
62      @Override
63      public Action service(AtmosphereRequest request, AtmosphereResponse response)
64              throws IOException, ServletException {
65  
66          Action action = suspended(request, response);
67          if (action.type() == Action.TYPE.SUSPEND) {
68              suspend(action, request, response);
69          } else if (action.type() == Action.TYPE.RESUME) {
70  
71              Action nextAction = resumed(request, response);
72              if (nextAction.type() == Action.TYPE.SUSPEND) {
73                  suspend(action, request, response);
74              }
75          }
76  
77          return action;
78      }
79  
80      /**
81       * Suspend the connection by invoking {@link AtmosphereRequest#startAsync()}
82       *
83       * @param action The {@link org.atmosphere.cpr.Action}
84       * @param req    the {@link AtmosphereRequest}
85       * @param res    the {@link AtmosphereResponse}
86       * @throws java.io.IOException
87       * @throws javax.servlet.ServletException
88       */
89      private void suspend(Action action, AtmosphereRequest req, AtmosphereResponse res)
90              throws IOException, ServletException {
91  
92          if (!req.isAsyncStarted() && !Utils.webSocketEnabled(req)) {
93              AsyncContext asyncContext = req.startAsync(req, res);
94              asyncContext.addListener(new CometListener(this, res.uuid()));
95              // Do nothing except setting the times out
96              if (action.timeout() != -1) {
97                  asyncContext.setTimeout(action.timeout());
98              } else {
99                  // Jetty 8 does something really weird if you set it to
100                 // Long.MAX_VALUE, which is to resume automatically.
101                 asyncContext.setTimeout(Integer.MAX_VALUE);
102             }
103             req.setAttribute(FrameworkConfig.ASYNC_CONTEXT, asyncContext);
104         }
105     }
106 
107     @Override
108     public void action(AtmosphereResourceImpl r) {
109         super.action(r);
110         if (r.action().type() == Action.TYPE.RESUME && r.isInScope()) {
111             endAsyncContext(r.getRequest(false));
112         }
113     }
114 
115     @Override
116     public AsyncSupport complete(AtmosphereResourceImpl r) {
117         endAsyncContext(r.getRequest(false));
118         return null;
119     }
120 
121     public void endAsyncContext(AtmosphereRequest request){
122         AsyncContext asyncContext = (AsyncContext) request.getAttribute(FrameworkConfig.ASYNC_CONTEXT);
123         if (asyncContext != null) {
124             try {
125                 asyncContext.complete();
126             } catch (IllegalStateException ex) {
127                 // Already completed. Jetty throw an exception on shutdown with log
128                 try {
129                     logger.trace("Already resumed!", ex);
130                 } catch (Exception ex2){};
131             } finally {
132                 request.removeAttribute(FrameworkConfig.ASYNC_CONTEXT);
133             }
134         }
135     }
136 
137     @Override
138     public Action cancelled(AtmosphereRequest req, AtmosphereResponse res)
139             throws IOException, ServletException {
140 
141         Action action = super.cancelled(req, res);
142         if (req.getAttribute(MAX_INACTIVE) != null && Long.class.cast(req.getAttribute(MAX_INACTIVE)) == -1) {
143             endAsyncContext(req);
144         }
145         return action;
146     }
147 
148     /**
149      * Servlet 3.0 async listener support.
150      */
151     private final static class CometListener implements AsyncListener {
152 
153         private final AsynchronousProcessor p;
154         private final String uuid;
155 
156         // For JBoss 7 https://github.com/Atmosphere/atmosphere/issues/240
157         public CometListener() {
158             this.uuid = "-1";
159             p = null;
160         }
161 
162         public CometListener(AsynchronousProcessor processor, String uuid) {
163             this.p = processor;
164             this.uuid = uuid;
165         }
166 
167         @Override
168         public void onComplete(AsyncEvent event) throws IOException {
169             logger.trace("Resumed (completed): event: {}", uuid);
170         }
171 
172         @Override
173         public void onTimeout(AsyncEvent event) throws IOException {
174             logger.trace("onTimeout(): event: {}", event.getAsyncContext().getRequest());
175 
176             if (p == null) {
177                 logger.error("Invalid state - CometListener");
178                 return;
179             }
180 
181             final AsyncContext asyncContext = event.getAsyncContext();
182             try {
183                 p.timedout((AtmosphereRequest) asyncContext.getRequest(),
184                         (AtmosphereResponse) asyncContext.getResponse());
185             } catch (ServletException ex) {
186                 logger.warn("onTimeout(): failed timing out comet response: " + event.getAsyncContext().getResponse(), ex);
187             } finally {
188                 try {
189                     asyncContext.complete();
190                 } catch (IllegalStateException ex) {
191                     // The complete method has been already called.
192                     // https://tomcat.apache.org/tomcat-7.0-doc/api/org/apache/coyote/AsyncStateMachine.html
193                     logger.trace("", ex);
194                 }
195             }
196         }
197 
198         @Override
199         public void onError(AsyncEvent event) {
200             logger.trace("onError(): event: {}", event.getAsyncContext().getResponse());
201 
202             if (p == null) {
203                 logger.error("Invalid state - CometListener");
204                 return;
205             }
206 
207             try {
208                 p.cancelled((AtmosphereRequest) event.getAsyncContext().getRequest(),
209                         (AtmosphereResponse) event.getAsyncContext().getResponse());
210             } catch (Throwable ex) {
211                 logger.warn("failed cancelling comet response: " + event.getAsyncContext().getResponse(), ex);
212             }
213         }
214 
215         @Override
216         public void onStartAsync(AsyncEvent event) {
217             logger.trace("onStartAsync(): event: {}", event.getAsyncContext().getResponse());
218         }
219     }
