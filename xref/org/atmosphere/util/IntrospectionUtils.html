61       * Call execute() - any ant-like task should work
62       */
63      public static void execute(Object proxy, String method) throws Exception {
64          Method executeM = null;
65          Class<?> c = proxy.getClass();
66          Class<?> params[] = new Class[0];
67          //    params[0]=args.getClass();
68          executeM = findMethod(c, method, params);
69          if (executeM == null) {
70              throw new RuntimeException("No execute in " + proxy.getClass());
71          }
72          executeM.invoke(proxy, (Object[]) null);//new Object[] { args });
73      }

75      /**
76       * Call void setAttribute( String ,Object )
77       */
78      public static void setAttribute(Object proxy, String name, Object value)
79              throws Exception {
80          if (proxy instanceof AttributeHolder) {
81              ((AttributeHolder) proxy).setAttribute(name, value);
82              return;
83          }
84  
85          Method executeM = null;
86          Class<?> c = proxy.getClass();
87          Class<?> params[] = new Class[2];
88          params[0] = String.class;
89          params[1] = Object.class;
90          executeM = findMethod(c, "setAttribute", params);
91          if (executeM == null) {
92              logger.debug("No setAttribute in {}", proxy.getClass());
93              return;
94          }
95  
96          logger.debug("Setting {}={} in proxy: {}", new Object[]{name, value, proxy});
97          executeM.invoke(proxy, new Object[]{name, value});
98          return;
99      }

116     /**
117      * Construct a URLClassLoader. Will compile and work in JDK1.1 too.
118      */
119     public static ClassLoader getURLClassLoader(URL urls[], ClassLoader parent) {
120         try {
121             Class<?> urlCL = Class.forName("java.net.URLClassLoader");
122             Class<?> paramT[] = new Class[2];
123             paramT[0] = urls.getClass();
124             paramT[1] = ClassLoader.class;
125             Method m = findMethod(urlCL, "newInstance", paramT);
126             if (m == null) {
127                 return null;
128             }
129 
130             ClassLoader cl = (ClassLoader) m.invoke(urlCL, new Object[]{urls, parent});
131             return cl;
132         } catch (ClassNotFoundException ex) {
133             // jdk1.1
134             return null;
135         } catch (Exception ex) {
136             logger.error("failed getting URLClassLoader", ex);
137             return null;
138         }
139     }

157     final static public boolean invokeProperty(Object object, String setter, String name, String value) {
158         try {
159             Method methods[] = findMethods(object.getClass());
160             Method setPropertyMethodVoid = null;
161             Method setPropertyMethodBool = null;
162 
163             // First, the ideal case - a setFoo( String ) method
164             for (int i = 0; i < methods.length; i++) {
165                 Class<?> paramT[] = methods[i].getParameterTypes();
166                 if (setter.equals(methods[i].getName()) && paramT.length == 1
167                         && "java.lang.String".equals(paramT[0].getName())) {
168 
169                     methods[i].invoke(object, new Object[]{value});
170                     return true;
171                 }
172             }
173 
174             // Try a setFoo ( int ) or ( boolean )
175             for (int i = 0; i < methods.length; i++) {
176                 boolean ok = true;
177                 if (setter.equals(methods[i].getName())
178                         && methods[i].getParameterTypes().length == 1) {
179 
180                     // match - find the type and invoke it
181                     Class<?> paramType = methods[i].getParameterTypes()[0];
182                     Object params[] = new Object[1];
183 
184                     // Try a setFoo ( int )
185                     if ("java.lang.Integer".equals(paramType.getName())
186                             || "int".equals(paramType.getName())) {
187                         try {
188                             params[0] = new Integer(value);
189                         } catch (NumberFormatException ex) {
190                             ok = false;
191                         }
192                         // Try a setFoo ( long )
193                     } else if ("java.lang.Long".equals(paramType.getName())
194                             || "long".equals(paramType.getName())) {
195                         try {
196                             params[0] = new Long(value);
197                         } catch (NumberFormatException ex) {
198                             ok = false;
199                         }
200 
201                         // Try a setFoo ( boolean )
202                     } else if ("java.lang.Boolean".equals(paramType.getName())
203                             || "boolean".equals(paramType.getName())) {
204                         params[0] = new Boolean(value);
205 
206                         // Try a setFoo ( InetAddress )
207                     } else if ("java.net.InetAddress".equals(paramType
208                             .getName())) {
209                         try {
210                             params[0] = InetAddress.getByName(value);
211                         } catch (UnknownHostException exc) {
212                             debug("Unable to resolve host name:" + value);
213                             ok = false;
214                         }
215 
216                         // Unknown type
217                     } else {
218                         debug("Unknown type " + paramType.getName());
219                     }
220 
221                     if (ok) {
222                         methods[i].invoke(object, params);
223                         return true;
224                     }
225                 }
226 
227                 // save "setProperty" for later
228                 if ("setProperty".equals(methods[i].getName())) {
229                     if (methods[i].getReturnType() == Boolean.TYPE) {
230                         setPropertyMethodBool = methods[i];
231                     } else {
232                         setPropertyMethodVoid = methods[i];
233                     }
234 
235                 }
236             }
237 
238             // Ok, no setXXX found, try a setProperty("name", "value")
239             if (setPropertyMethodBool != null || setPropertyMethodVoid != null) {
240                 Object params[] = new Object[2];
241                 params[0] = name;
242                 params[1] = value;
243                 if (setPropertyMethodBool != null) {
244                     try {
245                         return (Boolean) setPropertyMethodBool.invoke(object, params);
246                     } catch (IllegalArgumentException biae) {
247                         //the boolean method had the wrong
248                         //parameter types. lets try the other
249                         if (setPropertyMethodVoid != null) {
250                             setPropertyMethodVoid.invoke(object, params);
251                             return true;
252                         } else {
253                             throw biae;
254                         }
255                     }
256                 } else {
257                     setPropertyMethodVoid.invoke(object, params);
258                     return true;
259                 }
260             }
261 
262         } catch (IllegalArgumentException e) {
263             logger.info("failed, object: " + object + ", setter: " + setter + ", value: " + value, e);
264         } catch (Exception e) {
265             if (dbg > 0) {
266                 debug(e.getClass().getSimpleName() + " for " + object.getClass() + " " + setter + "=" + value + ")");
267             }
268             if (dbg > 1) {
269                 logger.debug("", e);
270             }
271         }
272 
273         return false;
274     }

276     public static Object getProperty(Object object, String name) {
277         String getter = "get" + capitalize(name);
278         String isGetter = "is" + capitalize(name);
279 
280         try {
281             Method methods[] = findMethods(object.getClass());
282             Method getPropertyMethod = null;
283 
284             // First, the ideal case - a getFoo() method
285             for (int i = 0; i < methods.length; i++) {
286                 Class<?> paramT[] = methods[i].getParameterTypes();
287                 if (getter.equals(methods[i].getName()) && paramT.length == 0) {
288                     return methods[i].invoke(object, (Object[]) null);
289                 }
290                 if (isGetter.equals(methods[i].getName()) && paramT.length == 0) {
291                     return methods[i].invoke(object, (Object[]) null);
292                 }
293 
294                 if ("getProperty".equals(methods[i].getName())) {
295                     getPropertyMethod = methods[i];
296                 }
297             }
298 
299             // Ok, no setXXX found, try a getProperty("name")
300             if (getPropertyMethod != null) {
301                 Object params[] = new Object[1];
302                 params[0] = name;
303                 return getPropertyMethod.invoke(object, params);
304             }
305 
306         } catch (IllegalArgumentException e) {
307             logger.info("failed, object: " + object + ", name: " + name, e);
308         } catch (Exception e) {
309             if (dbg > 0) {
310                 debug(e.getClass().getSimpleName() + " for " + object.getClass() + " " + name + ")");
311             }
312             if (dbg > 1) {
313                 logger.debug("", e);
314             }
315         }
316 
317         return null;
318     }

323         String setter = "set" + capitalize(name);
324         try {
325             Method methods[] = findMethods(object.getClass());
326             Method setPropertyMethod = null;
327             // find setFoo() method
328             for (int i = 0; i < methods.length; i++) {
329                 Class<?> paramT[] = methods[i].getParameterTypes();
330                 if (setter.equals(methods[i].getName()) && paramT.length == 0) {

338             }
339             if (dbg > 1) {
340                 logger.debug("", e);
341             }
342         }

345     /**
346      * Replace ${NAME} with the property value
347      */
348     public static String replaceProperties(String value,
349                                            Hashtable<String, String> staticProp, PropertySource dynamicProp[]) {
350         if (value.indexOf("$") < 0) {
351             return value;
352         }
353         StringBuilder sb = new StringBuilder();
354         int prev = 0;
355         // assert value!=nil
356         int pos;
357         while ((pos = value.indexOf("$", prev)) >= 0) {
358             if (pos > 0) {
359                 sb.append(value.substring(prev, pos));
360             }
361             if (pos == (value.length() - 1)) {
362                 sb.append('$');
363                 prev = pos + 1;
364             } else if (value.charAt(pos + 1) != '{') {
365                 sb.append('$');
366                 prev = pos + 1; // XXX
367             } else {
368                 int endName = value.indexOf('}', pos);
369                 if (endName < 0) {
370                     sb.append(value.substring(pos));
371                     prev = value.length();
372                     continue;
373                 }
374                 String n = value.substring(pos + 2, endName);
375                 String v = null;
376                 if (staticProp != null) {
377                     v = staticProp.get(n);
378                 }
379                 if (v == null && dynamicProp != null) {
380                     for (int i = 0; i < dynamicProp.length; i++) {
381                         v = dynamicProp[i].getProperty(n);
382                         if (v != null) {
383                             break;
384                         }
385                     }
386                 }
387                 if (v == null)
388                     v = "${" + n + "}";
389 
390                 sb.append(v);
391                 prev = endName + 1;
392             }
393         }
394         if (prev < value.length())
395             sb.append(value.substring(prev));
396         return sb.toString();
397     }

399     /**
400      * Reverse of Introspector.decapitalize
401      */
402     public static String capitalize(String name) {
403         if (name == null || name.length() == 0) {
404             return name;
405         }
406         char chars[] = name.toCharArray();
407         chars[0] = Character.toUpperCase(chars[0]);
408         return new String(chars);
409     }

411     public static String unCapitalize(String name) {
412         if (name == null || name.length() == 0) {
413             return name;
414         }
415         char chars[] = name.toCharArray();
416         chars[0] = Character.toLowerCase(chars[0]);
417         return new String(chars);
418     }

422     /**
423      * Add all the jar files in a dir to the classpath, represented as a Vector
424      * of URLs.
425      */
426     @SuppressWarnings("unchecked")
427     public static void addToClassPath(Vector<URL> cpV, String dir) {
428         try {
429             String cpComp[] = getFilesByExt(dir, ".jar");
430             if (cpComp != null) {
431                 int jarCount = cpComp.length;
432                 for (int i = 0; i < jarCount; i++) {
433                     URL url = getURL(dir, cpComp[i]);
434                     if (url != null)
435                         cpV.addElement(url);
436                 }
437             }
438         } catch (Exception ex) {
439             logger.debug("failed to add urls to classpath", ex);
440         }
441     }

443     @SuppressWarnings("unchecked")
444     public static void addToolsJar(Vector<URL> v) {
445         try {
446             // Add tools.jar in any case
447             File f = new File(System.getProperty("java.home")
448                     + "/../lib/tools.jar");
449 
450             if (!f.exists()) {
451                 // On some systems java.home gets set to the root of jdk.
452                 // That's a bug, but we can work around and be nice.
453                 f = new File(System.getProperty("java.home") + "/lib/tools.jar");
454                 if (f.exists()) {
455                     logger.debug("Detected strange java.home value {}, it should point to jre",
456                             System.getProperty("java.home"));
457                 }
458             }
459             URL url = new URL("file", "", f.getAbsolutePath());
460 
461             v.addElement(url);
462         } catch (MalformedURLException ex) {
463             logger.debug("failed to add tools jar url to vector", ex);
464         }
465     }

467     /**
468      * Return all files with a given extension in a dir
469      */
470     public static String[] getFilesByExt(String ld, String ext) {
471         File dir = new File(ld);
472         String[] names = null;
473         final String lext = ext;
474         if (dir.isDirectory()) {
475             names = dir.list(new FilenameFilter() {
476                 public boolean accept(File d, String name) {
477                     if (name.endsWith(lext)) {
478                         return true;
479                     }
480                     return false;
481                 }
482             });
483         }
484         return names;
485     }

487     /**
488      * Construct a file url from a file, using a base dir
489      */
490     public static URL getURL(String base, String file) {
491         try {
492             File baseF = new File(base);
493             File f = new File(baseF, file);
494             String path = f.getCanonicalPath();
495             if (f.isDirectory()) {
496                 path += "/";
497             }
498             if (!f.exists()) {
499                 return null;
500             }
501             return new URL("file", "", path);
502         } catch (Exception ex) {
503             logger.debug("failed to get url, base: " + base + ", file: " + file, ex);
504             return null;
505         }
506     }

508     /**
509      * Add elements from the classpath <i>cp </i> to a Vector <i>jars </i> as
510      * file URLs (We use Vector for JDK 1.1 compat).
511      * <p/>
512      *
513      * @param jars The jar list
514      * @param cp   a String classpath of directory or jar file elements
515      *             separated by path.separator delimiters.
516      * @throws IOException           If an I/O error occurs
517      * @throws MalformedURLException Doh ;)
518      */
519     @SuppressWarnings("unchecked")
520     public static void addJarsFromClassPath(Vector<URL> jars, String cp)
521             throws IOException, MalformedURLException {
522         String sep = System.getProperty("path.separator");
523         String token;
524         StringTokenizer st;
525         if (cp != null) {
526             st = new StringTokenizer(cp, sep);
527             while (st.hasMoreTokens()) {
528                 File f = new File(st.nextToken());
529                 String path = f.getCanonicalPath();
530                 if (f.isDirectory()) {
531                     path += "/";
532                 }
533                 URL url = new URL("file", "", path);
534                 if (!jars.contains(url)) {
535                     jars.addElement(url);
536                 }
537             }
538         }
539     }

552     /**
553      * Construct a URL classpath from files in a directory, a cpath property,
554      * and tools.jar.
555      */
556     @SuppressWarnings("unchecked")
557     public static URL[] getClassPath(String dir, String cpath,
558                                      String cpathProp, boolean addTools) throws IOException,
559             MalformedURLException {
560         Vector<URL> jarsV = new Vector<URL>();
561         if (dir != null) {
562             // Add dir/classes first, if it exists
563             URL url = getURL(dir, "classes");
564             if (url != null)
565                 jarsV.addElement(url);
566             addToClassPath(jarsV, dir);
567         }
568 
569         if (cpath != null)
570             addJarsFromClassPath(jarsV, cpath);
571 
572         if (cpathProp != null) {
573             String cpath1 = System.getProperty(cpathProp);
574             addJarsFromClassPath(jarsV, cpath1);
575         }
576 
577         if (addTools)
578             addToolsJar(jarsV);
579 
580         return getClassPath(jarsV);
581     }

585     public static boolean processArgs(Object proxy, String args[])
586             throws Exception {
587         String args0[] = null;
588         if (null != findMethod(proxy.getClass(), "getOptions1", new Class[]{})) {
589             args0 = (String[]) callMethod0(proxy, "getOptions1");
590         }
591 
592         if (args0 == null) {
593             //args0=findVoidSetters(proxy.getClass());
594             args0 = findBooleanSetters(proxy.getClass());
595         }
596         Hashtable<String, String> h = null;
597         if (null != findMethod(proxy.getClass(), "getOptionAliases",
598                 new Class[]{})) {
599             h = (Hashtable<String, String>) callMethod0(proxy, "getOptionAliases");
600         }
601         return processArgs(proxy, args, args0, null, h);
602     }

604     public static boolean processArgs(Object proxy, String args[],
605                                       String args0[], String args1[],
606                                       Hashtable<String, String> aliases) throws Exception {
607         for (int i = 0; i < args.length; i++) {
608             String arg = args[i];
609             if (arg.startsWith("-"))
610                 arg = arg.substring(1);
611             if (aliases != null && aliases.get(arg) != null)
612                 arg = aliases.get(arg);
613 
614             if (args0 != null) {
615                 boolean set = false;
616                 for (int j = 0; j < args0.length; j++) {
617                     if (args0[j].equalsIgnoreCase(arg)) {
618                         setProperty(proxy, args0[j], "true");
619                         set = true;
620                         break;
621                     }
622                 }
623                 if (set)
624                     continue;
625             }
626             if (args1 != null) {
627                 for (int j = 0; j < args1.length; j++) {
628                     if (args1[j].equalsIgnoreCase(arg)) {
629                         i++;
630                         if (i >= args.length)
631                             return false;
632                         setProperty(proxy, arg, args[i]);
633                         break;
634                     }
635                 }
636             } else {
637                 // if args1 is not specified,assume all other options have param
638                 i++;
639                 if (i >= args.length)
640                     return false;
641                 setProperty(proxy, arg, args[i]);
642             }
643 
644         }
645         return true;
646     }

654     @SuppressWarnings("unchecked")
655     public static String[] findVoidSetters(Class<?> c) {
656         Method m[] = findMethods(c);
657         if (m == null)
658             return null;
659         Vector<String> v = new Vector<String>();
660         for (int i = 0; i < m.length; i++) {
661             if (m[i].getName().startsWith("set")
662                     && m[i].getParameterTypes().length == 0) {
663                 String arg = m[i].getName().substring(3);
664                 v.addElement(unCapitalize(arg));
665             }
666         }
667         String s[] = new String[v.size()];
668         for (int i = 0; i < s.length; i++) {
669             s[i] = (String) v.elementAt(i);
670         }
671         return s;

699     @SuppressWarnings("unchecked")
700     public static Method[] findMethods(Class<?> c) {
701         Method methods[] = (Method[]) objectMethods.get(c);
702         if (methods != null)
703             return methods;
704 
705         methods = c.getMethods();
706         objectMethods.put(c, methods);
707         return methods;
708     }

710     public static Method findMethod(Class<?> c, String name,
711                                     Class<?> params[]) {
712         Method methods[] = findMethods(c);
713         if (methods == null)
714             return null;
715         for (int i = 0; i < methods.length; i++) {
716             if (methods[i].getName().equals(name)) {
717                 Class<?> methodParams[] = methods[i].getParameterTypes();
718                 if (methodParams == null)
719                     if (params == null || params.length == 0)
720                         return methods[i];
721                 if (params == null)
722                     if (methodParams == null || methodParams.length == 0)
723                         return methods[i];
724                 if (params.length != methodParams.length)
725                     continue;
726                 boolean found = true;
727                 for (int j = 0; j < params.length; j++) {
728                     if (params[j] != methodParams[j]) {
729                         found = false;
730                         break;
731                     }
732                 }
733                 if (found)
734                     return methods[i];
735             }
736         }
737         return null;
738     }

744     public static boolean hasHook(Object obj, String methodN) {
745         try {
746             Method myMethods[] = findMethods(obj.getClass());
747             for (int i = 0; i < myMethods.length; i++) {
748                 if (methodN.equals(myMethods[i].getName())) {
749                     // check if it's overriden
750                     Class<?> declaring = myMethods[i].getDeclaringClass();
751                     Class<?> parentOfDeclaring = declaring.getSuperclass();
752                     // this works only if the base class doesn't extend
753                     // another class.
754 
755                     // if the method is declared in a top level class
756                     // like BaseInterceptor parent is Object, otherwise
757                     // parent is BaseInterceptor or an intermediate class
758                     if (!"java.lang.Object".equals(parentOfDeclaring.getName())) {
759                         return true;
760                     }
761                 }
762             }
763         } catch (Exception ex) {
764             logger.debug("hasHook() failed", ex);
765         }
766         return false;
767     }

844     public static Object convert(String object, Class<?> paramType) {
845         Object result = null;
846         if ("java.lang.String".equals(paramType.getName())) {
847             result = object;
848         } else if ("java.lang.Integer".equals(paramType.getName())
849                 || "int".equals(paramType.getName())) {
850             try {
851                 result = new Integer(object);
852             } catch (NumberFormatException ex) {
853             }
854             // Try a setFoo ( boolean )
855         } else if ("java.lang.Boolean".equals(paramType.getName())
856                 || "boolean".equals(paramType.getName())) {
857             result = new Boolean(object);
858 
859             // Try a setFoo ( InetAddress )
860         } else if ("java.net.InetAddress".equals(paramType
861                 .getName())) {
862             try {
863                 result = InetAddress.getByName(object);
864             } catch (UnknownHostException exc) {
865                 debug("Unable to resolve host name:" + object);
866             }
867 
868             // Unknown type
869         } else {
870             debug("Unknown type " + paramType.getName());
871         }
872         if (result == null) {
873             throw new IllegalArgumentException("Can't convert argument: " + object);
874         }
875         return result;
876     }
