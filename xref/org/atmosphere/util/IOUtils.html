80          if (o != null && (String.class.isAssignableFrom(o.getClass()) && String.class.cast(o).isEmpty())
81                  || (Byte[].class.isAssignableFrom(o.getClass()) && Byte[].class.cast(o).length == 0)) {

88      public static StringBuilder readEntirelyAsString(AtmosphereResource r) {
89          final StringBuilder stringBuilder = new StringBuilder();
90          AtmosphereRequest request = r.getRequest();
91          if (request.body().isEmpty()) {
92              BufferedReader bufferedReader = null;
93              try {
94                  try {
95                      InputStream inputStream = request.getInputStream();
96                      if (inputStream != null) {
97                          bufferedReader = new BufferedReader(new InputStreamReader(inputStream));
98                      }
99                  } catch (IllegalStateException ex) {
100                     logger.trace("", ex);
101                     Reader reader = request.getReader();
102                     if (reader != null) {
103                         bufferedReader = new BufferedReader(reader);
104                     }
105                 }
106 
107                 if (bufferedReader != null) {
108                     char[] charBuffer = new char[8192];
109                     int bytesRead = -1;
110                     try {
111                         while ((bytesRead = bufferedReader.read(charBuffer)) > 0) {
112                             stringBuilder.append(charBuffer, 0, bytesRead);
113                         }
114                     } catch (NullPointerException ex) {
115                         // https://java.net/jira/browse/GRIZZLY-1676
116                     }
117                 } else {
118                     stringBuilder.append("");
119                 }
120             } catch (IOException ex) {
121                 logger.warn("", ex);
122             } finally {
123                 if (bufferedReader != null) {
124                     try {
125                         bufferedReader.close();
126                     } catch (IOException ex) {
127                         logger.warn("", ex);
128                     }
129                 }
130             }
131         } else {
132             AtmosphereRequest.Body body = request.body();
133             try {
134                 stringBuilder.append(body.hasString() ? body.asString() : new String(body.asBytes(), body.byteOffset(), body.byteLength(), request.getCharacterEncoding()));
135             } catch (UnsupportedEncodingException e) {
136                 logger.error("", e);
137             }
138         }
139         return stringBuilder;
140     }

142     public static byte[] readEntirelyAsByte(AtmosphereResource r) {
143         AtmosphereRequest request = r.getRequest();
144         AtmosphereRequest.Body body = request.body();
145         if (request.body().isEmpty()) {
146             BufferedInputStream bufferedStream = null;
147             ByteArrayOutputStream bbIS = new ByteArrayOutputStream();
148             try {
149                 try {
150                     InputStream inputStream = request.getInputStream();
151                     if (inputStream != null) {
152                         bufferedStream = new BufferedInputStream(inputStream);
153                     }
154                 } catch (IllegalStateException ex) {
155                     logger.trace("", ex);
156                     Reader reader = request.getReader();
157                     if (reader != null) {
158                         bufferedStream = new BufferedInputStream(new ReaderInputStream(reader));
159                     }
160                 }
161 
162                 if (bufferedStream != null) {
163                     byte[] bytes = new byte[8192];
164                     int bytesRead = 0;
165                     while (bytesRead != -1) {
166                         bytesRead = bufferedStream.read(bytes);
167                         if (bytesRead > 0)
168                             bbIS.write(bytes, 0, bytesRead);
169                     }
170 
171                 } else {
172                     bbIS.write("".getBytes());
173                 }
174             } catch (IOException ex) {
175                 logger.warn("", ex);
176             } finally {
177                 if (bufferedStream != null) {
178                     try {
179                         bufferedStream.close();
180                     } catch (IOException ex) {
181                         logger.warn("", ex);
182                     }
183                 }
184             }
185             return bbIS.toByteArray();
186         } else if (body.hasString()) {
187             try {
188                 return readEntirelyAsString(r).toString().getBytes(request.getCharacterEncoding());
189             } catch (UnsupportedEncodingException e) {
190                 logger.error("", e);
191             }
192         } else if (body.hasBytes()) {
193             return Arrays.copyOfRange(body.asBytes(), body.byteOffset(), body.byteLength());
194         }
195         throw new IllegalStateException("No body " + r);
196     }

211     /**
212      * Used to remove trailing slash and wildcard from a servlet path.<br/><br/>
213      * Examples :<br/>
214      * - "/foo/" becomes "/foo"<br/>
215      * - "foo/bar" becomes "/foo/bar"<br/>
216      *
217      * @param fullServletPath : Servlet mapping
218      * @return Servlet mapping without trailing slash and wildcard
219      */
220     public static String getCleanedServletPath(String fullServletPath) {
221         Matcher matcher = SERVLET_PATH_PATTERN.matcher(fullServletPath);
222 
223         // It should not happen if the servlet path is valid
224         if (!matcher.find()) return fullServletPath;
225 
226         String servletPath = matcher.group(0);
227         if (!servletPath.startsWith("/")) {
228             servletPath = "/" + servletPath;
229         }
230 
231         return servletPath;
232     }

234     private static boolean scanForAtmosphereFramework(Class<?> classToScan) {
235         if (classToScan == null) return false;
236 
237         logger.trace("Scanning {}", classToScan.getName());
238 
239         // Before doing the Siberian traversal, look locally
240         if (knownClasses.contains(classToScan.getName())) {
241             return true;
242         }
243 
244         try {
245             Field[] fields = classToScan.getDeclaredFields();
246             for (Field f : fields) {
247                 f.setAccessible(true);
248                 if (AtmosphereFramework.class.isAssignableFrom(f.getType())) {
249                     return true;
250                 }
251             }
252         } catch (Exception ex) {
253             logger.trace("", ex);
254         }
255 
256         // Now try with parent
257         if (scanForAtmosphereFramework(classToScan.getSuperclass())) return true;
258         return false;
259     }

283     /**
284      * <p>
285      * This method reads the given file stored under "META-INF/services" and accessed through the framework's class loader
286      * to specify a list of {@link org.atmosphere.cpr.AtmosphereFramework.MetaServiceAction actions} to be done on different
287      * service classes ({@link org.atmosphere.cpr.AtmosphereInterceptor}, {@link org.atmosphere.cpr.BroadcastFilter}, etc).
288      * </p>
289      * <p/>
290      * <p>
291      * The file content should follows the following format:
292      * <pre>
293      * INSTALL
294      * com.mycompany.MyInterceptor
295      * com.mycompany.MyFilter
296      * EXCLUDE
297      * org.atmosphere.interceptor.HeartbeatInterceptor
298      * </pre>
299      * </p>
300      * <p/>
301      * <p>
302      * If you don't specify any {@link org.atmosphere.cpr.AtmosphereFramework.MetaServiceAction} before a class, then
303      * default action will be {@link org.atmosphere.cpr.AtmosphereFramework.MetaServiceAction#INSTALL}.
304      * </p>
305      * <p/>
306      * <p>
307      * Important note: you must specify a class declared inside a package. Since creating classes in the source root is
308      * a bad practice, the method does not deal with it to improve its performances.
309      * </p>
310      *
311      * @param path the service file to read
312      * @return the map associating class to action
313      */
314     public static Map<String, AtmosphereFramework.MetaServiceAction> readServiceFile(final String path) {
315         final Map<String, AtmosphereFramework.MetaServiceAction> b = new LinkedHashMap<String, AtmosphereFramework.MetaServiceAction>();
316 
317         String line;
318         InputStream is = null;
319         BufferedReader reader = null;
320         AtmosphereFramework.MetaServiceAction action = AtmosphereFramework.MetaServiceAction.INSTALL;
321 
322         try {
323             is = AtmosphereFramework.class.getClassLoader().getResourceAsStream("META-INF/services/" + path);
324 
325             if (is == null) {
326                 logger.warn("META-INF/services/{} not found in class loader", path);
327                 return b;
328             }
329 
330             reader = new BufferedReader(new InputStreamReader(is));
331 
332             while (true) {
333                 line = reader.readLine();
334 
335                 if (line == null) {
336                     break;
337                 } else if (line.isEmpty()) {
338                     continue;
339                 } else if (line.indexOf('.') == -1) {
340                     action = AtmosphereFramework.MetaServiceAction.valueOf(line);
341                 } else {
342                     b.put(line, action);
343                 }
344             }
345         } catch (IOException e) {
346             logger.debug("Unable to read META-INF/services/{} from class loader", path, e);
347         } finally {
348             close(is, reader);
349         }
350 
351         return b;
352     }

354     /**
355      * <p>
356      * Tries to close the given objects and log the {@link IOException} at INFO level
357      * to make the code more readable when we assume that the {@link IOException} won't be managed.
358      * </p>
359      * <p/>
360      * <p>
361      * Also ignore {@code null} parameters.
362      * </p>
363      *
364      * @param closeableArray the objects to close
365      */
366     public static void close(final Closeable... closeableArray) {
367         for (Closeable closeable : closeableArray) {
368             try {
369                 if (closeable != null) {
370                     closeable.close();
371                 }
372             } catch (IOException ioe) {
373                 logger.info("Can't close the object", ioe);
374             }
375         }
376     }

378     public static String realPath(ServletContext servletContext, String targetPath) throws MalformedURLException {
379         String realPath = servletContext.getRealPath(targetPath);
380         if (realPath == null) {
381             URL u = servletContext.getResource(targetPath);
382             if (u != null) {
383                 realPath = u.getPath();
384             } else {
385                 return "";
386             }
387         }
388         return realPath;
