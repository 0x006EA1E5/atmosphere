47      public String buildURL(FocusPoint focusPoint) {
48  
49          int cookie = random.nextInt();
50          int randomValue = random.nextInt(2147483647) - 1;
51          long now = new Date().getTime();
52  
53          focusPoint.setParentTrackPoint(appFocusPoint);
54          StringBuffer url = new StringBuffer(TRACKING_URL_Prefix);
55          url.append("?utmwv=1"); //Urchin/Analytics version
56          url.append("&utmn=" + random.nextInt());
57          url.append("&utmcs=UTF-8"); //document encoding
58          url.append("&utmsr=1440x900"); //screen resolution
59          url.append("&utmsc=32-bit"); //color depth
60          url.append("&utmul=en-us"); //user language
61          url.append("&utmje=1"); //java enabled
62          url.append("&utmfl=9.0%20%20r28"); //flash
63          url.append("&utmcr=1"); //carriage return
64          url.append("&utmdt=" + focusPoint.getContentTitle()); //The optimum keyword density //document title
65          url.append("&utmhn=" + hostName);//document hostname
66          url.append("&utmr=" + refererURL); //referer URL
67          url.append("&utmp=" + focusPoint.getContentURI());//document page URL
68          url.append("&utmac=" + googleAnalyticsTrackingCode);//Google Analytics account
69          url.append("&utmcc=__utma%3D'" + cookie + "." + randomValue + "." + now + "." + now + "." + now + ".2%3B%2B__utmb%3D" + cookie + "%3B%2B__utmc%3D" + cookie + "%3B%2B__utmz%3D" + cookie + "." + now + ".2.2.utmccn%3D(direct)%7Cutmcsr%3D(direct)%7Cutmcmd%3D(none)%3B%2B__utmv%3D" + cookie);
70          return url.toString();
71      }
