16  package org.atmosphere.util.analytics;
17  
18  import java.net.InetAddress;
19  import java.net.UnknownHostException;
20  import java.util.Date;
21  import java.util.Random;
22  
23  /**
24   * Fork of https://code.google.com/p/jgoogleanalytics/
25   * <p/>
26   * URL building logic for the earlier versions of google analytics (urchin.js)
27   *
28   * @author : Siddique Hameed
29   * @version : 0.1
30   */
31  
32  public class GoogleAnalytics_v1_URLBuildingStrategy implements URLBuildingStrategy {
33      private FocusPoint appFocusPoint;
34      private String googleAnalyticsTrackingCode;
35      private String refererURL = "http://async-io.org";
36  
37      private static final String TRACKING_URL_Prefix = "http://www.google-analytics.com/__utm.gif";
38  
39      private static final Random random = new Random();
40      private static String hostName = "localhost";
41  
42      static {
43          try {
44              hostName = InetAddress.getLocalHost().getHostName();
45          } catch (UnknownHostException e) {
46              //ignore this
47          }
48      }
49  
50  
51      public GoogleAnalytics_v1_URLBuildingStrategy(String appName, String googleAnalyticsTrackingCode) {
52          this.googleAnalyticsTrackingCode = googleAnalyticsTrackingCode;
53          this.appFocusPoint = new FocusPoint(appName);
54      }
55  
56      public GoogleAnalytics_v1_URLBuildingStrategy(String appName, String appVersion, String googleAnalyticsTrackingCode) {
57          this.googleAnalyticsTrackingCode = googleAnalyticsTrackingCode;
58          this.appFocusPoint = new FocusPoint(appVersion, new FocusPoint(appName));
59      }
60  
61  
62      public String buildURL(FocusPoint focusPoint) {
63  
64          int cookie = random.nextInt();
65          int randomValue = random.nextInt(2147483647) - 1;
66          long now = new Date().getTime();
67  
68          focusPoint.setParentTrackPoint(appFocusPoint);
69          StringBuffer url = new StringBuffer(TRACKING_URL_Prefix);
70          url.append("?utmwv=1"); //Urchin/Analytics version
71          url.append("&utmn=" + random.nextInt());
72          url.append("&utmcs=UTF-8"); //document encoding
73          url.append("&utmsr=1440x900"); //screen resolution
74          url.append("&utmsc=32-bit"); //color depth
75          url.append("&utmul=en-us"); //user language
76          url.append("&utmje=1"); //java enabled
77          url.append("&utmfl=9.0%20%20r28"); //flash
78          url.append("&utmcr=1"); //carriage return
79          url.append("&utmdt=" + focusPoint.getContentTitle()); //The optimum keyword density //document title
80          url.append("&utmhn=" + hostName);//document hostname
81          url.append("&utmr=" + refererURL); //referer URL
82          url.append("&utmp=" + focusPoint.getContentURI());//document page URL
83          url.append("&utmac=" + googleAnalyticsTrackingCode);//Google Analytics account
84          url.append("&utmcc=__utma%3D'" + cookie + "." + randomValue + "." + now + "." + now + "." + now + ".2%3B%2B__utmb%3D" + cookie + "%3B%2B__utmc%3D" + cookie + "%3B%2B__utmz%3D" + cookie + "." + now + ".2.2.utmccn%3D(direct)%7Cutmcsr%3D(direct)%7Cutmcmd%3D(none)%3B%2B__utmv%3D" + cookie);
85          return url.toString();
86      }
87  
88      public void setRefererURL(String refererURL) {
89          this.refererURL = refererURL;
90      }
91  }
