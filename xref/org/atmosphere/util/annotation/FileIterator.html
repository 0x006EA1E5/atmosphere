39  package org.atmosphere.util.annotation;
40  
41  import java.io.File;
42  import java.io.IOException;
43  import java.util.Deque;
44  import java.util.LinkedList;
45  import java.util.NoSuchElementException;
46  
47  /**
48   * {@code FileIterator} enables iteration over all files in a directory and all
49   * its sub directories.
50   * <br/>
51   * Usage:
52   * <pre>
53   * FileIterator iter = new FileIterator(new File("./src"));
54   * File f;
55   * while ((f = iter.next()) != null) {
56   *     // do something with f
57   *     assert f == iter.getCurrent();
58   * }
59   * </pre>
60   * 
61   * @author <a href="mailto:rmuller@xiam.nl">Ronald K. Muller</a>
62   * @since annotation-detector 3.0.0
63   */
64  public final class FileIterator {
65  
66      private final Deque<File> stack = new LinkedList<File>();
67      private int rootCount;
68      private File current;
69      
70      /**
71       * Create a new {@code FileIterator} using the specified 'filesOrDirectories' as root.
72       * <br/>
73       * If 'filesOrDirectories' contains a file, the iterator just returns that single file. 
74       * If 'filesOrDirectories' contains a directory, all files in that directory 
75       * and its sub directories are returned (depth first).
76       * 
77       * @param filesOrDirectories Zero or more {@link java.io.File} objects, which are iterated
78       * in the specified order (depth first)
79       */
80      public FileIterator(final File... filesOrDirectories) {
81          addReverse(filesOrDirectories);
82          rootCount = stack.size();
83      }
84  
85      /**
86       * Return the last returned file or {@code null} if no more files are available.
87       *
88       * @see #next()
89       */
90      public File getFile() {
91          return current;
92      }
93  
94      /**
95       * Return {@code true} if the current file is one of the files originally
96       * specified as one of the constructor file parameters, i.e. is a root file
97       * or directory.
98       */
99      public boolean isRootFile() {
100         if (current == null) {
101             throw new NoSuchElementException();
102         }
103         return stack.size() < rootCount;
104     }
105 
106     /**
107      * Return the next {@link java.io.File} object or {@code null} if no more files are
108      * available.
109      * 
110      * @see #getFile()
111      */
112     public File next() throws IOException {
113         if (stack.isEmpty()) {
114             current = null;
115             return null;
116         } else {
117             current = stack.removeLast();
118             if (current.isDirectory()) {
119                 if (stack.size() < rootCount) {
120                     rootCount = stack.size();
121                 }
122                 addReverse(current.listFiles());
123                 return next();
124             } else {
125                 return current;
126             }
127         }
128     }
129 
130     /**
131      * Add the specified files in reverse order.
132      */
133     private void addReverse(final File[] files) {
134         for (int i = files.length - 1; i >=0; --i) {
135             stack.add(files[i]);
136         }
137     }
138 
139 }
