43  /**
44   * Splits an HTTP query string into a path string and key-value parameter pairs.
45   * This decoder is for one time use only.  Create a new instance for each URI:
46   * <pre>
47   * {@link QueryStringDecoder} decoder = new {@link QueryStringDecoder}("/hello?recipient=world&x=1;y=2");
48   * assert decoder.getPath().equals("/hello");
49   * assert decoder.getParameters().get("recipient").get(0).equals("world");
50   * assert decoder.getParameters().get("x").get(0).equals("1");
51   * assert decoder.getParameters().get("y").get(0).equals("2");
52   * </pre>
53   *
54   * This decoder can also decode the content of an HTTP POST request whose
55   * content type is <tt>application/x-www-form-urlencoded</tt>:
56   * <pre>
57   * {@link QueryStringDecoder} decoder = new {@link QueryStringDecoder}("recipient=world&x=1;y=2", false);
58   * ...
59   * </pre>
60   *
61   * <h3>HashDOS vulnerability fix</h3>
62   *
63   * As a workaround to the <a href="http://goo.gl/I4Nky">HashDOS</a> vulnerability, the decoder
64   * limits the maximum number of decoded key-value parameter pairs, up to {@literal 1024} by
65   * default, and you can configure it when you construct the decoder by passing an additional
66   * integer parameter.
67   *
68   * @apiviz.stereotype utility
69   * @apiviz.has        org.jboss.netty.handler.codec.http.HttpRequest oneway - - decodes
70   */
71  public class QueryStringDecoder {
72  
73      private static final int DEFAULT_MAX_PARAMS = 1024;
74      public static final Charset UTF_8 = Charset.forName("UTF-8");

78      private final boolean hasPath;
79      private final int maxParams;
80      private String path;
81      private Map<String, List<String>> params;

116     /**
117      * Creates a new decoder that decodes the specified URI encoded in the
118      * specified charset.
119      */
120     public QueryStringDecoder(String uri, Charset charset, boolean hasPath, int maxParams) {
121         if (uri == null) {
122             throw new NullPointerException("uri");
123         }
124         if (charset == null) {
125             throw new NullPointerException("charset");
126         }
127         if (maxParams <= 0) {
128             throw new IllegalArgumentException(
129                     "maxParams: " + maxParams + " (expected: a positive integer)");
130         }
131 
132         // http://en.wikipedia.org/wiki/Query_string
133         this.uri = uri.replace(';', '&');
134         this.charset = charset;
135         this.maxParams = maxParams;
136         this.hasPath = hasPath;
137     }

163     /**
164      * Creates a new decoder that decodes the specified URI encoded in the
165      * specified charset.
166      */
167     public QueryStringDecoder(URI uri, Charset charset, int maxParams) {
168         if (uri == null) {
169             throw new NullPointerException("uri");
170         }
171         if (charset == null) {
172             throw new NullPointerException("charset");
173         }
174         if (maxParams <= 0) {
175             throw new IllegalArgumentException(
176                     "maxParams: " + maxParams + " (expected: a positive integer)");
177         }
178 
179         String rawPath = uri.getRawPath();
180         if (rawPath != null) {
181             hasPath = true;
182         } else {
183             rawPath = "";
184             hasPath = false;
185         }
186         // Also take care of cut of things like "http://localhost"
187         String newUri = rawPath + "?" + uri.getRawQuery();
188 
189         // http://en.wikipedia.org/wiki/Query_string
190         this.uri = newUri.replace(';', '&');
191         this.charset = charset;
192         this.maxParams = maxParams;
193 
194     }

204     /**
205      * Returns the decoded path string of the URI.
206      */
207     public String getPath() {
208         if (path == null) {
209             if (!hasPath) {
210                 return path = "";
211             }
212 
213             int pathEndPos = uri.indexOf('?');
214             if (pathEndPos < 0) {
215                 path = uri;
216             } else {
217                 return path = uri.substring(0, pathEndPos);
218             }
219         }
220         return path;
221     }
222 
223     /**
224      * Returns the decoded key-value parameter pairs of the URI.
225      */
226     public Map<String, List<String>> getParameters() {
227         if (params == null) {
228             if (hasPath) {
229                 int pathLength = getPath().length();
230                 if (uri.length() == pathLength) {
231                     return Collections.emptyMap();
232                 }
233                 decodeParams(uri.substring(pathLength + 1));
234             } else {
235                 if (uri.length() == 0) {
236                     return Collections.emptyMap();
237                 }
238                 decodeParams(uri);
239             }
240         }
241         return params;
242     }
243 
244     private void decodeParams(String s) {
245         Map<String, List<String>> params = this.params = new LinkedHashMap<String, List<String>>();
246         nParams = 0;
247         String name = null;
248         int pos = 0; // Beginning of the unprocessed region
249         int i;       // End of the unprocessed region
250         char c = 0;  // Current character
251         for (i = 0; i < s.length(); i++) {
252             c = s.charAt(i);
253             if (c == '=' && name == null) {
254                 if (pos != i) {
255                     name = decodeComponent(s.substring(pos, i), charset);
256                 }
257                 pos = i + 1;
258             } else if (c == '&') {
259                 if (name == null && pos != i) {
260                     // We haven't seen an `=' so far but moved forward.
261                     // Must be a param of the form '&a&' so add it with
262                     // an empty value.
263                     if (!addParam(params, decodeComponent(s.substring(pos, i), charset), "")) {
264                         return;
265                     }
266                 } else if (name != null) {
267                     if (!addParam(params, name, decodeComponent(s.substring(pos, i), charset))) {
268                         return;
269                     }
270                     name = null;
271                 }
272                 pos = i + 1;
273             }
274         }
275 
276         if (pos != i) {  // Are there characters we haven't dealt with?
277             if (name == null) {     // Yes and we haven't seen any `='.
278                 if (!addParam(params, decodeComponent(s.substring(pos, i), charset), "")) {
279                     return;
280                 }
281             } else {                // Yes and this must be the last value.
282                 if (!addParam(params, name, decodeComponent(s.substring(pos, i), charset))) {
283                     return;
284                 }
285             }
286         } else if (name != null) {  // Have we seen a name without value?
287             if (!addParam(params, name, "")) {
288                 return;
289             }
290         }
291     }
292 
293     private boolean addParam(Map<String, List<String>> params, String name, String value) {
294         if (nParams >= maxParams) {
295             return false;
296         }
297 
298         List<String> values = params.get(name);
299         if (values == null) {
300             values = new ArrayList<String>(1);  // Often there's only 1 value.
301             params.put(name, values);
302         }
303         values.add(value);
304         nParams ++;
305         return true;
306     }

308     /**
309      * Decodes a bit of an URL encoded by a browser.
310      * <p>
311      * This is equivalent to calling {@link #decodeComponent(String, Charset)}
312      * with the UTF-8 charset (recommended to comply with RFC 3986, Section 2).
313      * @param s The string to decode (can be empty).
314      * @return The decoded string, or {@code s} if there's nothing to decode.
315      * If the string to decode is {@code null}, returns an empty string.
316      * @throws IllegalArgumentException if the string contains a malformed
317      * escape sequence.
318      */
319     public static String decodeComponent(final String s) {
320         return decodeComponent(s, UTF_8);
321     }
322 
323     /**
324      * Decodes a bit of an URL encoded by a browser.
325      * <p>
326      * The string is expected to be encoded as per RFC 3986, Section 2.
327      * This is the encoding used by JavaScript functions {@code encodeURI}
328      * and {@code encodeURIComponent}, but not {@code escape}.  For example
329      * in this encoding, &eacute; (in Unicode {@code U+00E9} or in UTF-8
330      * {@code 0xC3 0xA9}) is encoded as {@code %C3%A9} or {@code %c3%a9}.
331      * <p>
332      * This is essentially equivalent to calling
333      *   <code>{@link URLDecoder#decode(String, String) URLDecoder.decode}(s, charset.name())</code>
334      * except that it's over 2x faster and generates less garbage for the GC.
335      * Actually this function doesn't allocate any memory if there's nothing
336      * to decode, the argument itself is returned.
337      * @param s The string to decode (can be empty).
338      * @param charset The charset to use to decode the string (should really
339      * be {@link #UTF_8}.
340      * @return The decoded string, or {@code s} if there's nothing to decode.
341      * If the string to decode is {@code null}, returns an empty string.
342      * @throws IllegalArgumentException if the string contains a malformed
343      * escape sequence.
344      */
345     @SuppressWarnings("fallthrough")
346     public static String decodeComponent(final String s,
347                                          final Charset charset) {
348         if (s == null) {
349             return "";
350         }
351         final int size = s.length();
352         boolean modified = false;
353         for (int i = 0; i < size; i++) {
354             final char c = s.charAt(i);
355             switch (c) {
356                 case '%':
357                     i++;  // We can skip at least one char, e.g. `%%'.
358                     // Fall through.
359                 case '+':
360                     modified = true;
361                     break;
362             }
363         }
364         if (!modified) {
365             return s;
366         }
367         final byte[] buf = new byte[size];
368         int pos = 0;  // position in `buf'.
369         for (int i = 0; i < size; i++) {
370             char c = s.charAt(i);
371             switch (c) {
372                 case '+':
373                     buf[pos++] = ' ';  // "+" -> " "
374                     break;
375                 case '%':
376                     if (i == size - 1) {
377                         throw new IllegalArgumentException("unterminated escape"
378                                 + " sequence at end of string: " + s);
379                     }
380                     c = s.charAt(++i);
381                     if (c == '%') {
382                         buf[pos++] = '%';  // "%%" -> "%"
383                         break;
384                     } else if (i == size - 1) {
385                         throw new IllegalArgumentException("partial escape"
386                                 + " sequence at end of string: " + s);
387                     }
388                     c = decodeHexNibble(c);
389                     final char c2 = decodeHexNibble(s.charAt(++i));
390                     if (c == Character.MAX_VALUE || c2 == Character.MAX_VALUE) {
391                         throw new IllegalArgumentException(
392                                 "invalid escape sequence `%" + s.charAt(i - 1)
393                                 + s.charAt(i) + "' at index " + (i - 2)
394                                 + " of: " + s);
395                     }
396                     c = (char) (c * 16 + c2);
397                     // Fall through.
398                 default:
399                     buf[pos++] = (byte) c;
400                     break;
401             }
402         }
403         try {
404             return new String(buf, 0, pos, charset.name());
405         } catch (UnsupportedEncodingException e) {
406             throw new IllegalArgumentException("unsupported encoding: " + charset.name());
407         }
408     }

410     /**
411      * Helper to decode half of a hexadecimal number from a string.
412      * @param c The ASCII character of the hexadecimal number to decode.
413      * Must be in the range {@code [0-9a-fA-F]}.
414      * @return The hexadecimal value represented in the ASCII character
415      * given, or {@link Character#MAX_VALUE} if the character is invalid.
416      */
417     private static char decodeHexNibble(final char c) {
418         if ('0' <= c && c <= '9') {
419             return (char) (c - '0');
420         } else if ('a' <= c && c <= 'f') {
421             return (char) (c - 'a' + 10);
422         } else if ('A' <= c && c <= 'F') {
423             return (char) (c - 'A' + 10);
424         } else {
425             return Character.MAX_VALUE;
426         }
427     }
