16  package org.atmosphere.jetty;
17  
18  import org.atmosphere.cpr.AtmosphereFramework;
19  import org.atmosphere.cpr.AtmosphereRequest;
20  import org.atmosphere.websocket.WebSocketEventListener;
21  import org.atmosphere.websocket.WebSocketProcessor;
22  import org.atmosphere.websocket.WebSocketProtocol;
23  import org.eclipse.jetty.websocket.WebSocket;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import java.io.UnsupportedEncodingException;
28  
29  import static org.atmosphere.websocket.WebSocketEventListener.WebSocketEvent.TYPE.DISCONNECT;
30  import static org.atmosphere.websocket.WebSocketEventListener.WebSocketEvent.TYPE.MESSAGE;
31  
32  /**
33   * Jetty 7 & 8 WebSocket support.
34   */
35  public class JettyWebSocketDraft08Handler implements WebSocket {
36  
37      private static final Logger logger = LoggerFactory.getLogger(JettyWebSocketDraft08Handler.class);
38  
39      private WebSocketProcessor webSocketProcessor;
40      private final AtmosphereRequest request;
41      private final AtmosphereFramework framework;
42      private final WebSocketProtocol webSocketProtocol;
43  
44      public JettyWebSocketDraft08Handler(AtmosphereRequest request, AtmosphereFramework framework, WebSocketProtocol webSocketProtocol) {
45          this.request = request;
46          this.framework = framework;
47          this.webSocketProtocol = webSocketProtocol;
48      }
49  
50      @Override
51      public void onConnect(org.eclipse.jetty.websocket.WebSocket.Outbound outbound) {
52  
53          logger.debug("WebSocket.onConnect (outbound)");
54          try {
55              webSocketProcessor = new WebSocketProcessor(framework, new JettyWebSocket(outbound), webSocketProtocol);
56              webSocketProcessor.dispatch(request);
57          } catch (Exception e) {
58              logger.warn("failed to connect to web socket", e);
59          }
60      }
61  
62      @Override
63      public void onMessage(byte frame, String data) {
64          logger.trace("WebSocket.onMessage (frame/string)");
65          webSocketProcessor.invokeWebSocketProtocol(data);
66          webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent(data, MESSAGE, webSocketProcessor.webSocket()));
67      }
68  
69      @Override
70      public void onMessage(byte frame, byte[] data, int offset, int length) {
71          logger.trace("WebSocket.onMessage (frame)");
72          webSocketProcessor.invokeWebSocketProtocol(new String(data, offset, length));
73          try {
74              webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent(new String(data, offset, length, "UTF-8"), MESSAGE, webSocketProcessor.webSocket()));
75          } catch (UnsupportedEncodingException e) {
76              logger.warn("UnsupportedEncodingException", e);
77  
78          }
79      }
80  
81      @Override
82      public void onFragment(boolean more, byte opcode, byte[] data, int offset, int length) {
83          logger.trace("WebSocket.onFragment");
84          webSocketProcessor.invokeWebSocketProtocol(new String(data, offset, length));
85          try {
86              webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent(new String(data, offset, length, "UTF-8"), MESSAGE, webSocketProcessor.webSocket()));
87          } catch (UnsupportedEncodingException e) {
88              logger.warn("UnsupportedEncodingException", e);
89  
90          }
91      }
92  
93      @Override
94      public void onDisconnect() {
95          request.destroy();
96          logger.trace("WebSocket.onDisconnect");
97          webSocketProcessor.close(1000);
98          webSocketProcessor.notifyListener(new WebSocketEventListener.WebSocketEvent("", DISCONNECT, webSocketProcessor.webSocket()));
99      }
100 
101 }
