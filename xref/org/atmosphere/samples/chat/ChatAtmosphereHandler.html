50  import javax.servlet.http.HttpServletRequest;
51  import javax.servlet.http.HttpServletResponse;
52  import java.io.IOException;

58  /**
59   * Simple AtmosphereHandler that implement the logic to build a Chat application.
60   *
61   * @author Jeanfrancois Arcand
62   * @author TAKAI Naoto (original author for the Comet based Chat).
63   */
64  public class ChatAtmosphereHandler implements AtmosphereHandler<HttpServletRequest, HttpServletResponse> {
65  
66      private static final Logger logger = LoggerFactory.getLogger(ChatAtmosphereHandler.class);
67  
68      private final static String BEGIN_SCRIPT_TAG = "<script type='text/javascript'>\n";
69      private final static String END_SCRIPT_TAG = "</script>\n";
70      private final static long serialVersionUID = -2919167206889576860L;
71  
72      private final static String CLUSTER = "org.atmosphere.useCluster";
73  
74      private AtomicBoolean filterAdded = new AtomicBoolean(false);
75  
76      public ChatAtmosphereHandler() {
77      }
78  
79      /**
80       * When the {@link AtmosphereServlet} detect an {@link HttpServletRequest}
81       * maps to this {@link AtmosphereHandler}, the  {@link AtmosphereHandler#onRequest}
82       * gets invoked and the response will be suspended depending on the http
83       * method, e.g. GET will suspend the connection, POST will broadcast chat
84       * message to suspended connection.
85       *
86       * @param event An {@link AtmosphereResource}
87       * @throws java.io.IOException
88       */
89      public void onRequest(AtmosphereResource<HttpServletRequest,
90              HttpServletResponse> event) throws IOException {
91  
92          HttpServletRequest req = event.getRequest();
93          HttpServletResponse res = event.getResponse();
94  
95          res.setContentType("text/html;charset=ISO-8859-1");
96          if (req.getMethod().equalsIgnoreCase("GET")) {
97              event.suspend();
98  
99              Broadcaster bc = event.getBroadcaster();
100             String clusterType = event.getAtmosphereConfig().getInitParameter(CLUSTER);
101             if (!filterAdded.getAndSet(true) && clusterType != null) {
102                 if (clusterType.equals("jgroups")) {
103                     event.getAtmosphereConfig().getServletContext().log("JGroupsFilter enabled");
104                     bc.getBroadcasterConfig().addFilter(
105                             new JGroupsFilter(bc));
106                 }
107             }
108 
109             //Simple Broadcast
110             bc.getBroadcasterConfig().addFilter(new XSSHtmlFilter());
111             Future<String> f = bc.broadcast(event.getAtmosphereConfig().getWebServerName()
112                     + "**has suspended a connection from "
113                     + req.getRemoteAddr());
114 
115             try {
116                 // Wait for the push to occurs. This is blocking a thread as the Broadcast operation
117                 // is usually asynchronous, e.g executed using an {@link ExecuturService}
118                 f.get();
119             } catch (InterruptedException ex) {
120                 logger.error("", ex);
121             } catch (ExecutionException ex) {
122                 logger.error("", ex);
123             }
124 
125             // Ping the connection every 30 seconds
126             bc.scheduleFixedBroadcast(req.getRemoteAddr() + "**is still listening", 30, TimeUnit.SECONDS);
127 
128             // Delay a message until the next broadcast.
129             bc.delayBroadcast("Delayed Chat message");
130         } else if (req.getMethod().equalsIgnoreCase("POST")) {
131             String action = req.getParameterValues("action")[0];
132             String name = req.getParameterValues("name")[0];
133 
134             if ("login".equals(action)) {
135                 req.getSession().setAttribute("name", name);
136                 event.getBroadcaster().broadcast("System Message from "
137                         + event.getAtmosphereConfig().getWebServerName() + "**" + name + " has joined.");
138 
139             } else if ("post".equals(action)) {
140                 String message = req.getParameterValues("message")[0];
141                 event.getBroadcaster().broadcast(name + "**" + message);
142             } else {
143                 res.setStatus(422);
144             }
145             res.getWriter().write("success");
146             res.getWriter().flush();
147         }
148     }
149 
150     /**
151      * Invoked when a call to {@link Broadcaster#broadcast(java.lang.Object)} is
152      * issued or when the response times out, e.g whne the value
153      * {@link AtmosphereResource#suspend(long)}
154      * expires.
155      *
156      * @param event An {@link AtmosphereResourceEvent}
157      * @throws java.io.IOException
158      */
159     public void onStateChange(AtmosphereResourceEvent<HttpServletRequest,
160             HttpServletResponse> event) throws IOException {
161 
162         HttpServletRequest req = event.getResource().getRequest();
163         HttpServletResponse res = event.getResource().getResponse();
164 
165         if (event.getMessage() == null) return;
166 
167         String e = event.getMessage().toString();
168 
169         String name = e;
170         String message = "";
171 
172         if (e.indexOf("**") > 0) {
173             name = e.substring(0, e.indexOf("**"));
174             message = e.substring(e.indexOf("**") + 2);
175         }
176 
177         String msg = BEGIN_SCRIPT_TAG + toJsonp(name, message) + END_SCRIPT_TAG;
178 
179         if (event.isCancelled()) {
180             event.getResource().getBroadcaster()
181                     .broadcast(req.getSession().getAttribute("name") + " has left");
182         } else if (event.isResuming() || event.isResumedOnTimeout()) {
183             String script = "<script>window.parent.app.listen();\n</script>";
184 
185             res.getWriter().write(script);
186         } else {
187             res.getWriter().write(msg);
188         }
189         res.getWriter().flush();
190     }
191 
192     private String toJsonp(String name, String message) {
193         return "window.parent.app.update({ name: \"" + name + "\", message: \""
194                 + message + "\" });\n";
195     }

198     }
