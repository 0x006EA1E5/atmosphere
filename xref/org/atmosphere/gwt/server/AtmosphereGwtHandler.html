73      /**
74       * This is the main entrypoint on the server that you will want to hook into and override.
75       * This method is called when a client has request a new connection. Best practice is to do all your
76       * required setup here and tie the AtmosphereResource to a Broadcaster, but do not send anything to
77       * the client yet. If you wish to do so it is best to let the client send a notification to the server
78       * using the {@link AtmosphereClient#post} method in the onConnected event.
79       * 
80       * @param resource
81       * @return
82       * @throws ServletException
83       * @throws IOException 
84       */
85      public int doComet(GwtAtmosphereResource resource) throws ServletException, IOException {
86          Broadcaster broadcaster = BroadcasterFactory.getDefault().lookup(Broadcaster.class, GWT_BROADCASTER_ID);
87          if (broadcaster == null) {
88              broadcaster = BroadcasterFactory.getDefault().get(DefaultBroadcaster.class, GWT_BROADCASTER_ID);
89          }
90          resource.getAtmosphereResource().setBroadcaster(broadcaster);
91          return NO_TIMEOUT;
92      }

125      * You will need to pass the connectionID, which you can pass as an url parameter {getConnectionID()} or
126      * directly in your remote call
127      *
128      * @param connectionId
129      * @return

131     protected GwtAtmosphereResource lookupResource(int connectionId) {
132         if (resources == null) {
133             return null;
134         }
135         GwtAtmosphereResource r = resources.get(connectionId);
136         if (r != null) {
137             return r;
138         } else {
139             logger.info("Failed to find resource for [" + connectionId + "]");
140         }
141         return null;
142     }

195     protected void reapResources() {
196         if (resources != null) {
197             for (GwtAtmosphereResource resource : resources.values()) {
198                 if (!resource.isAlive()) {
199                     resources.remove(resource.getConnectionID());
200                 }
201             }
202         }

205     @Override
206     public void onRequest(AtmosphereResource resource) throws IOException {
207 
208         HttpServletRequest request = resource.getRequest();
209 
210         String servertransport = request.getParameter("servertransport");
211         Object webSocketSubProtocol = resource.getRequest().getAttribute(FrameworkConfig.WEBSOCKET_SUBPROTOCOL);
212         if ("rpcprotocol".equals(servertransport)) {
213 
214             Integer connectionID = Integer.parseInt(request.getParameter("connectionID"));
215             doServerMessage(request, resource.getResponse(), connectionID);
216             return;
217 
218         } else if (webSocketSubProtocol != null
219                   && webSocketSubProtocol.equals(FrameworkConfig.SIMPLE_HTTP_OVER_WEBSOCKET)) {
220 
221             Integer connectionID = (Integer) request.getAttribute(AtmosphereGwtHandler.class.getName()
222                                         + ".connectionID");
223             doServerMessage(request, resource.getResponse(), connectionID);
224             return;
225         }
226 
227         try {
228             int requestHeartbeat = getRequestedHeartbeat(request);
229             boolean requestEscapeText = getRequestedEscapeOfText(request);
230 
231             GwtAtmosphereResourceImpl resourceWrapper =
232                     new GwtAtmosphereResourceImpl(resource, this, requestHeartbeat, requestEscapeText);
233             request.setAttribute(AtmosphereGwtHandler.class.getName() + ".connectionID",
234                     (Integer) resourceWrapper.getConnectionID());
235             doCometImpl(resourceWrapper);
236         } catch (IOException e) {
237 //            GwtAtmosphereResourceImpl resource = new GwtAtmosphereResourceImpl(atm, this, -1);
238             logger.error("Unable to initiated comet" + e.getMessage(), e);
239 //			resource.getResponseWriter().sendError(HttpServletResponse.SC_BAD_REQUEST, e.getMessage());
240         }

244         boolean requestEscapeText = this.escapeText;

252     private int getRequestedHeartbeat(HttpServletRequest request) throws IOException {
253         int requestHeartbeat = heartbeat;
254         String requestedHeartbeat = request.getParameter("heartbeat");
255         if (requestedHeartbeat != null) {
256             try {
257                 requestHeartbeat = Integer.parseInt(requestedHeartbeat);
258                 if (requestHeartbeat <= 0) {
259                     throw new IOException("invalid heartbeat parameter");
260                 }
261                 requestHeartbeat = computeHeartbeat(requestHeartbeat);
262             } catch (NumberFormatException e) {
263                 throw new IOException("invalid heartbeat parameter");
264             }
265         }
266         return requestHeartbeat;
267     }

271     protected void doServerMessage(HttpServletRequest request, HttpServletResponse response, int connectionID)
272         throws IOException{
273         BufferedReader data = request.getReader();
274         List<Object> postMessages = new ArrayList<Object>();
275         GwtAtmosphereResource resource = lookupResource(connectionID);
276         if (resource == null) {
277             return;
278         }
279         String mode = resource.getRequest().getParameter(Constants.CLIENT_SERIALZE_MODE_PARAMETER);
280         SerialMode serialMode;
281         if (mode != null) {
282             serialMode = SerialMode.valueOf(mode);
283         } else {
284             serialMode = SerialMode.RPC;
285         }
286 
287         try {
288             while (true) {
289                 String event = data.readLine();
290                 if (event == null) {
291                     break;
292                 }
293                 String action = data.readLine();
294 
295                 if (logger.isTraceEnabled()) {
296                     logger.trace("[" + connectionID + "] Server message received: " + event + ";" + action);
297                 }
298                 if (event.equals("o") || event.equals("s")) {
299                     int length = Integer.parseInt(data.readLine());
300                     char[] messageData = new char[length];
301                     int totalRead = 0;
302                     int read = 0;
303                     while ((read = data.read(messageData, totalRead, length - totalRead)) != -1) {
304                         totalRead += read;
305                         if (totalRead == length) {
306                             break;
307                         }
308                     }
309                     if (totalRead != length) {
310                         throw new IllegalStateException("Corrupt message received");
311                     }
312                     Object message = null;
313                     if (event.equals("o")) {
314                         try {
315                             message = deserialize(messageData, serialMode);
316                         } catch (SerializationException ex) {
317                             logger.error("Failed to deserialize message", ex);
318                         }
319                     } else {
320                         message = String.copyValueOf(messageData);
321                     }
322                     if (message != null) {
323                         if (action.equals("p")) {
324                             postMessages.add(message);
325                         } else if (action.equals("b")) {
326                             broadcast(message, resource);
327                         }
328                     }
329                 } else if (event.equals("c")) {
330                     if (action.equals("d")) {
331                         disconnect(resource);
332                     }
333                 }
334             }
335         } catch (IOException ex) {
336             logger.error("[" + connectionID + "] Failed to read", ex);
337         }
338 
339         if (postMessages.size() > 0) {
340             post(request, response, postMessages, resource);
341         }
342     }

352     protected Object deserialize(char[] data, SerialMode mode) throws SerializationException {
353         return deserialize(String.copyValueOf(data), mode);

355     protected Object deserialize(String data, SerialMode mode) throws SerializationException {
356         switch (mode) {
357             default:
358             case RPC:
359                 return getGwtRpc().deserialize(data);
360                 
361             case JSON:
362                 return getJSONDeserializer().deserialize(data);
363             case PLAIN:
364                 return data;
365         }
366     }

447     private void doCometImpl(GwtAtmosphereResourceImpl resource) throws IOException {
448 
449         try {
450             // setup the request
451             resource.getWriterImpl().initiate();
452             if (resources == null) {
453                 resources = new ConcurrentHashMap<Integer, GwtAtmosphereResource>(5);
454                 scheduler.scheduleWithFixedDelay(new Runnable() {
455                     @Override
456                     public void run() {
457                         reapResources();
458                     }
459                 }, 30, 10, TimeUnit.SECONDS);
460             }
461             resources.put(resource.getConnectionID(), resource);
462         } catch (IOException e) {
463             logger.error("Error initiating GwtComet", e);
464             return;
465         }
466 
467         int timeout;
468         try {
469             // call the application code
470             timeout = doComet(resource);
471             if (timeout == -1) {
472                 logger.info("You have set an infinite timeout for your comet connection this is not recommended");
473             }
474         } catch (ServletException e) {
475             logger.error("Error calling doComet()", e);
476 //			resource.getResponseWriter().sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e.getMessage());
477             return;
478         } catch (IOException e) {
479             logger.error("Error calling doComet()", e);
480 //			resource.getResponseWriter().sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e.getMessage());
481             return;
482         }
483 
484         // at this point the application may have spawned threads to process this response
485         // so we have to be careful about concurrency from here on
486         resource.suspend(timeout);
487     }
