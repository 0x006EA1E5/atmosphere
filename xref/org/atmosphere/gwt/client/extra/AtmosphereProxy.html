104     protected void initialize() {
105         if (WindowSocket.exists(Window.current(), EVENT_SOCKET)) {
106             throw new IllegalStateException("Only one AtmosphereProxy instance is allowed per window");
107         }
108         
109         eventSocket = new WindowSocket();
110         eventSocket.addHandler(new WindowSocket.MessageHandler() {
111           @Override
112           public void onMessage(Window source, String message) {
113                 AtmosphereProxyEvent event = deserialize(message);
114                 onEvent(source, event, message);
115         }});
116         eventSocket.bind(EVENT_SOCKET);
117         Window current = Window.current();
118         
119         if (current.opener() != null && current != current.opener() 
120                 && WindowSocket.exists(current.opener(), EVENT_SOCKET)) {
121             logger.info("Connecting new client window");
122             parent = current.opener();
123             dispatchEvent(parent, event(EventType.ANNOUNCE_NEW_CHILD));
124         } else {
125             logger.info("Starting master connection");
126             masterConnection = new AtmosphereClient(url, serializer, masterListener);
127             masterConnection.start();
128             parent = null;
129         }
130         
131         com.google.gwt.user.client.Window.addWindowClosingHandler(new ClosingHandler() {
132             @Override
133             public void onWindowClosing(ClosingEvent event) {
134                 AtmosphereProxy.this.onWindowClosing(event);
135             }
136         });
137     }

139     protected void onWindowClosing(ClosingEvent event) {
140         if (parent != null) {
141             dispatchEvent(parent, event(EventType.ANNOUNCE_CHILD_DEATH));
142         }
143         if (windowList.size() > 0) {
144             Window fosterParent = parent != null ? parent : windowList.get(0);
145             JSONArray arr = new JSONArray();
146             List<Window> orphans;
147             if (parent == fosterParent) {
148                 orphans = windowList;
149             } else {
150                 orphans = windowList.subList(1, windowList.size());
151             }
152             int i = 0;
153             for (Window w : orphans) {
154                 arr.set(i++, new JSONObject(w));
155             }
156             if (arr.size() > 0) {
157                 fosterParent.set("orphans", arr.getJavaScriptObject());
158             }
159             if (masterConnection != null) {
160                 dispatchEvent(fosterParent, event(EventType.ELECT_MASTER));
161                 masterConnection.stop();
162                 masterConnection = null;
163             } else if (arr.size() > 0){
164                 dispatchEvent(fosterParent, event(EventType.ADOPT_ORPHANS));
165             }
166         }

169     protected void onEvent(Window source, AtmosphereProxyEvent event, String rawEvent) {
170         switch (event.getEventType()) {
171             // do not bubble these events
172             // they are specificly meant for us
173             case ANNOUNCE_NEW_CHILD:
174                 logger.fine("Adding new child");
175                 windowList.add(source);
176                 return;
177             case ANNOUNCE_CHILD_DEATH:
178                 logger.fine("Removing child");
179                 windowList.remove(source);
180                 return;
181             case ANNOUNCE_NEW_PARENT:
182                 logger.fine("Set new parent");
183                 parent = source;
184                 return;
185             case ELECT_MASTER:
186                 logger.fine("Become new master");
187                 parent = null;
188                 masterConnection = new AtmosphereClient(url, serializer, masterListener);
189                 // same code for adoption
190             case ADOPT_ORPHANS:
191                 JavaScriptObject orphanArray = Window.current().getObject("orphans");
192                 if (orphanArray != null) {
193                     JSONArray arr = new JSONArray(orphanArray);
194                     // adopt orphans
195                     logger.fine("Adopting " + arr.size() + " orphans");
196                     for (int i=0; i < arr.size(); i++) {
197                         final Window w = (Window)((JSONObject)arr.get(i)).getJavaScriptObject();
198                         windowList.add(w);
199                         Scheduler.get().scheduleDeferred(new Scheduler.ScheduledCommand() {
200                             @Override
201                             public void execute() {
202                                 dispatchEvent(w, event(EventType.ANNOUNCE_NEW_PARENT));
203                             }
204                         });
205                     }
206                     Window.current().remove("orphans");
207                 }
208                 if (masterConnection != null) {
209                     masterConnection.start();
210                 }
211                 return;
212                 
213             // these events need to dispatched to our children as well as to the client listener
214             // they originate from the master connection and need to be propagated top-down
215             case ON_CONNECTED:
216                 clientListener.onConnected(-1, -1);
217                 break;
218             case ON_BEFORE_DISCONNECTED:
219                 clientListener.onBeforeDisconnected();
220                 break;
221             case ON_DISCONNECTED:
222                 clientListener.onDisconnected();
223                 break;
224             case ON_ERROR:
225 //                clientListener.onError(new AtmosphereClientException((String)event.getData()), true);
226                 break;
227             case ON_HEARTBEAT:
228                 clientListener.onHeartbeat();
229                 break;
230             case ON_REFRESH:
231                 clientListener.onRefresh();
232                 break;
233             case ON_AFTER_REFRESH:
234                 clientListener.onAfterRefresh();
235                 break;
236             case ON_MESSAGE:
237                 clientListener.onMessage(Collections.singletonList(event.getData()));
238                 break;
239                 
240             // these events originate from the client and are meant for the master connection
241             case POST:
242                 if (masterConnection != null) {
243                     masterConnection.post(event.getData());
244                 } else if (parent != null) {
245                     // we are not the master propagate up
246                     dispatchRawEvent(parent, rawEvent);
247                 } else {
248                     throw new IllegalStateException("Failed to find master connection for post");
249                 }
250                 return;
251             case BROADCAST:
252                 if (masterConnection != null) {
253                     masterConnection.broadcast(event.getData());
254                 } else if (parent != null) {
255                     // we are not the master propagate up
256                     dispatchRawEvent(parent, rawEvent);
257                 } else {
258                     throw new IllegalStateException("Failed to find master connection for broadcast");
259                 }
260                 return;
261             case LOCAL_BROADCAST:
262                 if (masterConnection != null) {
263                     masterListener.onMessage(Collections.singletonList(event.getData()));
264                 } else if (parent != null) {
265                     // we are not the master propagate up
266                     dispatchRawEvent(parent, rawEvent);
267                 } else {
268                     throw new IllegalStateException("Failed to find master connection for local broadcast");
269                 }
270                 return;
271         }
272         // propagate message to children
273         dispatchRawEvent(rawEvent);
274     }

310     protected String serialize(AtmosphereProxyEvent event) {
311         String data;
312         if (event.getData() != null) {
313             try {
314                 data = jsonSerializer.serialize(event.getData());
315             } catch (SerializationException ex) {
316                 throw new IllegalStateException(ex);
317             }
318         } else {
319             data = "";
320         }
321         return event.getEventType().name() + ";" + data;

333      protected AtmosphereProxyEvent deserialize(String data) {
334          AtmosphereProxyEvent event = new AtmosphereProxyEvent();
335          int pos = data.indexOf(";");
336          event.setEventType(EventType.valueOf(data.substring(0, pos)));
337          if (pos + 1 < data.length()) {
338             try {
339                 event.setData(jsonSerializer.deserialize(data.substring(pos+1)));
340             } catch (SerializationException ex) {
341                 throw new IllegalStateException(ex);
342             }
343          }
344          return event;
345     }
