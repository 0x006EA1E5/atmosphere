33      public final static void broadcast(final AtmosphereResource<?, ?> r, final AtmosphereResourceEvent e) {
34          HttpServletRequest request = (HttpServletRequest) r.getRequest();
35  
36          try {
37              ContainerResponse cr = (ContainerResponse) request.getAttribute(FrameworkConfig.CONTAINER_RESPONSE);
38  
39              if (cr == null) {
40                  logger.debug("Retrieving HttpServletRequest {} with ContainerResponse {}", request, cr);
41                  logger.error("Unexpected state. ContainerResponse cannot be null. The connection hasn't been suspended yet");
42                  r.getBroadcaster().removeAtmosphereResource(r);
43                  return;
44              }
45  
46              MediaType m = (MediaType) cr.getHttpHeaders().getFirst(HttpHeaders.CONTENT_TYPE);
47              if (e.getMessage() instanceof Response) {
48                  cr.setResponse((Response) e.getMessage());
49                  cr.getHttpHeaders().add(HttpHeaders.CONTENT_TYPE, m);
50                  cr.write();
51                  cr.getOutputStream().flush();
52              } else if (e.getMessage() instanceof List) {
53                  for (Object msg : (List<Object>) e.getMessage()) {
54                      cr.setResponse(Response.ok(msg).build());
55                      cr.getHttpHeaders().add(HttpHeaders.CONTENT_TYPE, m);
56                      cr.write();
57                      cr.getOutputStream().flush();
58                  }
59              } else {
60                  cr.setResponse(Response.ok(e.getMessage()).build());
61                  cr.getHttpHeaders().add(HttpHeaders.CONTENT_TYPE, m);
62                  cr.write();
63                  cr.getOutputStream().flush();
64              }
65          } catch (Throwable t) {
66              onException(t, r);
67          } finally {
68              Boolean resumeOnBroadcast = (Boolean) request.getAttribute(ApplicationConfig.RESUME_ON_BROADCAST);
69              if (resumeOnBroadcast != null && resumeOnBroadcast) {
70  
71                  String uuid = (String) request.getAttribute(AtmosphereFilter.RESUME_UUID);
72                  if (uuid != null) {
73                      if (request.getAttribute(AtmosphereFilter.RESUME_CANDIDATES) != null) {
74                          ((ConcurrentHashMap<String, AtmosphereResource<?, ?>>) request.getAttribute(AtmosphereFilter.RESUME_CANDIDATES)).remove(uuid);
75                      }
76                  }
77                  r.resume();
78              }
79          }
80      }
81  
82      final static void onException(Throwable t, AtmosphereResource<?, ?> r) {
83          try {
84              logger.debug("onException()", t);
85  
86              if (t instanceof IOException && r instanceof AtmosphereEventLifecycle) {
87                  ((AtmosphereEventLifecycle) r).notifyListeners(new AtmosphereResourceEventImpl((AtmosphereResourceImpl) r, true, false));
88                  ((AtmosphereEventLifecycle) r).removeEventListeners();
89              }
90          } finally {
91              try {
92                  r.getBroadcaster().removeAtmosphereResource(r);
93              } catch (IllegalStateException ex) {
94                  logger.trace(ex.getMessage(), ex);
95              }
96              BroadcasterFactory.getDefault().removeAllAtmosphereResource(r);
97          }
98      }
