16  package org.atmosphere.websocket;
17  
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  
21  import java.io.IOException;
22  import java.io.InputStream;
23  import java.io.Reader;
24  
25  /**
26   * An Adapter for {@link WebSocketStreamingHandlerAdapter}
27   *
28   * @author Jeanfrancois Arcand
29   */
30  public class WebSocketStreamingHandlerAdapter implements WebSocketStreamingHandler {
31  
32      private final Logger logger = LoggerFactory.getLogger(WebSocketStreamingHandlerAdapter.class);
33      @Override
34      public void onBinaryStream(WebSocket webSocket, InputStream inputStream) throws IOException {
35          logger.trace("onBinaryStream {}", webSocket);
36      }
37  
38      @Override
39      public void onTextStream(WebSocket webSocket, Reader reader) throws IOException {
40          logger.trace("onTextStream {}", webSocket);
41      }
42  
43      @Override
44      public void onByteMessage(WebSocket webSocket, byte[] data, int offset, int length) throws IOException {
45          logger.trace("onByteMessage {}", webSocket);
46      }
47  
48      @Override
49      public void onTextMessage(WebSocket webSocket, String data) throws IOException {
50          logger.trace("onTextMessage {}", webSocket);
51      }
52  
53      @Override
54      public void onOpen(WebSocket webSocket) throws IOException {
55          logger.trace("onOpen {}", webSocket);
56      }
57  
58      @Override
59      public void onClose(WebSocket webSocket) {
60          logger.trace("onClose {}", webSocket);
61      }
62  
63      @Override
64      public void onError(WebSocket webSocket, WebSocketProcessor.WebSocketException t) {
65          logger.trace("onError {}", webSocket, t);
66      }
67  }
