73  /**
74   * Like the {@link org.atmosphere.cpr.AsynchronousProcessor} class, this class is responsible for dispatching WebSocket request to the
75   * proper {@link org.atmosphere.websocket.WebSocket} implementation. This class can be extended in order to support any protocol
76   * running on top  websocket.
77   *
78   * @author Jeanfrancois Arcand
79   */
80  public class DefaultWebSocketProcessor implements WebSocketProcessor, Serializable {

97      public DefaultWebSocketProcessor(AtmosphereFramework framework) {
98          this.framework = framework;
99          this.webSocketProtocol = framework.getWebSocketProtocol();
100 
101         String s = framework.getAtmosphereConfig().getInitParameter(RECYCLE_ATMOSPHERE_REQUEST_RESPONSE);
102         if (s != null && Boolean.valueOf(s)) {
103             destroyable = true;
104         } else {
105             destroyable = false;
106         }
107 
108         s = framework.getAtmosphereConfig().getInitParameter(WEBSOCKET_PROTOCOL_EXECUTION);
109         if (s != null && Boolean.valueOf(s)) {
110             executeAsync = true;
111         } else {
112             executeAsync = false;
113         }
114 
115         s = framework.getAtmosphereConfig().getInitParameter(IN_MEMORY_STREAMING_BUFFER_SIZE);
116         if (s != null) {
117             byteBufferMaxSize = Integer.valueOf(s);
118             charBufferMaxSize = byteBufferMaxSize;
119         }
120 
121         AtmosphereConfig config = framework.getAtmosphereConfig();
122         if (executeAsync) {
123             asyncExecutor = ExecutorsFactory.getAsyncOperationExecutor(config, "WebSocket");
124         } else {
125             asyncExecutor = VoidExecutorService.VOID;
126         }
127 
128         scheduler = ExecutorsFactory.getScheduler(config);
129         optimizeMapping();
130     }

146     @Override
147     public final void open(final WebSocket webSocket, final AtmosphereRequest request, final AtmosphereResponse response) throws IOException {
148         // TODO: Fix this. Instead add an Interceptor.
149         if (framework.getAtmosphereConfig().handlers().size() == 0) {
150             framework.addAtmosphereHandler("/*", REFLECTOR_ATMOSPHEREHANDLER);
151         }
152 
153         request.headers(configureHeader(request)).setAttribute(WebSocket.WEBSOCKET_SUSPEND, true);
154 
155         AtmosphereResource r = AtmosphereResourceFactory.getDefault().create(framework.getAtmosphereConfig(),
156                 response,
157                 framework.getAsyncSupport());
158 
159         request.setAttribute(INJECTED_ATMOSPHERE_RESOURCE, r);
160         request.setAttribute(SUSPENDED_ATMOSPHERE_RESOURCE_UUID, r.uuid());
161 
162         webSocket.resource(r);
163         webSocketProtocol.onOpen(webSocket);
164         WebSocketHandler proxy = null;
165         if (handlers.size() != 0) {
166             WebSocketHandlerProxy handler = mapper.map(request, handlers);
167             if (handler == null) {
168                 logger.debug("No WebSocketHandler maps request for {} with mapping {}", request.getRequestURI(), handlers);
169                 throw new AtmosphereMappingException("No AtmosphereHandler maps request for " + request.getRequestURI());
170             }
171             proxy = postProcessMapping(webSocket, request, handler);
172         }
173 
174         // We must dispatch to execute AtmosphereInterceptor
175         dispatch(webSocket, request, response);
176 
177         if (proxy != null) {
178             proxy.onOpen(webSocket);
179             webSocket.webSocketHandler(proxy).resource().suspend(-1);
180         }
181 
182         request.removeAttribute(INJECTED_ATMOSPHERE_RESOURCE);
183 
184         if (webSocket.resource() != null) {
185             final AsynchronousProcessor.AsynchronousProcessorHook hook =
186                     new AsynchronousProcessor.AsynchronousProcessorHook((AtmosphereResourceImpl) webSocket.resource());
187             request.setAttribute(ASYNCHRONOUS_HOOK, hook);
188 
189             final Action action = ((AtmosphereResourceImpl) webSocket.resource()).action();
190             if (action.timeout() != -1 && !framework.getAsyncSupport().getContainerName().contains("Netty")) {
191                 final AtomicReference<Future<?>> f = new AtomicReference();
192                 f.set(scheduler.scheduleAtFixedRate(new Runnable() {
193                     @Override
194                     public void run() {
195                         if (WebSocket.class.isAssignableFrom(webSocket.getClass())
196                                 && System.currentTimeMillis() - WebSocket.class.cast(webSocket).lastWriteTimeStampInMilliseconds() > action.timeout()) {
197                             hook.timedOut();
198                             f.get().cancel(true);
199                         }
200                     }
201                 }, action.timeout(), action.timeout(), TimeUnit.MILLISECONDS));
202             }
203         } else {
204             logger.warn("AtmosphereResource was null");
205         }
206         notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent("", CONNECT, webSocket));
207     }

209     protected WebSocketHandler postProcessMapping(WebSocket webSocket, AtmosphereRequest request, WebSocketHandlerProxy w) {
210         WebSocketHandlerProxy p = null;
211         String path = w.path();
212         if (wildcardMapping()) {
213             String pathInfo = null;
214             try {
215                 pathInfo = request.getPathInfo();
216             } catch (IllegalStateException ex) {
217                 // http://java.net/jira/browse/GRIZZLY-1301
218             }
219 
220             if (pathInfo != null) {
221                 path = request.getServletPath() + pathInfo;
222             } else {
223                 path = request.getServletPath();
224             }
225 
226             if (path == null || path.isEmpty()) {
227                 path = "/";
228             }
229 
230             synchronized (handlers) {
231                 p = handlers.get(path);
232                 if (p == null) {
233                     // AtmosphereHandlerService
234                     WebSocketHandlerService a = w.proxied.getClass().getAnnotation(WebSocketHandlerService.class);
235                     if (a != null) {
236                         String targetPath = a.path();
237                         if (targetPath.indexOf("{") != -1 && targetPath.indexOf("}") != -1) {
238                             try {
239                                 boolean singleton = w.proxied.getClass().getAnnotation(Singleton.class) != null;
240                                 if (!singleton) {
241                                     registerWebSocketHandler(path, new WebSocketHandlerProxy(a.broadcaster(),
242                                             framework.newClassInstance(w.proxied.getClass())));
243                                 } else {
244                                     registerWebSocketHandler(path, new WebSocketHandlerProxy(a.broadcaster(), w));
245                                 }
246                                 p = handlers.get(path);
247                             } catch (Throwable e) {
248                                 logger.warn("Unable to create WebSocketHandler", e);
249                             }
250                         }
251                     }
252                 }
253             }
254         }
255 
256         try {
257             webSocket.resource().setBroadcaster(AnnotationUtil.broadcaster(framework, p != null ? p.broadcasterClazz : w.broadcasterClazz, path));
258         } catch (Exception e) {
259             logger.error("", e);
260         }
261 
262         return p != null ? p.proxied : w.proxied;
263     }

265     private void dispatch(final WebSocket webSocket, List<AtmosphereRequest> list) {
266         if (list == null) return;
267 
268         for (final AtmosphereRequest r : list) {
269             if (r != null) {
270 
271                 r.dispatchRequestAsynchronously();
272                 asyncExecutor.execute(new Runnable() {
273                     @Override
274                     public void run() {
275                         AtmosphereResponse w = new AtmosphereResponse(webSocket, r, destroyable);
276                         try {
277                             dispatch(webSocket, r, w);
278                         } finally {
279                             r.destroy();
280                             w.destroy();
281                         }
282                     }
283                 });
284             }
285         }
286     }

288     @Override
289     public void invokeWebSocketProtocol(final WebSocket webSocket, String webSocketMessage) {
290         WebSocketHandler webSocketHandler = webSocket.webSocketHandler();
291 
292         if (webSocketHandler == null) {
293             if (!WebSocketProtocolStream.class.isAssignableFrom(webSocketProtocol.getClass())) {
294                 List<AtmosphereRequest> list = webSocketProtocol.onMessage(webSocket, webSocketMessage);
295                 dispatch(webSocket, list);
296             } else {
297                 logger.debug("The WebServer doesn't support streaming. Wrapping the message as stream.");
298                 invokeWebSocketProtocol(webSocket, new StringReader(webSocketMessage));
299                 return;
300             }
301         } else {
302             if (!WebSocketStreamingHandler.class.isAssignableFrom(webSocketHandler.getClass())) {
303                 try {
304                     webSocketHandler.onTextMessage(webSocket, webSocketMessage);
305                 } catch (Exception ex) {
306                     handleException(ex, webSocket, webSocketHandler);
307                 }
308             } else {
309                 logger.debug("The WebServer doesn't support streaming. Wrapping the message as stream.");
310                 invokeWebSocketProtocol(webSocket, new StringReader(webSocketMessage));
311                 return;
312             }
313         }
314         notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent(webSocketMessage, MESSAGE, webSocket));
315     }

317     @Override
318     public void invokeWebSocketProtocol(WebSocket webSocket, byte[] data, int offset, int length) {
319         WebSocketHandler webSocketHandler = webSocket.webSocketHandler();
320 
321         if (webSocketHandler == null) {
322             if (!WebSocketProtocolStream.class.isAssignableFrom(webSocketProtocol.getClass())) {
323                 List<AtmosphereRequest> list = webSocketProtocol.onMessage(webSocket, data, offset, length);
324                 dispatch(webSocket, list);
325             } else {
326                 logger.debug("The WebServer doesn't support streaming. Wrapping the message as stream.");
327                 invokeWebSocketProtocol(webSocket, new ByteArrayInputStream(data, offset, length));
328                 return;
329             }
330         } else {
331             if (!WebSocketStreamingHandler.class.isAssignableFrom(webSocketHandler.getClass())) {
332                 try {
333                     webSocketHandler.onByteMessage(webSocket, data, offset, length);
334                 } catch (Exception ex) {
335                     handleException(ex, webSocket, webSocketHandler);
336                 }
337             } else {
338                 logger.debug("The WebServer doesn't support streaming. Wrapping the message as stream.");
339                 invokeWebSocketProtocol(webSocket, new ByteArrayInputStream(data, offset, length));
340                 return;
341             }
342         }
343         notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent<byte[]>(data, MESSAGE, webSocket));
344     }

346     private void handleException(Exception ex, WebSocket webSocket, WebSocketHandler webSocketHandler) {
347         logger.error("", ex);
348         AtmosphereResource r = webSocket.resource();
349         webSocketHandler.onError(webSocket, new WebSocketException(ex,
350                 new AtmosphereResponse.Builder()
351                         .request(r != null ? AtmosphereResourceImpl.class.cast(r).getRequest(false) : null)
352                         .status(500)
353                         .statusMessage("Server Error").build()));
354     }

356     @Override
357     public void invokeWebSocketProtocol(WebSocket webSocket, InputStream stream) {
358         WebSocketHandler webSocketHandler = webSocket.webSocketHandler();
359         try {
360             if (webSocketHandler == null) {
361                 if (WebSocketProtocolStream.class.isAssignableFrom(webSocketProtocol.getClass())) {
362                     List<AtmosphereRequest> list = WebSocketProtocolStream.class.cast(webSocketProtocol).onBinaryStream(webSocket, stream);
363                     dispatch(webSocket, list);
364                 } else {
365                     dispatchStream(webSocket, stream);
366                     return;
367                 }
368             } else {
369                 if (WebSocketStreamingHandler.class.isAssignableFrom(webSocketHandler.getClass())) {
370                     WebSocketStreamingHandler.class.cast(webSocketHandler).onBinaryStream(webSocket, stream);
371                 } else {
372                     dispatchStream(webSocket, stream);
373                     return;
374                 }
375 
376             }
377         } catch (Exception ex) {
378             handleException(ex, webSocket, webSocketHandler);
379         }
380 
381         notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent<InputStream>(stream, MESSAGE, webSocket));
382     }

412     /**
413      * Dispatch to request/response to the {@link org.atmosphere.cpr.AsyncSupport} implementation as it was a normal HTTP request.
414      *
415      * @param request a {@link AtmosphereRequest}
416      * @param r       a {@link AtmosphereResponse}
417      */
418     public final void dispatch(WebSocket webSocket, final AtmosphereRequest request, final AtmosphereResponse r) {
419         if (request == null) return;
420 
421         try {
422             framework.doCometSupport(request, r);
423         } catch (Throwable e) {
424             logger.warn("Failed invoking AtmosphereFramework.doCometSupport()", e);
425             webSocketProtocol.onError(webSocket, new WebSocketException(e,
426                     new AtmosphereResponse.Builder()
427                             .request(request)
428                             .status(500)
429                             .statusMessage("Server Error").build()));
430             return;
431         }
432 
433         if (r.getStatus() >= 400) {
434             webSocketProtocol.onError(webSocket, new WebSocketException("Status code higher or equal than 400", r));
435         }
436     }

438     @Override
439     public void close(WebSocket webSocket, int closeCode) {
440         logger.trace("WebSocket closed with {}", closeCode);
441 
442         WebSocketHandler webSocketHandler = webSocket.webSocketHandler();
443         // A message might be in the process of being processed and the websocket gets closed. In that corner
444         // case the webSocket.resource will be set to false and that might cause NPE in some WebSocketProcol implementation
445         // We could potentially synchronize on webSocket but since it is a rare case, it is better to not synchronize.
446         // synchronized (webSocket) {
447 
448         closeCode = closeCode(closeCode);
449         notifyListener(webSocket, new WebSocketEventListener.WebSocketEvent(closeCode, CLOSE, webSocket));
450         AtmosphereResourceImpl resource = (AtmosphereResourceImpl) webSocket.resource();
451 
452         if (resource == null) {
453             logger.warn("Unable to retrieve AtmosphereResource for {}", webSocket);
454         } else {
455             logger.trace("About to close AtmosphereResource for {}", resource.uuid());
456             AtmosphereRequest r = resource.getRequest(false);
457             AtmosphereResponse s = resource.getResponse(false);
458             try {
459                 webSocketProtocol.onClose(webSocket);
460 
461                 if (resource != null && resource.isInScope()) {
462 
463                     if (webSocketHandler != null) {
464                         webSocketHandler.onClose(webSocket);
465                     }
466 
467                     Object o = r.getAttribute(ASYNCHRONOUS_HOOK);
468                     AsynchronousProcessor.AsynchronousProcessorHook h;
469                     if (o != null && AsynchronousProcessor.class.isAssignableFrom(o.getClass())) {
470                         h = (AsynchronousProcessor.AsynchronousProcessorHook) o;
471                         if (!resource.isCancelled()) {
472                             if (closeCode == 1005) {
473                                 h.closed();
474                             } else {
475                                 h.timedOut();
476                             }
477                         }
478                     }
479 
480                     resource.setIsInScope(false);
481                     try {
482                         resource.cancel();
483                     } catch (IOException e) {
484                         logger.trace("", e);
485                     }
486                     AtmosphereResourceImpl.class.cast(resource)._destroy();
487                 }
488 
489             } finally {
490                 if (webSocket != null) {
491                     try {
492                         r.setAttribute(WebSocket.CLEAN_CLOSE, Boolean.TRUE);
493                         webSocket.resource(null).close(s);
494                     } catch (IOException e) {
495                         logger.trace("", e);
496                     }
497                 }
498 
499                 if (r != null) {
500                     r.destroy(true);
501                 }
502 
503                 if (s != null) {
504                     s.destroy(true);
505                 }
506 
507             }
508         }
509     }

512     public void destroy() {
513         boolean shared = framework.isShareExecutorServices();
514         if (asyncExecutor != null && !shared) {
515             asyncExecutor.shutdown();
516         }
517 
518         if (scheduler != null && !shared) {
519             scheduler.shutdown();
520         }

523     private int closeCode(int closeCode) {
524         // Tomcat and Jetty differ, same with browser
525         if (closeCode == 1000 && framework.getAsyncSupport().getContainerName().contains("Tomcat")) {
526             closeCode = 1005;
527         }
528         return closeCode;
529     }

532     @Override
533     public void notifyListener(WebSocket webSocket, WebSocketEventListener.WebSocketEvent event) {
534         AtmosphereResource resource = webSocket.resource();
535         if (resource == null) return;
536 
537         AtmosphereResourceImpl r = AtmosphereResourceImpl.class.cast(resource);
538         for (AtmosphereResourceEventListener l : r.atmosphereResourceEventListener()) {
539             if (WebSocketEventListener.class.isAssignableFrom(l.getClass())) {
540                 try {
541                     switch (event.type()) {
542                         case CONNECT:
543                             WebSocketEventListener.class.cast(l).onConnect(event);
544                             break;
545                         case DISCONNECT:
546                             WebSocketEventListener.class.cast(l).onDisconnect(event);
547                             break;
548                         case CONTROL:
549                             WebSocketEventListener.class.cast(l).onControl(event);
550                             break;
551                         case MESSAGE:
552                             WebSocketEventListener.class.cast(l).onMessage(event);
553                             break;
554                         case HANDSHAKE:
555                             WebSocketEventListener.class.cast(l).onHandshake(event);
556                             break;
557                         case CLOSE:
558                             boolean isClosedByClient = r.getAtmosphereResourceEvent().isClosedByClient();
559                             l.onDisconnect(new AtmosphereResourceEventImpl(r, !isClosedByClient, false, isClosedByClient, null));
560                             WebSocketEventListener.class.cast(l).onDisconnect(event);
561                             WebSocketEventListener.class.cast(l).onClose(event);
562                             break;
563                     }
564                 } catch (Throwable t) {
565                     logger.debug("Listener error {}", t);
566                     try {
567                         WebSocketEventListener.class.cast(l).onThrowable(new AtmosphereResourceEventImpl(r, false, false, t));
568                     } catch (Throwable t2) {
569                         logger.warn("Listener error {}", t2);
570                     }
571                 }
572             }
573         }
574     }

576     public static final Map<String, String> configureHeader(AtmosphereRequest request) {
577         Map<String, String> headers = new HashMap<String, String>();
578 
579         Enumeration<String> e = request.getParameterNames();
580         String s;
581         while (e.hasMoreElements()) {
582             s = e.nextElement();
583             headers.put(s, request.getParameter(s));
584         }
585 
586         headers.put(HeaderConfig.X_ATMOSPHERE_TRANSPORT, HeaderConfig.WEBSOCKET_TRANSPORT);
587         return headers;
588     }

590     protected void dispatchStream(WebSocket webSocket, InputStream is) throws IOException {
591         int read = 0;
592         ByteBuffer bb = webSocket.bb;
593         try {
594             while (read > -1) {
595                 bb.position(bb.position() + read);
596                 if (bb.remaining() == 0) {
597                     resizeByteBuffer(webSocket);
598                 }
599                 read = is.read(bb.array(), bb.position(), bb.remaining());
600             }
601             bb.flip();
602 
603             invokeWebSocketProtocol(webSocket, bb.array(), 0, bb.limit());
604         } finally {
605             bb.clear();
606         }
607     }
608 
609     protected void dispatchReader(WebSocket webSocket, Reader r) throws IOException {
610         int read = 0;
611         CharBuffer cb = webSocket.cb;
612         try {
613             while (read > -1) {
614                 cb.position(cb.position() + read);
615                 if (cb.remaining() == 0) {
616                     resizeCharBuffer(webSocket);
617                 }
618                 read = r.read(cb.array(), cb.position(), cb.remaining());
619             }
620             cb.flip();
621             invokeWebSocketProtocol(webSocket, cb.toString());
622         } finally {
623             cb.clear();
624         }
625     }

627     private void resizeByteBuffer(WebSocket webSocket) throws IOException {
628         int maxSize = getByteBufferMaxSize();
629         ByteBuffer bb = webSocket.bb;
630         if (bb.limit() >= maxSize) {
631             throw new IOException("Message Buffer too small. Use " + StreamingHttpProtocol.class.getName() + " when streaming over websocket.");
632         }
633 
634         long newSize = bb.limit() * 2;
635         if (newSize > maxSize) {
636             newSize = maxSize;
637         }
638 
639         // Cast is safe. newSize < maxSize and maxSize is an int
640         ByteBuffer newBuffer = ByteBuffer.allocate((int) newSize);
641         bb.rewind();
642         newBuffer.put(bb);
643         webSocket.bb = newBuffer;
644     }

646     private void resizeCharBuffer(WebSocket webSocket) throws IOException {
647         int maxSize = getCharBufferMaxSize();
648         CharBuffer cb = webSocket.cb;
649         if (cb.limit() >= maxSize) {
650             throw new IOException("Message Buffer too small. Use " + StreamingHttpProtocol.class.getName() + " when streaming over websocket.");
651         }
652 
653         long newSize = cb.limit() * 2;
654         if (newSize > maxSize) {
655             newSize = maxSize;
656         }
657 
658         // Cast is safe. newSize < maxSize and maxSize is an int
659         CharBuffer newBuffer = CharBuffer.allocate((int) newSize);
660         cb.rewind();
661         newBuffer.put(cb);
662         webSocket.cb = newBuffer;
663     }
