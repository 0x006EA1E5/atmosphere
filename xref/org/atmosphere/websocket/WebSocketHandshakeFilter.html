16  package org.atmosphere.websocket;
17  
18  import org.atmosphere.cpr.ApplicationConfig;
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  import javax.servlet.Filter;
23  import javax.servlet.FilterChain;
24  import javax.servlet.FilterConfig;
25  import javax.servlet.ServletException;
26  import javax.servlet.ServletRequest;
27  import javax.servlet.ServletResponse;
28  import javax.servlet.http.HttpServletRequest;
29  import javax.servlet.http.HttpServletResponse;
30  import java.io.IOException;
31  
32  import static org.atmosphere.cpr.HeaderConfig.X_ATMOSPHERE_ERROR;
33  
34  /**
35   * A Servlet Filter for configuring which WebSocket protocol version an application want to support.
36   *
37   * @author Jeanfrancois Arcand
38   */
39  public class WebSocketHandshakeFilter implements Filter {
40  
41      private static final Logger logger = LoggerFactory.getLogger(WebSocketHandshakeFilter.class);
42      private String[] bannedVersion;
43  
44      @Override
45      public void init(FilterConfig filterConfig) throws ServletException {
46          String draft = filterConfig.getInitParameter(ApplicationConfig.WEB_SOCKET_BANNED_VERSION);
47          if (draft != null) {
48              bannedVersion = draft.split(",");
49              logger.debug("Blocked WebSocket Draft version {}", draft);
50          }
51      }
52  
53      @Override
54      public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException, ServletException {
55  
56          if (HttpServletRequest.class.cast(request).getHeader("Connection") != null && HttpServletRequest.class.cast(request).getHeader("Connection").equalsIgnoreCase("upgrade")) {
57              int draft = HttpServletRequest.class.cast(request).getIntHeader("Sec-WebSocket-Version");
58              if (draft < 0) {
59                  draft = HttpServletRequest.class.cast(request).getIntHeader("Sec-WebSocket-Draft");
60              }
61  
62              if (bannedVersion != null) {
63                  for (String s : bannedVersion) {
64                      if (Integer.getInteger(s) == draft) {
65                          HttpServletResponse.class.cast(response).addHeader(X_ATMOSPHERE_ERROR, "Websocket protocol not supported");
66                          HttpServletResponse.class.cast(response).sendError(202, "Websocket protocol not supported");
67                          return;
68                      }
69                  }
70              }
71          } else if (HttpServletRequest.class.cast(request).getIntHeader("Sec-WebSocket-Version") > 0) {
72              logger.error("Invalid WebSocket Specification {} with {} ", HttpServletRequest.class.cast(request).getHeader("Connection"), HttpServletRequest.class.cast(request).getIntHeader("Sec-WebSocket-Version"));
73              HttpServletResponse.class.cast(response).addHeader(X_ATMOSPHERE_ERROR, "Websocket protocol not supported");
74              HttpServletResponse.class.cast(response).sendError(202, "Websocket protocol not supported");
75              return;
76          }
77          chain.doFilter(request, response);
78      }
79  
80      @Override
81      public void destroy() {
82      }
83  }
