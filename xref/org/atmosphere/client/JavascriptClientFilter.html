17  package org.atmosphere.client;
18  
19  import org.atmosphere.cpr.PerRequestBroadcastFilter;
20  
21  import javax.servlet.http.HttpServletRequest;
22  import javax.servlet.http.HttpServletResponse;
23  import java.util.concurrent.atomic.AtomicInteger;
24  
25  /**
26   * Filter that inject Javascript code to a broadcast so it can be used with the Atmosphere JQuery Plugin.
27   *
28   * @author Jeanfrancois Arcand
29   */
30  public class JavascriptClientFilter implements PerRequestBroadcastFilter {
31  
32      private final AtomicInteger uniqueScriptToken = new AtomicInteger();
33  
34      @Override
35      public BroadcastAction filter(Object originalMessage, Object message) {
36  
37          if (message instanceof String) {
38              StringBuilder sb = new StringBuilder("<script id=\"atmosphere_")
39                      .append(uniqueScriptToken.getAndIncrement())
40                      .append("\">")
41                      .append("parent.callback")
42                      .append("('")
43                      .append(message.toString())
44                      .append("');</script>");
45              message = sb.toString();
46          }
47          return new BroadcastAction(BroadcastAction.ACTION.CONTINUE, message);
48      }
49  
50      @Override
51      public BroadcastAction filter(HttpServletRequest request, HttpServletResponse response, Object message) {
52  
53          String userAgent = request.getHeader("User-Agent") == null ? "" : request.getHeader("User-Agent").toLowerCase();
54          String transport = request.getHeader("X-Atmosphere-Transport") == null ? "streaming" : request.getHeader("X-Atmosphere-Transport").toString();
55          if (transport.equals("streaming") && userAgent.startsWith("opera")) {
56              StringBuilder sb = new StringBuilder("<script id=\"atmosphere_")
57                      .append(uniqueScriptToken.getAndIncrement())
58                      .append("\">")
59                      .append("window.parent.$.atmosphere.streamingCallback")
60                      .append("('")
61                      .append(message.toString())
62                      .append("');</script>");
63              message = sb.toString();
64              return new BroadcastAction(BroadcastAction.ACTION.CONTINUE, message);
65          }
66          return new BroadcastAction(BroadcastAction.ACTION.CONTINUE, message);
67      }
68  }
