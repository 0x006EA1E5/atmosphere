26  /**
27   * Use this annotation with the {@link ManagedService}. A method annotated with this annotation will be invoked when the
28   * connection has been suspended and ready.
29   *
30   * @author Jeanfrancois Arcand
31   */
32  @Retention(RetentionPolicy.RUNTIME)
33  @Target(ElementType.METHOD)
34  public @interface Ready {
35  
36      static enum DELIVER_TO { RESOURCE, BROADCASTER, ALL}
37  
38      /**
39       * Broadcast the returned value to only the calling resource {@link org.atmosphere.config.service.Ready.DELIVER_TO#RESOURCE},
40       * to it's associated Broadcaster {@link org.atmosphere.config.service.Ready.DELIVER_TO#BROADCASTER}
41       * or to all created Broadcaster {@link org.atmosphere.config.service.Ready.DELIVER_TO#ALL}
42       * @return the {@link org.atmosphere.config.service.Ready.DELIVER_TO}
43       */
44      DELIVER_TO value() default  DELIVER_TO.RESOURCE;
45  
46      /**
47       * A list of {@link org.atmosphere.config.managed.Encoder}
48       */
49      Class<? extends Encoder>[] encoders() default {};
50  }
