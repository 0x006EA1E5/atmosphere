38  /**
39   * This class is responsible for creating {@link Broadcaster} instances. You can also add and remove {@link Broadcaster}
40   * and lookup using {@link BroadcasterFactory#getDefault()} ()} from any classes loaded using the same class loader.
41   *
42   * @author Jeanfrancois Arcand
43   * @author Jason Burgess
44   */
45  public class DefaultBroadcasterFactory extends BroadcasterFactory {
46  
47      private static final Logger logger = LoggerFactory.getLogger(DefaultBroadcasterFactory.class);
48  
49      private final ConcurrentHashMap<Object, Broadcaster> store = new ConcurrentHashMap<Object, Broadcaster>();
50  
51      private final Class<? extends Broadcaster> clazz;
52  
53      private BroadcasterLifeCyclePolicy policy =
54              new BroadcasterLifeCyclePolicy.Builder().policy(NEVER).build();
55      protected Broadcaster.POLICY defaultPolicy = Broadcaster.POLICY.FIFO;
56      protected int defaultPolicyInteger = -1;
57      private final URI legacyBroadcasterURI = URI.create("http://127.0.0.0");
58  
59      protected DefaultBroadcasterFactory(Class<? extends Broadcaster> clazz, String broadcasterLifeCyclePolicy, AtmosphereConfig c) {
60          this.clazz = clazz;
61          this.factory = this;
62          config = c;
63          configure(broadcasterLifeCyclePolicy);
64      }
65  
66      private void configure(String broadcasterLifeCyclePolicy) {
67  
68          int maxIdleTime = 5 * 60 * 1000;
69          String s = config.getInitParameter(ApplicationConfig.BROADCASTER_LIFECYCLE_POLICY_IDLETIME);
70          if (s != null) {
71              maxIdleTime = Integer.parseInt(s);
72          }
73  
74          s = config.getInitParameter(BROADCASTER_POLICY);
75          if (s != null) {
76              defaultPolicy = s.equalsIgnoreCase(Broadcaster.POLICY.REJECT.name()) ? Broadcaster.POLICY.REJECT : Broadcaster.POLICY.FIFO;
77          }
78  
79          s = config.getInitParameter(BROADCASTER_POLICY_TIMEOUT);
80          if (s != null) {
81              defaultPolicyInteger = Integer.valueOf(s);
82          }
83  
84          if (EMPTY.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
85              policy = new BroadcasterLifeCyclePolicy.Builder().policy(EMPTY).build();
86          } else if (EMPTY_DESTROY.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
87              policy = new BroadcasterLifeCyclePolicy.Builder().policy(EMPTY_DESTROY).build();
88          } else if (IDLE.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
89              policy = new BroadcasterLifeCyclePolicy.Builder().policy(IDLE).idleTimeInMS(maxIdleTime).build();
90          } else if (IDLE_DESTROY.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
91              policy = new BroadcasterLifeCyclePolicy.Builder().policy(IDLE_DESTROY).idleTimeInMS(maxIdleTime).build();
92          } else if (IDLE_RESUME.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
93              policy = new BroadcasterLifeCyclePolicy.Builder().policy(IDLE_RESUME).idleTimeInMS(maxIdleTime).build();
94          } else if (NEVER.name().equalsIgnoreCase(broadcasterLifeCyclePolicy)) {
95              policy = new BroadcasterLifeCyclePolicy.Builder().policy(NEVER).build();
96          } else {
97              logger.warn("Unsupported BroadcasterLifeCyclePolicy policy {}", broadcasterLifeCyclePolicy);
98          }
99      }

111     @Override
112     public final <T extends Broadcaster> T get(Class<T> c, Object id) {
113 
114         if (id == null) {
115             throw new NullPointerException("id is null");
116         }
117         if (c == null) {
118             throw new NullPointerException("Class is null");
119         }
120 
121         return lookup(c, id, true, true);
122     }
123 
124     private <T extends Broadcaster> T createBroadcaster(Class<T> c, Object id) throws BroadcasterCreationException {
125         try {
126             T b = config.framework().newClassInstance(c, c);
127             b.initialize(id.toString(), legacyBroadcasterURI, config);
128             b.setSuspendPolicy(defaultPolicyInteger, defaultPolicy);
129 
130             if (b.getBroadcasterConfig() == null) {
131                 b.setBroadcasterConfig(new BroadcasterConfig(config.framework().broadcasterFilters, config, id.toString()).init());
132             }
133 
134             b.setBroadcasterLifeCyclePolicy(policy);
135             if (DefaultBroadcaster.class.isAssignableFrom(clazz)) {
136                 DefaultBroadcaster.class.cast(b).start();
137             }
138 
139             for (BroadcasterListener l : broadcasterListeners) {
140                 b.addBroadcasterListener(l);
141             }
142             logger.trace("Broadcaster {} was created {}", id, b);
143 
144             notifyOnPostCreate(b);
145             return b;
146         } catch (Throwable t) {
147             throw new BroadcasterCreationException(t);
148         }
149     }

156     @Override
157     public boolean remove(Broadcaster b, Object id) {
158         boolean removed = store.remove(id, b);
159         if (removed && logger.isDebugEnabled()) {
160             logger.debug("Removing Broadcaster {} factory size now {} ", id, store.size());
161         }
162         return removed;
163     }

185     public <T extends Broadcaster> T lookup(Class<T> c, Object id, boolean createIfNull, boolean unique) {
186         synchronized (c) {
187             logger.trace("About to create {}", id);
188             if (unique && store.get(id) != null) {
189                 throw new IllegalStateException("Broadcaster already exists " + id + ". Use BroadcasterFactory.lookup instead");
190             }
191 
192             T b = (T) store.get(id);
193             logger.trace("Looking in the store using {} returned {}", id, b);
194             if (b != null && !c.isAssignableFrom(b.getClass())) {
195                 String msg = "Invalid lookup class " + c.getName() + ". Cached class is: " + b.getClass().getName();
196                 logger.debug(msg);
197                 throw new IllegalStateException(msg);
198             }
199 
200             if ((b == null && createIfNull) || (b != null && b.isDestroyed())) {
201                 if (b != null) {
202                     logger.trace("Removing destroyed Broadcaster {}", b.getID());
203                     store.remove(b.getID(), b);
204                 }
205 
206                 Broadcaster nb = store.get(id);
207                 if (nb == null) {
208                     nb = createBroadcaster(c, id);
209                     store.put(id, nb);
210                 }
211 
212                 if (nb == null && logger.isTraceEnabled()) {
213                     logger.trace("Added Broadcaster {} . Factory size: {}", id, store.size());
214                 }
215 
216                 b = (T) nb;
217             }
218             return b;
219         }
220     }

222     @Override
223     public void removeAllAtmosphereResource(AtmosphereResource r) {
224         // Remove inside all Broadcaster as well.
225         try {
226             if (store.size() > 0) {
227                 for (Broadcaster b : lookupAll()) {
228                     try {
229                         // Prevent deadlock
230                         if (b.getAtmosphereResources().contains(r)) {
231                             b.removeAtmosphereResource(r);
232                         }
233                     } catch (IllegalStateException ex) {
234                         logger.trace(ex.getMessage(), ex);
235                     }
236                 }
237             }
238         } catch (Exception ex) {
239             logger.warn(ex.getMessage(), ex);
240         }
241     }

243     @Override
244     public boolean remove(Object id) {
245         return store.remove(id) != null;

248     @Override

250         return Collections.unmodifiableCollection(store.values());

253     @Override
254     public synchronized void destroy() {
255 
256         String s = config.getInitParameter(ApplicationConfig.SHARED);
257         if (s != null && s.equalsIgnoreCase("true")) {
258             logger.warn("Factory shared, will not be destroyed. This can possibly cause memory leaks if" +
259                     "Broadcasters were created. Make sure you destroy them manually.");
260             return;
261         }
262 
263         Enumeration<Broadcaster> e = store.elements();
264         Broadcaster b;
265         // We just need one when shared.
266         BroadcasterConfig bc = null;
267         while (e.hasMoreElements()) {
268             try {
269                 b = e.nextElement();
270                 b.resumeAll();
271                 bc = b.getBroadcasterConfig();
272                 bc.forceDestroy();
273                 b.destroy();
274             } catch (Throwable t) {
275                 // Shield us from any bad behaviour
276                 logger.debug("Destroy", t);
277             }
278         }
279         broadcasterListeners.clear();
280         store.clear();
281         factory = null;
282     }
283 
284     public void notifyOnPostCreate(Broadcaster b) {
285         for (BroadcasterListener l : broadcasterListeners) {
286             try {
287                 l.onPostCreate(b);
288             } catch (Exception ex) {
289                 logger.warn("onPostCreate", ex);
290             }
291         }
292     }

294     /**
295      * Build a default {@link BroadcasterFactory} returned when invoking {@link #getDefault()} ()}.
296      *
297      * @param clazz A class implementing {@link Broadcaster}
298      * @param c     An instance of {@link AtmosphereConfig}
299      * @return the default {@link BroadcasterFactory}.
300      * @throws InstantiationException
301      * @throws IllegalAccessException
302      */
303     public static BroadcasterFactory buildAndReplaceDefaultfactory(Class<? extends Broadcaster> clazz, AtmosphereConfig c)
304             throws InstantiationException, IllegalAccessException {
305 
306         factory = new DefaultBroadcasterFactory(clazz, "NEVER", c);
307         c.framework().setBroadcasterFactory(factory);
308         return factory;
309     }

311     public static final class BroadcasterCreationException extends RuntimeException {
312         public BroadcasterCreationException(Throwable t) {
313             super(t);
314         }
315     }

317     @Override
318     public BroadcasterFactory addBroadcasterListener(BroadcasterListener l) {
319         super.addBroadcasterListener(l);
320         for (Broadcaster b : store.values()) {
321             b.addBroadcasterListener(l);
322         }
323         return this;
324     }
325 
326     @Override
327     public BroadcasterFactory removeBroadcasterListener(BroadcasterListener l) {
328         super.removeBroadcasterListener(l);
329         for (Broadcaster b : store.values()) {
330             b.removeBroadcasterListener(l);
331         }
332         return this;
333     }
