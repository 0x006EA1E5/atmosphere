17  package org.atmosphere.cpr;
18  
19  /**
20   * An AtmosphereResourceEvent is created every time an event occurs, like when a
21   * {@link Broadcaster#broadcast(java.lang.Object)} is executed, when a browser remotely closes the connection or
22   * when a suspended resource times out or gets resumed. When such events occur, an instance of that class will be
23   * created and its associated {@link AtmosphereHandler#onStateChange(org.atmosphere.cpr.AtmosphereResourceEvent)} will be invoked.
24   *
25   * @author Jeanfrancois Arcand
26   */
27  public interface AtmosphereResourceEvent {
28  
29      /**
30       * Return the object that was passed to {@link Broadcaster#broadcast(java.lang.Object)}.
31       *
32       * @return the object that was passed to {@link Broadcaster#broadcast(java.lang.Object)}
33       */
34      public Object getMessage();
35  
36      /**
37       * Set an Object that can be retrieved with {@link #getMessage()}. Note that the value may be overridden when
38       * {@link Broadcaster#broadcast(java.lang.Object)} gets invoked.
39       *
40       * @param o an Object that can be retrieved with {@link #getMessage()}.
41       */
42      public AtmosphereResourceEvent setMessage(Object o);
43  
44      /**
45       * Return true is the response gets resumed after a timeout.
46       *
47       * @return true is the response gets resumed after a timeout.
48       */
49      public boolean isResumedOnTimeout();
50  
51      /**
52       * Return true when the remote client close the connection.
53       *
54       * @return true when the remote client close the connection.
55       */
56      public boolean isCancelled();
57  
58      /**
59       * Return <tt>true</tt> if {@link AtmosphereResource#suspend()} has been invoked and set to <tt>true</tt>.
60       *
61       * @return <tt>true</tt> if {@link AtmosphereResource#suspend()} has been invoked and set to <tt>true</tt>
62       */
63      public boolean isSuspended();
64  
65      /**
66       * Return <tt>true</tt> if {@link AtmosphereResource#resume()} has been invoked.
67       *
68       * @return <tt>true</tt> if {@link AtmosphereResource#resume()} has been invoked and set to <tt>true</tt>
69       */
70      public boolean isResuming();
71  
72      /**
73       * Return the {@link AtmosphereResource} associated with this event.
74       *
75       * @return {@link AtmosphereResource}
76       */
77      public AtmosphereResource getResource();
78  
79      /**
80       * Return true if the client closed the connection and send the Atmosphere close message. You must
81       * use the {@link org.atmosphere.interceptor.OnDisconnectInterceptor} in order to receive the proper value,
82       * and atmosphereProtocol must be set to true on the client side (enabledProtocol is true by default).
83       *
84       * @return
85       */
86      public boolean isClosedByClient();
87  
88      /**
89       * Return true if the application closed the connection using {@link org.atmosphere.cpr.AtmosphereResource#close()}.
90       *
91       * @return if the application.
92       */
93      public boolean isClosedByApplication();
94  
95      /**
96       * Return a {@link Throwable} if an unexpected exception occured.
97       *
98       * @return {@link Throwable} if an unexpected exception occured.
99       */
100     public Throwable throwable();
101 
102     /**
103      * Return the broadcaster associated with the {@link AtmosphereResource} this object contains.
104      */
105     public Broadcaster broadcaster();
106 }
