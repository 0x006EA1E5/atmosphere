16  package org.atmosphere.cpr;
17  
18  import javax.servlet.http.HttpSession;
19  import java.io.IOException;
20  import java.util.concurrent.TimeUnit;
21  
22  /**
23   * An {@link AtmosphereResource} encapsulates the mechanism to {@link #suspend()}, {@link #resume()} and
24   * broadcast ({@link #getBroadcaster()}) messages among suspended resources. {@link AtmosphereResource}s are passed to
25   * an instance of {@link AtmosphereHandler} at runtime.
26   *
27   * @author Jeanfrancois Arcand
28   */
29  public interface AtmosphereResource {
30  
31      enum TRANSPORT {POLLING, LONG_POLLING, STREAMING, WEBSOCKET, JSONP, UNDEFINED, SSE, AJAX, HTMLFILE, CLOSE}
32  
33      /**
34       * Return the current {@link TRANSPORT}. The transport value is retrieved using the {@link HeaderConfig#X_ATMOSPHERE_TRANSPORT}
35       * header value.
36       */
37      TRANSPORT transport();
38  
39      /**
40       * Set to true to resume the response after the first broadcast. False by default.
41       *
42       * @param resumeOnBroadcast
43       */
44      AtmosphereResource resumeOnBroadcast(boolean resumeOnBroadcast);
45  
46      /**
47       * Return true if the {@link org.atmosphere.cpr.AtmosphereResource#suspend()} has been invoked.
48       *
49       * @return true if the {@link org.atmosphere.cpr.AtmosphereResource#suspend()} has been invoked
50       */
51      boolean isSuspended();
52  
53      /**
54       * Return true if this AtmosphereResource is resumed after the first broadcast.
55       *
56       * @see org.atmosphere.cpr.AtmosphereResource#resumeOnBroadcast()
57       */
58      boolean resumeOnBroadcast();
59  
60      /**
61       * Return true if this object has been resumed.
62       *
63       * @return true if this object has been resumed
64       */
65      boolean isResumed();
66  
67      /**
68       * Return true if this object has been cancelled.
69       *
70       * @return true if this object has been cancelled
71       */
72      boolean isCancelled();
73  
74      /**
75       * Complete the {@link AtmosphereResponse} and finish/commit it. If the {@link AtmosphereResponse} is in the
76       * process of being resumed, invoking this method has no effect.
77       */
78      AtmosphereResource resume();
79  
80      /**
81       * Suspend the {@link AtmosphereResponse} indefinitely.
82       * Suspending a {@link AtmosphereResponse} will tell the underlying container to avoid recycling objects associated
83       * with the current instance, and also to avoid committing a response.
84       * <p/>
85       * The Framework will output some HTML comments when suspending the response in order to make sure all browsers
86       * work well with suspended responses.
87       */
88      AtmosphereResource suspend();
89  
90      /**
91       * Suspend the {@link AtmosphereResponse}. Suspending a {@link AtmosphereResponse} will tell the underlying
92       * container to avoid recycling objects associated with the current instance, and also to avoid committing response.
93       * Invoking this method when a request is being timed out (e.g. {@link AtmosphereResourceEvent#isResumedOnTimeout}
94       * returns true) has no effect.
95       * <p/>
96       *
97       * @param timeout The maximum amount of time, in milliseconds, a {@link AtmosphereResponse} can be suspended. When
98       *                the timeout expires, the {@link AtmosphereResponse} will be automatically resumed and committed.
99       *                Usage of any methods of a {@link AtmosphereResponse} that times out will throw an {@link IllegalStateException}.
100      */
101     AtmosphereResource suspend(long timeout);
102 
103     /**
104      * Suspend the {@link AtmosphereResponse}. Suspending a {@link AtmosphereResponse} will tell the underlying
105      * container to avoid recycling objects associated with the current instance, and also to avoid committing response.
106      * Invoking this method when a request is being timed out (e.g. {@link AtmosphereResourceEvent#isResumedOnTimeout}
107      * returns true) has no effect.
108      * <p/>
109      *
110      * @param timeout  The maximum amount of time a {@link AtmosphereResponse} can be suspended. When the timeout
111      *                 expires, the {@link AtmosphereResponse} will be automatically resumed and committed. Usage of any
112      *                 methods of a {@link AtmosphereResponse} that times out will throw an {@link IllegalStateException}.
113      * @param timeunit The time unit of the timeout value
114      */
115 
116     AtmosphereResource suspend(long timeout, TimeUnit timeunit);
117 
118     /**
119      * Return the underlying {@link AtmosphereRequest} request.
120      *
121      * @return {@link AtmosphereRequest} the underlying request.
122      */
123     AtmosphereRequest getRequest();
124 
125     /**
126      * Return the {@link AtmosphereResponse}.
127      *
128      * @return {@link AtmosphereResponse} the underlying response.
129      */
130     AtmosphereResponse getResponse();
131 
132     /**
133      * Return the {@link AtmosphereConfig}.
134      *
135      * @return the {@link AtmosphereConfig}
136      */
137     AtmosphereConfig getAtmosphereConfig();
138 
139     /**
140      * Return the current {@link Broadcaster}.
141      *
142      * @return the current {@link Broadcaster}
143      */
144     Broadcaster getBroadcaster();
145 
146     /**
147      * Set the current {@link Broadcaster}. If null, a new Broadcaster will be created with {@link Broadcaster.SCOPE#REQUEST}
148      * if that resource hasn't been suspended yet.
149      *
150      * @param broadcaster
151      * @return this
152      */
153     AtmosphereResource setBroadcaster(Broadcaster broadcaster);
154 
155     /**
156      * Set the {@link Serializer} to use when {@link AtmosphereResource#write} execute the operation.
157      * By default, the {@link Serializer} is null.
158      *
159      * @param s the {@link Serializer}
160      * @return this
161      */
162     AtmosphereResource setSerializer(Serializer s);
163 
164     /**
165      * Write the String. If {@link #resumeOnBroadcast()} is true, the underlying connection will be resumed (@link #resume());
166      *
167      * @param s
168      * @return this
169      */
170     AtmosphereResource write(String s);
171 
172     /**
173      * Write the bytes If {@link #resumeOnBroadcast()} is true, the underlying connection will be resumed (@link #resume());
174      *
175      * @param s
176      * @return this
177      */
178     AtmosphereResource write(byte[] s);
179 
180     /**
181      * Get the {@link Serializer} or null if not defined.
182      *
183      * @return the {@link Serializer} or null if not defined
184      */
185     Serializer getSerializer();
186 
187     /**
188      * Return the current {@link AtmosphereResourceEvent}.
189      */
190     AtmosphereResourceEvent getAtmosphereResourceEvent();
191 
192     /**
193      * Return the associated {@link AtmosphereHandler} associated with this resource.
194      *
195      * @return the associated {@link AtmosphereHandler} associated with this resource
196      */
197     AtmosphereHandler getAtmosphereHandler();
198 
199     /**
200      * Set a message that will be written when the resource times out. Can be an {@link Object} or {@link java.util.concurrent.Callable}.
201      *
202      * @return this
203      */
204     AtmosphereResource writeOnTimeout(Object o);
205 
206     /**
207      * Return the object that will be written when the resource times out.
208      */
209     Object writeOnTimeout();
210 
211     /**
212      * Return the unique ID associated with this AtmosphereResource.
213      *
214      * @return the unique ID associated with this AtmosphereResource
215      */
216     String uuid();
217 
218     /**
219      * Add a {@link AtmosphereResourceEventListener}.
220      *
221      * @param e an instance of AtmosphereResourceEventListener
222      * @return this
223      */
224     AtmosphereResource addEventListener(AtmosphereResourceEventListener e);
225 
226     /**
227      * Remove a {@link AtmosphereResourceEventListener}.
228      *
229      * @param e
230      * @return this
231      */
232     AtmosphereResource removeEventListener(AtmosphereResourceEventListener e);
233 
234     /**
235      * Remove all {@link AtmosphereResourceEventListener}s.
236      *
237      * @return this
238      */
239     AtmosphereResource removeEventListeners();
240 
241     /**
242      * Notify all {@link AtmosphereResourceEventListener}.
243      *
244      * @param e an instance of AtmosphereResourceEvent
245      * @return this
246      */
247     AtmosphereResource notifyListeners(AtmosphereResourceEvent e);
248 
249     /**
250      * Notify all {@link AtmosphereResourceEventListener}s.
251      *
252      * @return this
253      */
254     AtmosphereResource notifyListeners();
255 
256     /**
257      * Return the {@link HttpSession} if supported, null if not
258      *
259      * @return the {@link HttpSession} if supported, null if not
260      */
261     HttpSession session();
262 
263     /**
264      * Return the {@link HttpSession} if supported, and creates it if not already created.
265      *
266      * @return the {@link HttpSession} if supported, and creates it if not already created
267      */
268     HttpSession session(boolean create);
269 
270     /**
271      * Close the underlying connection. Invoking this method will close the underlying connection and resume the
272      * {@link AtmosphereResource}}.
273      */
274     void close() throws IOException;
275 
276     /**
277      * Force binary write and never write String value.
278      * return this
279      */
280     AtmosphereResource forceBinaryWrite(boolean force);
281 
282     /**
283      * Return true when binary write is forced.
284      *
285      * @return true when binary write is forced.
286      */
287     boolean forceBinaryWrite();
288 
289     /**
290      * Initialize an {@link AtmosphereResource}.
291      *
292      * @param config            The {@link org.atmosphere.cpr.AtmosphereConfig}
293      * @param broadcaster       The {@link org.atmosphere.cpr.Broadcaster}.
294      * @param req               The {@link AtmosphereRequest}
295      * @param response          The {@link AtmosphereResource}
296      * @param asyncSupport      The {@link AsyncSupport}
297      * @param atmosphereHandler The {@link AtmosphereHandler}
298      * @return this
299      */
300     public AtmosphereResource initialize(AtmosphereConfig config, Broadcaster broadcaster,
301                                          AtmosphereRequest req, AtmosphereResponse response,
302                                          AsyncSupport asyncSupport, AtmosphereHandler atmosphereHandler);
303 }
