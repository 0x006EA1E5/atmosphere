16  package org.atmosphere.cpr;
17  
18  /**
19   * Factory for {@link AtmosphereResourceSession} instances
20   *
21   * @author uklance (https://github.com/uklance)
22   */
23  public abstract class AtmosphereResourceSessionFactory {
24      // TODO: support IOC
25      // TO Make this Happend, we gonna need to pass an AtmosphereFramework's instance to the constructor and
26      // use the newInstance method, or add support for some sort of @Inject.
27      private static final AtmosphereResourceSessionFactory DEFAULT = new DefaultAtmosphereResourceSessionFactory();
28  
29      /**
30       * Return the default {@link AtmosphereResourceSessionFactory}
31       *
32       * @return the default {@link AtmosphereResourceSessionFactory}
33       */
34      public static AtmosphereResourceSessionFactory getDefault() {
35          return DEFAULT;
36      }
37  
38      /**
39       * Returns the current session associated with the
40       * {@link AtmosphereResource} or, if there is no current session and create
41       * is true, returns a new session.
42       * <p/>
43       * If create is false and the request has no valid HttpSession, this method
44       * returns null.
45       *
46       * @param resource An {@link AtmosphereResource}
47       * @param create   true to create a new session if necessary; false to return
48       *                 null if there's no current session
49       * @return the session associated with this request or null if create is
50       * false and the resource has no valid session
51       */
52      public abstract AtmosphereResourceSession getSession(AtmosphereResource resource, boolean create);
53  
54      /**
55       * Returns the current session associated with the
56       * {@link AtmosphereResource}, or creates one if it does not yet exist.
57       *
58       * @param resource An {@link AtmosphereResource}
59       * @return the current session associated with the
60       * {@link AtmosphereResource}, or creates one if it does not yet
61       * exist.
62       */
63      public AtmosphereResourceSession getSession(AtmosphereResource resource) {
64          return getSession(resource, true);
65      }
66  }
