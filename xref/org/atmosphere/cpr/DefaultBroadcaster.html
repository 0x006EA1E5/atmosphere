69  /**
70   * {@link Broadcaster} implementation.
71   * <p/>
72   * Broadcast messages to suspended response using the caller's Thread.
73   * This basic {@link Broadcaster} use an {@link java.util.concurrent.ExecutorService}
74   * to broadcast message, hence the broadcast operation is asynchronous. Make sure
75   * you block on {@link #broadcast(Object)}.get()} if you need synchronous operations.
76   *
77   * @author Jeanfrancois Arcand
78   */

129     public synchronized void destroy() {
130         if (destroyed.get()) return;
131 
132         try {
133             logger.trace("Broadcaster {} is being destroyed and cannot be re-used", getID());
134 
135             if (BroadcasterFactory.getDefault() != null) {
136                 BroadcasterFactory.getDefault().remove(this, this.getID());
137             }
138 
139             if (currentLifecycleTask != null) {
140                 currentLifecycleTask.cancel(true);
141             }
142             started.set(false);
143             destroyed.set(true);
144 
145             releaseExternalResources();
146             if (notifierFuture != null) {
147                 notifierFuture.cancel(true);
148             }
149 
150             if (asyncWriteFuture != null) {
151                 asyncWriteFuture.cancel(true);
152             }
153 
154             if (bc != null) {
155                 bc.destroy();
156             }
157 
158             if (broadcasterCache != null) {
159                 broadcasterCache.stop();
160             }
161             resources.clear();
162             broadcastOnResume.clear();
163             messages.clear();
164             asyncWriteQueue.clear();
165             delayedBroadcast.clear();
166             broadcasterCache = null;
167         } catch (Throwable t) {
168             logger.error("Unexpected exception during Broadcaster destroy {}", getID(), t);
169         }
170     }

182     public void setScope(SCOPE scope) {
183         this.scope = scope;
184         if (scope != SCOPE.REQUEST) {
185             return;
186         }
187 
188         logger.debug("Changing broadcaster scope for {}. This broadcaster will be destroyed.", getID());
189 
190         synchronized (resources) {
191             try {
192                 // Next, we need to create a new broadcaster per resource.
193                 for (AtmosphereResource<?, ?> resource : resources) {
194                     Broadcaster b = BroadcasterFactory.getDefault()
195                             .get(getClass(), getClass().getSimpleName() + "/" + UUID.randomUUID());
196 
197                     if (DefaultBroadcaster.class.isAssignableFrom(this.getClass())) {
198                         BroadcasterCache cache = bc.getBroadcasterCache().getClass().newInstance();
199                         InjectorProvider.getInjector().inject(cache);
200                         DefaultBroadcaster.class.cast(b).broadcasterCache = cache;
201                     }
202                     resource.setBroadcaster(b);
203                     b.setScope(SCOPE.REQUEST);
204                     if (resource.getAtmosphereResourceEvent().isSuspended()) {
205                         b.addAtmosphereResource(resource);
206                     }
207                     logger.debug("Resource {} not using broadcaster {}", resource, b.getID());
208                 }
209 
210 
211                 destroy();
212             } catch (Exception e) {
213                 logger.error("Failed to set request scope for current resources", e);
214             }
215         }
216     }

228     public synchronized void setID(String id) {
229         if (id == null) {
230             id = getClass().getSimpleName() + "/" + UUID.randomUUID();
231         }
232 
233         Broadcaster b = BroadcasterFactory.getDefault().lookup(this.getClass(), id);
234         if (b != null && b.getScope() == SCOPE.REQUEST) {
235             throw new IllegalStateException("Broadcaster ID already assigned to SCOPE.REQUEST. Cannot change the id");
236         } else if (b != null) {
237             return;
238         }
239 
240         BroadcasterFactory.getDefault().remove(this, name);
241         this.name = id;
242         BroadcasterFactory.getDefault().add(this, name);
243     }

278     public void setBroadcasterLifeCyclePolicy(final BroadcasterLifeCyclePolicy lifeCyclePolicy) {
279         this.lifeCyclePolicy = lifeCyclePolicy;
280         if (currentLifecycleTask != null) {
281             currentLifecycleTask.cancel(false);
282         }
283 
284         if (bc != null && bc.getScheduledExecutorService() == null) {
285             logger.error("No Broadcaster's SchedulerExecutorService has been configured on {}. BroadcasterLifeCyclePolicy won't work.", getID());
286             return;
287         }
288 
289         if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE
290                 || lifeCyclePolicy.getLifeCyclePolicy() == IDLE_RESUME
291                 || lifeCyclePolicy.getLifeCyclePolicy() == IDLE_DESTROY) {
292 
293             int time = lifeCyclePolicy.getTimeout();
294             if (time == -1) {
295                 throw new IllegalStateException("BroadcasterLifeCyclePolicy time is not set");
296             }
297 
298             final AtomicReference<Future<?>> ref = new AtomicReference<Future<?>>();
299             currentLifecycleTask = bc.getScheduledExecutorService().scheduleAtFixedRate(new Runnable() {
300 
301                 @Override
302                 public void run() {
303                     try {
304                         if (resources.isEmpty()) {
305                             if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE) {
306                                 notifyEmptyListener();
307                                 notifyIdleListener();
308 
309                                 releaseExternalResources();
310                                 logger.debug("Applying BroadcasterLifeCyclePolicy IDLE policy to Broadcaster {}", getID());
311                             } else if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE_DESTROY) {
312                                 notifyEmptyListener();
313                                 notifyIdleListener();
314 
315                                 destroy(false);
316                                 logger.debug("Applying BroadcasterLifeCyclePolicy IDLE_DESTROY policy to Broadcaster {}", getID());
317                             }
318                         } else if (lifeCyclePolicy.getLifeCyclePolicy() == IDLE_RESUME) {
319                             notifyIdleListener();
320 
321                             destroy(true);
322                             logger.debug("Applying BroadcasterLifeCyclePolicy IDLE_RESUME policy to Broadcaster {}", getID());
323                         }
324                     } catch (Throwable t) {
325                         if (destroyed.get()) {
326                             logger.trace("Scheduled BroadcasterLifeCyclePolicy exception", t);
327                         } else {
328                             logger.warn("Scheduled BroadcasterLifeCyclePolicy exception", t);
329                         }
330                     }
331                 }
332 
333                 void destroy(boolean resume) {
334                     notifyDestroyListener();
335 
336                     if (resume) {
337                         logger.info("All AtmosphereResource will now be resumed from Broadcaster {}", getID());
338                         resumeAll();
339                     }
340 
341                     DefaultBroadcaster.this.destroy();
342                     /**
343                      * The value may be null if the timeout is too low. Hopefully next execution will
344                      * cancel the task properly.
345                      */
346                     if (ref.get() != null) {
347                         currentLifecycleTask.cancel(true);
348                     }
349                 }
350 
351             }, time, time, lifeCyclePolicy.getTimeUnit());
352             ref.set(currentLifecycleTask);
353         }
354     }

413     protected Runnable getBroadcastHandler() {
414         return new Runnable() {
415             public void run() {
416                 Entry msg = null;
417                 while (started.get()) {
418                     try {
419                         msg = messages.take();
420                         push(msg);
421                     } catch (Throwable ex) {
422                         if (!started.get() || destroyed.get()) {
423                             logger.trace("Failed to submit broadcast handler runnable on shutdown for Broadcaster {}", getID(), ex);
424                         } else {
425                             logger.warn("This message {} will be lost", msg);
426                             logger.debug("Failed to submit broadcast handler runnable to for Broadcaster {}", getID(), ex);
427                         }
428                     }
429                 }
430             }
431         };
432     }
433 
434     protected void start() {
435         if (!started.getAndSet(true)) {
436             broadcasterCache = bc.getBroadcasterCache();
437             broadcasterCache.start();
438 
439             notifierFuture = bc.getExecutorService().submit(getBroadcastHandler());
440             asyncWriteFuture = bc.getAsyncWriteService().submit(getAsyncWriteHandler());
441             setID(name);
442         }
443     }
444 
445     protected void push(Entry entry) {
446 
447         if (destroyed.get()) {
448             return;
449         }
450 
451         String prevMessage = entry.message.toString();
452         if (!delayedBroadcast.isEmpty()) {
453             Iterator<Entry> i = delayedBroadcast.iterator();
454             StringBuilder b = new StringBuilder();
455             while (i.hasNext()) {
456                 Entry e = i.next();
457                 e.future.cancel(true);
458                 try {
459                     // Append so we do a single flush
460                     if (e.message instanceof String
461                             && entry.message instanceof String) {
462                         b.append(e.message);
463                     } else {
464                         push(e);
465                     }
466                 } finally {
467                     i.remove();
468                 }
469             }
470 
471             if (b.length() > 0) {
472                 entry.message = b.append(entry.message).toString();
473             }
474         }
475 
476         if (resources.isEmpty()) {
477             logger.debug("Broadcaster {} doesn't have any associated resource", getID());
478             trackBroadcastMessage(null, entry.message);
479             if (entry.future != null) {
480                 entry.future.done();
481             }
482             return;
483         }
484 
485         Object finalMsg = translate(entry.message);
486         entry.message = finalMsg;
487         try {
488             if (entry.multipleAtmoResources == null) {
489                 for (AtmosphereResource<?, ?> r : resources) {
490                     finalMsg = perRequestFilter(r, entry);
491 
492                     if (finalMsg == null) {
493                         logger.debug("Skipping broadcast delivery resource {} ", r);
494                         continue;
495                     }
496 
497                     if (entry.writeLocally) {
498                         queueWriteIO(r, finalMsg, entry);
499                     }
500                 }
501             } else if (entry.multipleAtmoResources instanceof AtmosphereResource<?, ?>) {
502                 finalMsg = perRequestFilter((AtmosphereResource<?, ?>) entry.multipleAtmoResources, entry);
503 
504                 if (finalMsg == null) {
505                     logger.debug("Skipping broadcast delivery resource {} ", entry.multipleAtmoResources);
506                     return;
507                 }
508 
509                 if (entry.writeLocally) {
510                     queueWriteIO((AtmosphereResource<?, ?>) entry.multipleAtmoResources, finalMsg, entry);
511                 }
512             } else if (entry.multipleAtmoResources instanceof Set) {
513                 Set<AtmosphereResource<?, ?>> sub = (Set<AtmosphereResource<?, ?>>) entry.multipleAtmoResources;
514                 for (AtmosphereResource<?, ?> r : sub) {
515                     finalMsg = perRequestFilter(r, entry);
516 
517                     if (finalMsg == null) {
518                         logger.debug("Skipping broadcast delivery resource {} ", r);
519                         continue;
520                     }
521 
522                     if (entry.writeLocally) {
523                         queueWriteIO(r, finalMsg, entry);
524                     }
525                 }
526             }
527             entry.message = prevMessage;
528         } catch (InterruptedException ex) {
529             logger.debug(ex.getMessage(), ex);
530         }
531     }
532 
533     protected void queueWriteIO(AtmosphereResource<?, ?> r, Object finalMsg, Entry entry) throws InterruptedException {
534         asyncWriteQueue.put(new AsyncWriteToken(r, finalMsg, entry.future));
535     }
536 
537     protected Object perRequestFilter(AtmosphereResource<?, ?> r, Entry msg) {
538         Object finalMsg = msg.message;
539 
540         if (AtmosphereResourceImpl.class.isAssignableFrom(r.getClass())) {
541             if (AtmosphereResourceImpl.class.cast(r).isInScope()) {
542                 if (r.getRequest() instanceof HttpServletRequest && bc.hasPerRequestFilters()) {
543                     Object message = msg.originalMessage;
544                     BroadcastAction a = bc.filter((HttpServletRequest) r.getRequest(), (HttpServletResponse) r.getResponse(), message);
545                     if (a.action() == BroadcastAction.ACTION.ABORT) {
546                         return null;
547                     }
548                     if (a.message() != msg.originalMessage) {
549                         finalMsg = a.message();
550                     }
551                 }
552                 trackBroadcastMessage(r, finalMsg);
553             } else {
554                 // The resource is no longer valid.
555                 removeAtmosphereResource(r);
556                 BroadcasterFactory.getDefault().removeAllAtmosphereResource(r);
557             }
558 
559         }
560         return finalMsg;
561     }

563     private Object translate(Object msg) {
564         if (Callable.class.isAssignableFrom(msg.getClass())) {
565             try {
566                 return Callable.class.cast(msg).call();
567             } catch (Exception e) {
568                 logger.error("failed to cast message: " + msg, e);
569             }
570         }
571         return msg;
572     }

574     protected void executeAsyncWrite(final AtmosphereResource<?, ?> resource, final Object msg, final BroadcasterFuture future) {
575         boolean notifyListeners = true;
576         try {
577             final AtmosphereResourceEventImpl event = (AtmosphereResourceEventImpl) resource.getAtmosphereResourceEvent();
578             event.setMessage(msg);
579 
580             // Check again to make sure we are still valid. Remove and silently ignore.
581             if (!AtmosphereResourceImpl.class.cast(resource).isInScope()) {
582                 resources.remove(resource);
583                 return;
584             }
585 
586             try {
587                 HttpServletRequest.class.cast(resource.getRequest())
588                         .setAttribute(MAX_INACTIVE, System.currentTimeMillis());
589             } catch (Throwable t) {
590                 logger.error("Invalid AtmosphereResource state {}", event);
591                 logger.error("If you are using Tomcat 7.0.22 and lower, your most probably hitting http://is.gd/NqicFT");
592                 logger.error("", t);
593                 // The Request/Response associated with the AtmosphereResource has already been written and commited
594                 removeAtmosphereResource(resource);
595                 BroadcasterFactory.getDefault().removeAllAtmosphereResource(resource);
596                 event.setCancelled(true);
597                 event.setThrowable(t);
598                 return;
599             }
600 
601             broadcast(resource, event);
602         } finally {
603             if (notifyListeners) {
604                 resource.notifyListeners();
605             }
606 
607             if (future != null) {
608                 future.done();
609             }
610         }
611     }

613     protected Runnable getAsyncWriteHandler() {
614         return new Runnable() {
615             public void run() {
616                 AsyncWriteToken token = null;
617                 try {
618                     token = asyncWriteQueue.take();
619                     synchronized (token.resource) {
620 
621                         // We want this thread to wait for the write operation to happens to kept the order
622                         bc.getAsyncWriteService().submit(this);
623 
624                         // If the resource is no longer in scope, skip the processing.
625                         if (AtmosphereResourceImpl.class.cast(token.resource).isInScope()) {
626                             executeAsyncWrite(token.resource, token.msg, token.future);
627                         }
628                     }
629                 } catch (Throwable ex) {
630                     if (!started.get() || destroyed.get()) {
631                         logger.trace("Failed to execute a write operation. Broadcaster is destroyed or not yet started for Broadcaster {}", getID(), ex);
632                     } else {
633                         logger.warn("This message {} will be lost", token.msg);
634                         logger.debug("Failed to execute a write operation for Broadcaster {}", getID(), ex);
635                     }
636                 }
637             }
638         };
639     }

641     protected void checkCachedAndPush(final AtmosphereResource<?, ?> r, final AtmosphereResourceEvent e) {
642         retrieveTrackedBroadcast(r, e);
643         if (e.getMessage() instanceof List && !((List) e.getMessage()).isEmpty()) {
644             HttpServletRequest.class.cast(r.getRequest()).setAttribute(CACHED, "true");
645             // Must make sure execute only one thread
646             synchronized (r) {
647                 broadcast(r, e);
648             }
649         }
650     }
651 
652     protected boolean retrieveTrackedBroadcast(final AtmosphereResource<?, ?> r, final AtmosphereResourceEvent e) {
653         List<?> missedMsg = broadcasterCache.retrieveFromCache(r);
654         if (!missedMsg.isEmpty()) {
655             e.setMessage(missedMsg);
656             return true;
657         }
658         return false;
659     }
660 
661     protected void trackBroadcastMessage(final AtmosphereResource<?, ?> r, Object msg) {
662         broadcasterCache.addToCache(r, msg);
663     }

665     protected void broadcast(final AtmosphereResource<?, ?> r, final AtmosphereResourceEvent e) {
666         try {
667             r.getAtmosphereHandler().onStateChange(e);
668         } catch (Throwable t) {
669             onException(t, r);
670         }
671     }
672 
673     protected void onException(Throwable t, final AtmosphereResource<?, ?> r) {
674         logger.debug("onException()", t);
675 
676         // Remove to prevent other broadcast to re-use it.
677         removeAtmosphereResource(r);
678 
679         final AtmosphereResourceEventImpl event = (AtmosphereResourceEventImpl) r.getAtmosphereResourceEvent();
680         event.setThrowable(t);
681 
682         r.notifyListeners(event);
683         r.removeEventListeners();
684 
685         /**
686          * Make sure we resume the connection on every IOException.
687          */
688         if (bc != null && bc.getAsyncWriteService() != null) {
689             bc.getAsyncWriteService().execute(new Runnable() {
690                 @Override
691                 public void run() {
692                     try {
693                         r.resume();
694                     } catch (Throwable t) {
695                         logger.warn("Was unable to resume a corrupted AtmosphereResource {}", r);
696                         logger.warn("Cause", t);
697                     }
698                 }
699             });
700         } else {
701             r.resume();
702         }
703     }

711     /**
712      * {@inheritDoc}
713      */
714     @Override
715     public <T> Future<T> broadcast(T msg) {
716 
717         if (destroyed.get()) {
718             logger.debug(DESTROYED, getID(), "broadcast(T msg)");
719             return null;
720         }
721

823 
824         if (destroyed.get()) {
825             logger.debug(DESTROYED, getID(), "addAtmosphereResource(AtmosphereResource<?, ?> r");
826             return r;
827         }
828 
829         start();
830         if (scope == SCOPE.REQUEST && requestScoped.getAndSet(true)) {
831             throw new IllegalStateException("Broadcaster " + this
832                     + " cannot be used as its scope is set to REQUEST");
833         }
834 
835         // To avoid excessive synchronization, we allow resources.size() to get larger that maxSuspendResource
836         if (maxSuspendResource.get() > 0 && resources.size() >= maxSuspendResource.get()) {
837             // Resume the first in.
838             if (policy == POLICY.FIFO) {
839                 // TODO handle null return from poll()
840                 AtmosphereResource<?, ?> resource = resources.poll();
841                 try {
842                     logger.warn("Too many resource. Forcing resume of {} ", resource);
843                     resource.resume();
844                 } catch (Throwable t) {
845                     logger.warn("failed to resume resource {} ", resource, t);
846                 }
847             } else if (policy == POLICY.REJECT) {
848                 throw new RejectedExecutionException(String.format("Maximum suspended AtmosphereResources %s", maxSuspendResource));
849             }
850         }
851 
852         if (resources.contains(r)) {
853             return r;
854         }
855 
856         // Re-add yourself
857         if (resources.isEmpty()) {
858             BroadcasterFactory.getDefault().add(this, name);
859         }
860 
861         resources.add(r);
862         checkCachedAndPush(r, r.getAtmosphereResourceEvent());
863         return r;
