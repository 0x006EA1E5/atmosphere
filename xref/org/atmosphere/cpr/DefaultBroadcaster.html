127     public Broadcaster initialize(String name, URI uri, AtmosphereConfig config) {
128         this.name = name;
129         this.uri = uri;
130         this.config = config;
131 
132         bc = createBroadcasterConfig(config);
133         String s = config.getInitParameter(BROADCASTER_CACHE_STRATEGY);
134         if (s != null) {
135             logger.warn("{} is no longer supported. Use BroadcastInterceptor instead. By default the original message will be cached.", BROADCASTER_CACHE_STRATEGY);
136         }
137         s = config.getInitParameter(OUT_OF_ORDER_BROADCAST);
138         if (s != null) {
139             outOfOrderBroadcastSupported.set(Boolean.valueOf(s));
140         }
141 
142         s = config.getInitParameter(BROADCASTER_WAIT_TIME);
143         if (s != null) {
144             waitTime = Integer.valueOf(s);
145         }
146 
147         s = config.getInitParameter(WRITE_TIMEOUT);
148         if (s != null) {
149             writeTimeoutInSecond = Integer.valueOf(s);
150         }
151         if (outOfOrderBroadcastSupported.get()) {
152             logger.trace("{} supports Out Of Order Broadcast: {}", name, outOfOrderBroadcastSupported.get());
153         }
154         initialized.set(true);
155         backwardCompatible = Boolean.parseBoolean(config.getInitParameter(BACKWARD_COMPATIBLE_WEBSOCKET_BEHAVIOR));
156         cacheOnIOFlushException = config.getInitParameter(CACHE_MESSAGE_ON_IO_FLUSH_EXCEPTION, true);
157         sharedListeners = config.getInitParameter(BROADCASTER_SHAREABLE_LISTENERS, false);
158 
159         if (sharedListeners) {
160             broadcasterListeners = config.getBroadcasterFactory().broadcasterListeners();
161         } else {
162             broadcasterListeners = new ConcurrentLinkedQueue<BroadcasterListener>();
163         }
164 
165         candidateForPoolable = PoolableBroadcasterFactory.class.isAssignableFrom(config.getBroadcasterFactory().getClass());
166 
167         return this;
168     }

184     @Override
185     public synchronized void destroy() {
186         try {
187             logger.trace("Broadcaster {} will be pooled: {}", getID(), candidateForPoolable);
188             if (!candidateForPoolable) {
189                 if (notifyOnPreDestroy()) return;
190 
191                 if (destroyed.getAndSet(true)) return;
192 
193                 logger.trace("Broadcaster {} is being destroyed and cannot be re-used. Policy was {}", getID(), policy);
194                 logger.trace("Broadcaster {} is being destroyed and cannot be re-used. Resources are {}", getID(), resources);
195 
196                 started.set(false);
197 
198                 releaseExternalResources();
199                 killReactiveThreads();
200 
201                 if (bc != null) {
202                     bc.destroy();
203                 }
204                 lifeCycleListeners.clear();
205                 delayedBroadcast.clear();
206                 if (!sharedListeners) {
207                     broadcasterListeners.clear();
208                 }
209             }
210 
211             resources.clear();
212             broadcastOnResume.clear();
213             messages.clear();
214             writeQueues.clear();
215 
216             if (config.getBroadcasterFactory() != null) {
217                 config.getBroadcasterFactory().remove(this, this.getID());
218             }
219         } catch (Throwable t) {
220             logger.error("Unexpected exception during Broadcaster destroy {}", getID(), t);
221         }
222     }

229     @Override
230     public void setScope(SCOPE scope) {
231         if (destroyed.get()) {
232             logger.debug(DESTROYED, getID(), "setScope");
233             return;
234         }
235 
236         this.scope = scope;
237         if (scope != SCOPE.REQUEST) {
238             return;
239         }
240 
241         logger.debug("Changing broadcaster scope for {}. This broadcaster will be destroyed.", getID());
242         synchronized (resources) {
243             try {
244                 // Next, we need to create a new broadcaster per resource.
245                 for (AtmosphereResource resource : resources) {
246                     Broadcaster b = config.getBroadcasterFactory()
247                             .get(getClass(), getClass().getSimpleName() + "/" + config.uuidProvider().generateUuid());
248 
249                     /**
250                      * REQUEST_SCOPE means one BroadcasterCache per Broadcaster,
251                      */
252                     if (DefaultBroadcaster.class.isAssignableFrom(this.getClass())) {
253                         BroadcasterCache cache = config.framework().newClassInstance(BroadcasterCache.class, bc.getBroadcasterCache().getClass());
254                         cache.configure(config);
255                         b.getBroadcasterConfig().setBroadcasterCache(cache);
256                     }
257 
258                     resource.setBroadcaster(b);
259                     b.setScope(SCOPE.REQUEST);
260                     if (resource.getAtmosphereResourceEvent().isSuspended()) {
261                         b.addAtmosphereResource(resource);
262                     }
263                     logger.debug("Resource {} not using broadcaster {}", resource, b.getID());
264                 }
265 
266                 // Do not destroy because this is a new Broadcaster
267                 if (resources.isEmpty()) {
268                     return;
269                 }
270 
271                 destroy();
272             } catch (Exception e) {
273                 logger.error("Failed to set request scope for current resources", e);
274             }
275         }
276     }

283     @Override
284     public synchronized void setID(String id) {
285         if (id == null) {
286             id = getClass().getSimpleName() + "/" + config.uuidProvider().generateUuid();
287         }
288 
289         if (config.getBroadcasterFactory() == null)
290             return; // we are shutdown or destroyed, but someone still reference
291 
292         Broadcaster b = config.getBroadcasterFactory().lookup(this.getClass(), id);
293         if (b != null && b.getScope() == SCOPE.REQUEST) {
294             throw new IllegalStateException("Broadcaster ID already assigned to SCOPE.REQUEST. Cannot change the id");
295         } else if (b != null) {
296             return;
297         }
298 
299         config.getBroadcasterFactory().remove(this, name);
300         this.name = id;
301         config.getBroadcasterFactory().add(this, name);
302 
303         bc.broadcasterID(name);
304     }

399     protected Runnable getBroadcastHandler() {
400         return new Runnable() {
401             public void run() {
402                 while (!isDestroyed()) {
403                     Deliver msg = null;
404                     try {
405                         msg = messages.poll(waitTime, TimeUnit.MILLISECONDS);
406                         if (msg == null) {
407                             dispatchThread.decrementAndGet();
408                             return;
409                         }
410                     } catch (InterruptedException ex) {
411                         logger.trace("{} got interrupted for Broadcaster {}", Thread.currentThread().getName(), getID());
412                         logger.trace("", ex);
413                         return;
414                     } finally {
415                         if (outOfOrderBroadcastSupported.get()) {
416                             bc.getExecutorService().submit(this);
417                         }
418                     }
419 
420                     try {
421                         logger.trace("{} is about to broadcast {}", getID(), msg);
422                         push(msg);
423                     } catch (Throwable ex) {
424                         if (!started.get() || destroyed.get()) {
425                             logger.trace("Failed to submit broadcast handler runnable on shutdown for Broadcaster {}", getID(), ex);
426                             return;
427                         } else {
428                             logger.warn("This message {} will be lost", msg);
429                             logger.warn("Failed to submit broadcast handler runnable to for Broadcaster" + getID(), ex);
430                         }
431                     } finally {
432                         if (outOfOrderBroadcastSupported.get()) {
433                             return;
434                         }
435                     }
436                 }
437             }
438         };
439     }

441     protected Runnable getAsyncWriteHandler(final WriteQueue writeQueue) {
442         return new Runnable() {
443             public void run() {
444                 while (!isDestroyed()) {
445                     AsyncWriteToken token = null;
446                     try {
447                         token = writeQueue.queue.poll(waitTime, TimeUnit.MILLISECONDS);
448                         if (token == null && !outOfOrderBroadcastSupported.get()) {
449                             synchronized (writeQueue) {
450                                 if (writeQueue.queue.isEmpty()) {
451                                     writeQueue.monitored.set(false);
452                                     writeQueues.remove(writeQueue.uuid);
453                                     return;
454                                 }
455                             }
456                         } else if (token == null) {
457                             return;
458                         }
459                     } catch (InterruptedException ex) {
460                         logger.trace("{} got interrupted for Broadcaster {}", Thread.currentThread().getName(), getID());
461                         logger.trace("", ex);
462                         return;
463                     } finally {
464                         if (!bc.getAsyncWriteService().isShutdown() && outOfOrderBroadcastSupported.get()) {
465                             bc.getAsyncWriteService().submit(this);
466                         }
467                     }
468 
469                     // Shield us from https://github.com/Atmosphere/atmosphere/issues/1187
470                     if (token != null) {
471                         synchronized (token.resource) {
472                             try {
473                                 logger.trace("About to write to {}", token.resource);
474                                 executeAsyncWrite(token);
475                             } catch (Throwable ex) {
476                                 if (!started.get() || destroyed.get()) {
477                                     logger.trace("Failed to execute a write operation. Broadcaster is destroyed or not yet started for Broadcaster {}", getID(), ex);
478                                     return;
479                                 } else {
480                                     try {
481                                         if (token != null) {
482                                             logger.warn("This message {} will be lost for AtmosphereResource {}, adding it to the BroadcasterCache",
483                                                     token.originalMessage, token.resource != null ? token.resource.uuid() : "null");
484                                             cacheLostMessage(token.resource, token, true);
485                                         }
486                                     } finally {
487                                         if (token != null) {
488                                             removeAtmosphereResource(token.resource, false);
489                                         }
490                                         logger.warn("Failed to execute a write operation for Broadcaster " + getID(), ex);
491                                     }
492                                 }
493                             } finally {
494                                 if (!bc.getAsyncWriteService().isShutdown() && outOfOrderBroadcastSupported.get()) {
495                                     return;
496                                 }
497                             }
498                         }
499                     }
500                 }
501             }
502         };
503     }

505     protected void start() {
506         if (!initialized.get()) {
507             logger.warn("Broadcaster {} not initialized", getID());
508         }
509 
510         if (!started.getAndSet(true)) {
511             bc.getBroadcasterCache().start();
512 
513             // Only start if we know a child haven't started them.
514             if (notifierFuture == null && asyncWriteFuture == null) {
515                 spawnReactor();
516             }
517         }
518     }

520     protected void spawnReactor() {
521         killReactiveThreads();
522 
523         int threads = outOfOrderBroadcastSupported.get() ? reactiveThreadsCount() : 1;
524         notifierFuture = new Future<?>[threads];
525 
526         if (outOfOrderBroadcastSupported.get()) {
527             asyncWriteFuture = new Future<?>[threads];
528             for (int i = 0; i < threads; i++) {
529                 notifierFuture[i] = bc.getExecutorService().submit(getBroadcastHandler());
530                 asyncWriteFuture[i] = bc.getAsyncWriteService().submit(getAsyncWriteHandler(uniqueWriteQueue));
531             }
532         } else {
533             notifierFuture[0] = bc.getExecutorService().submit(getBroadcastHandler());
534         }
535         dispatchThread.set(threads);
536     }
537 
538     protected void killReactiveThreads() {
539         if (notifierFuture != null) {
540             for (Future<?> f : notifierFuture) {
541                 if (f != null)
542                     f.cancel(false);
543             }
544         }
545 
546         if (asyncWriteFuture != null) {
547             for (Future<?> f : asyncWriteFuture) {
548                 if (f != null)
549                     f.cancel(false);
550             }
551         }
552     }

572     protected void deliverPush(Deliver deliver, boolean rec) {
573         recentActivity.set(true);
574 
575         String prevMessage = deliver.message.toString();
576         if (rec && !delayedBroadcast.isEmpty()) {
577             Iterator<Deliver> i = delayedBroadcast.iterator();
578             StringBuilder b = new StringBuilder();
579             while (i.hasNext()) {
580                 Deliver e = i.next();
581                 e.future.cancel(true);
582                 try {
583                     // Append so we do a single flush
584                     if (e.message instanceof String
585                             && deliver.message instanceof String) {
586                         b.append(e.message);
587                     } else {
588                         deliverPush(e, false);
589                     }
590                 } finally {
591                     i.remove();
592                 }
593             }
594 
595             if (b.length() > 0) {
596                 deliver.message = b.append(deliver.message).toString();
597             }
598         }
599 
600         Object finalMsg = callable(deliver.message);
601         if (finalMsg == null) {
602             logger.error("Callable exception. Please catch all exceptions from your callable. Message {} will be lost and all AtmosphereResource " +
603                     "associated with this Broadcaster resumed.", deliver.message);
604             entryDone(deliver.future);
605             switch (deliver.type) {
606                 case ALL:
607                     synchronized (resources) {
608                         for (AtmosphereResource r : resources) {
609                             if (Utils.resumableTransport(r.transport()))
610                                 try {
611                                     r.resume();
612                                 } catch (Throwable t) {
613                                     logger.trace("resumeAll", t);
614                                 }
615                         }
616                     }
617                     break;
618                 case RESOURCE:
619                     deliver.resource.resume();
620                     break;
621                 case SET:
622                     for (AtmosphereResource r : deliver.resources) {
623                         r.resume();
624                     }
625                     break;
626             }
627             return;
628         }
629 
630         notifyOnMessage(deliver);
631         Object prevM = deliver.originalMessage;
632         deliver.originalMessage = (deliver.originalMessage != deliver.message ? callable(deliver.originalMessage) : finalMsg);
633 
634         if (deliver.originalMessage == null) {
635             logger.trace("Broadcasted message was null {}", prevM);
636             entryDone(deliver.future);
637             return;
638         }
639 
640         deliver.message = finalMsg;
641 
642         Map<String, CacheMessage> cacheForSet = deliver.type == Deliver.TYPE.SET ? new HashMap<String, CacheMessage>() : null;
643         // We cache first, and if the broadcast succeed, we will remove it.
644         switch (deliver.type) {
645             case ALL:
646                 deliver.cache = bc.getBroadcasterCache().addToCache(getID(), BroadcasterCache.NULL, new BroadcastMessage(deliver.originalMessage));
647                 break;
648             case RESOURCE:
649                 deliver.cache = bc.getBroadcasterCache().addToCache(getID(), deliver.resource.uuid(), new BroadcastMessage(deliver.originalMessage));
650                 break;
651             case SET:
652                 for (AtmosphereResource r : deliver.resources) {
653                     cacheForSet.put(r.uuid(), bc.getBroadcasterCache().addToCache(getID(), r.uuid(), new BroadcastMessage(deliver.originalMessage)));
654                 }
655                 break;
656         }
657 
658         if (resources.isEmpty()) {
659             logger.trace("No resource available for {} and message {}", getID(), finalMsg);
660             entryDone(deliver.future);
661             if (cacheForSet != null) {
662                 cacheForSet.clear();
663             }
664             return;
665         }
666 
667         try {
668             if (logger.isTraceEnabled()) {
669                 for (AtmosphereResource r : resources) {
670                     logger.trace("AtmosphereResource {} available for {}", r.uuid(), deliver.message);
671                 }
672             }
673 
674             boolean hasFilters = bc.hasPerRequestFilters();
675             Object beforeProcessingMessage = deliver.message;
676             switch (deliver.type) {
677                 case ALL:
678                     AtomicInteger count = new AtomicInteger(resources.size());
679 
680                     for (AtmosphereResource r : resources) {
681                         deliver.message = beforeProcessingMessage;
682                         boolean deliverMessage = perRequestFilter(r, deliver);
683 
684                         if (endBroadcast(deliver, r, deliver.cache, deliverMessage)) continue;
685 
686                         if (deliver.writeLocally) {
687                             queueWriteIO(r, hasFilters ? new Deliver(r, deliver) : deliver, count);
688                         }
689                     }
690                     break;
691                 case RESOURCE:
692                     boolean deliverMessage = perRequestFilter(deliver.resource, deliver);
693 
694                     if (endBroadcast(deliver, deliver.resource, deliver.cache,  deliverMessage)) return;
695 
696                     if (deliver.writeLocally) {
697                         queueWriteIO(deliver.resource, deliver, new AtomicInteger(1));
698                     }
699                     break;
700                 case SET:
701                     count = new AtomicInteger(deliver.resources.size());
702 
703                     for (AtmosphereResource r : deliver.resources) {
704                         deliver.message = beforeProcessingMessage;
705                         deliverMessage = perRequestFilter(r, deliver);
706 
707                         CacheMessage cacheMsg = cacheForSet.remove(r.uuid());
708 
709                         if (endBroadcast(deliver, r, cacheMsg, deliverMessage)) continue;
710 
711                         if (deliver.writeLocally) {
712                             queueWriteIO(r, new Deliver(r, deliver, cacheMsg), count);
713                         }
714                     }
715                     break;
716             }
717 
718             deliver.message = prevMessage;
719         } catch (InterruptedException ex) {
720             logger.debug(ex.getMessage(), ex);
721             if (cacheForSet != null) {
722                 cacheForSet.clear();
723             }
724         }
725     }

727     protected boolean endBroadcast(Deliver deliver, AtmosphereResource r, CacheMessage cacheMsg, boolean deliverMessage) {
728         if (!deliverMessage || deliver.message == null) {
729             logger.debug("Skipping broadcast delivery {} for resource {} ", deliver.message, deliver.resource != null ? deliver.resource.uuid() : "null");
730             bc.getBroadcasterCache().clearCache(getID(), r.uuid(), cacheMsg);
731             entryDone(deliver.future);
732 
733             return true;
734         }
735         return false;

738     protected void queueWriteIO(AtmosphereResource r, Deliver deliver, AtomicInteger count) throws InterruptedException {
739         if (deliver.async) {
740             // The onStateChange/onRequest may change the isResumed value, hence we need to make sure only one thread flip
741             // the switch to garantee the Entry will be cached in the order it was broadcasted.
742             // Without synchronizing we may end up with a out of order BroadcasterCache queue.
743             if (!bc.getBroadcasterCache().getClass().equals(BroadcasterCache.DEFAULT.getClass().getName())) {
744                 if (r.isResumed() || r.isCancelled()) {
745                     logger.trace("AtmosphereResource {} has been resumed or cancelled, unable to Broadcast message {}", r.uuid(), deliver.message);
746 
747                     /**
748                      * https://github.com/Atmosphere/atmosphere/issues/1886
749                      * Before caching the message, double check if the client has reconnected, and if true, send the
750                      * cached message.
751                      */
752                     AtmosphereResource r2 = config.resourcesFactory().find(r.uuid());
753                     logger.trace("Found an AtmosphereResource {} in state {}", r2, r.isSuspended());
754                     if (r2 != null && r2.isSuspended() && r.hashCode() != r2.hashCode()) {
755                         // Prevent other Broadcast to happens
756                         removeAtmosphereResource(r2);
757                         checkCachedAndPush(r2, r2.getAtmosphereResourceEvent());
758                     }
759                     return;
760                 }
761             }
762 
763             AsyncWriteToken w = new AsyncWriteToken(r, deliver.message, deliver.future, deliver.originalMessage, deliver.cache, count);
764             if (!outOfOrderBroadcastSupported.get()) {
765                 WriteQueue writeQueue = writeQueues.get(r.uuid());
766                 if (writeQueue == null) {
767                     writeQueue = new WriteQueue(r.uuid());
768                     writeQueues.put(r.uuid(), writeQueue);
769                 }
770 
771                 writeQueue.queue.put(w);
772                 synchronized (writeQueue) {
773                     if (!writeQueue.monitored.getAndSet(true)) {
774                         logger.trace("Broadcaster {} is about to queueWriteIO for AtmosphereResource {}", name, r.uuid());
775                         bc.getAsyncWriteService().submit(getAsyncWriteHandler(writeQueue));
776                     }
777                 }
778             } else {
779                 uniqueWriteQueue.queue.offer(w);
780             }
781         } else {
782             executeBlockingWrite(r, deliver, count);
783         }
784     }

788         synchronized (r) {
789             executeAsyncWrite(new AsyncWriteToken(r, deliver.message, deliver.future, deliver.originalMessage, deliver.cache, count));

840     protected void executeAsyncWrite(final AsyncWriteToken token) {
841         boolean notifyListeners = true;
842         boolean lostCandidate = false;
843 
844         if (token.resource == null) throw new NullPointerException();
845 
846         final AtmosphereResourceEventImpl event = (AtmosphereResourceEventImpl) token.resource.getAtmosphereResourceEvent();
847         final AtmosphereResourceImpl r = AtmosphereResourceImpl.class.cast(token.resource);
848         final boolean willBeResumed = Utils.resumableTransport(r.transport());
849         List<AtmosphereResourceEventListener> listeners = willBeResumed ? new ArrayList() : EMPTY_LISTENERS;
850         final AtmosphereRequest request = r.getRequest(false);
851         try {
852 
853             event.setMessage(token.msg);
854 
855             // Make sure we cache the message in case the AtmosphereResource has been cancelled, resumed or the client disconnected.
856             if (!isAtmosphereResourceValid(r)) {
857                 logger.trace("AtmosphereResource {} state is invalid for Broadcaster {}. Message will be cached", r.uuid(), name);
858                 removeAtmosphereResource(r, false);
859                 return;
860             }
861 
862             bc.getBroadcasterCache().clearCache(getID(), r.uuid(), token.cache);
863             try {
864                 request.setAttribute(getID(), token.future);
865                 request.setAttribute(MAX_INACTIVE, System.currentTimeMillis());
866                 request.setAttribute(usingTokenIdForAttribute, token);
867 
868                 if (willBeResumed && !r.atmosphereResourceEventListener().isEmpty()) {
869                     listeners.addAll(r.atmosphereResourceEventListener());
870                 }
871                 prepareInvokeOnStateChange(r, event);
872             } catch (Throwable t) {
873                 logger.debug("Invalid AtmosphereResource state {}. The connection has been remotely" +
874                         " closed and message {} will be added to the configured BroadcasterCache for later retrieval", r.uuid(), event.getMessage());
875                 logger.trace("If you are using Tomcat 7.0.22 and lower, you're most probably hitting http://is.gd/NqicFT");
876                 logger.trace("ApplicationConfig.CACHE_MESSAGE_ON_IO_FLUSH_EXCEPTION {}", cacheOnIOFlushException, t);
877 
878                 lostCandidate = cacheOnIOFlushException ? cacheOnIOFlushException : cacheMessageOnIOException(t);
879                 // The Request/Response associated with the AtmosphereResource has already been written and commited
880                 removeAtmosphereResource(r, false);
881                 r.removeFromAllBroadcasters();
882                 event.setCancelled(true);
883                 event.setThrowable(t);
884                 r.setIsInScope(false);
885                 return;
886             }
887         } finally {
888             if (notifyListeners) {
889                 // Long Polling listener will be cleared when the resume() is called.
890                 if (willBeResumed) {
891                     event.setMessage(token.msg);
892                     for (AtmosphereResourceEventListener e : listeners) {
893                         e.onBroadcast(event);
894                     }
895                     // Listener wil be called later
896                 } else if (!event.isResumedOnTimeout()) {
897                     r.notifyListeners();
898                 }
899             }
900 
901             if (token.lastBroadcasted()) {
902                 notifyBroadcastListener();
903             }
904 
905             if (token.future != null) token.future.done();
906 
907             if (lostCandidate) {
908                 cacheLostMessage(r, token, true);
909             }
910 
911             try {
912                 request.removeAttribute(getID());
913                 request.removeAttribute(usingTokenIdForAttribute);
914             } catch (NullPointerException ex) {
915                 logger.trace("NPE after the message has been written for {}", r.uuid());
916             }
917             token.destroy();
918         }
919     }

930     protected boolean checkCachedAndPush(final AtmosphereResource r, final AtmosphereResourceEvent e) {
931         boolean cache = retrieveTrackedBroadcast(r, e);
932 
933         if (!cache) return false;
934 
935         if (!((List) e.getMessage()).isEmpty()) {
936             logger.debug("Sending cached message {} to {}", e.getMessage(), r.uuid());
937 
938             List<Object> cacheMessages = (List) e.getMessage();
939             BroadcasterFuture<Object> f = new BroadcasterFuture<Object>(e.getMessage(), 1);
940             LinkedList<Object> filteredMessage = new LinkedList<Object>();
941             LinkedList<Object> filteredMessageClone = null;
942             Deliver deliver;
943             Object newMessage;
944             for (Object o : cacheMessages) {
945                 newMessage = filter(o);
946                 if (newMessage == null) {
947                     continue;
948                 }
949 
950                 deliver = new Deliver(newMessage, r, f, o);
951                 // Can be aborted by a Filter
952                 if (!perRequestFilter(r, deliver)) {
953                     continue;
954                 }
955 
956                 if (deliver.message != null) {
957                     filteredMessage.addLast(deliver.message);
958                 }
959             }
960 
961             if (filteredMessage.isEmpty()) {
962                 return false;
963             }
964             e.setMessage(filteredMessage);
965 
966             final boolean willBeResumed = Utils.resumableTransport(r.transport());
967 
968             if (willBeResumed) {
969                 filteredMessageClone = (LinkedList<Object>) filteredMessage.clone();
970             }
971 
972             List<AtmosphereResourceEventListener> listeners = willBeResumed ? new ArrayList() : EMPTY_LISTENERS;
973             AtmosphereResourceImpl rImpl = AtmosphereResourceImpl.class.cast(r);
974             if (willBeResumed && !rImpl.atmosphereResourceEventListener().isEmpty()) {
975                 listeners.addAll(rImpl.atmosphereResourceEventListener());
976             }
977 
978             // Must make sure execute only one thread
979             synchronized (rImpl) {
980                 try {
981                     rImpl.getRequest().setAttribute(CACHED, "true");
982                     prepareInvokeOnStateChange(r, e);
983                 } catch (Throwable t) {
984                     // An exception occurred
985                     logger.error("Unable to write cached message {} for {}", e.getMessage(), r.uuid());
986                     logger.error("", t);
987                     for (Object o : cacheMessages) {
988                         bc.getBroadcasterCache().addToCache(getID(), r != null ? r.uuid() : BroadcasterCache.NULL, new BroadcastMessage(o));
989                     }
990                     return true;
991                 }
992 
993                 // If long-polling or JSONP is used we need to set the messages for the event again, because onResume() have cleared them
994                 if (willBeResumed) {
995                     e.setMessage(filteredMessageClone);
996                 }
997 
998                 for (AtmosphereResourceEventListener l : willBeResumed ? listeners : rImpl.atmosphereResourceEventListener()) {
999                     l.onBroadcast(e);
1000                 }
1001 
1002                 switch (r.transport()) {
1003                     case UNDEFINED:
1004                     case JSONP:
1005                     case AJAX:
1006                     case LONG_POLLING:
1007                         return true;
1008                     case SSE:
1009                         break;
1010                     default:
1011                         try {
1012                             r.getResponse().flushBuffer();
1013                         } catch (IOException ioe) {
1014                             logger.trace("", ioe);
1015                             AtmosphereResourceImpl.class.cast(r)._destroy();
1016                         }
1017                         break;
1018                 }
1019             }
1020         }
1021         return false;
1022     }

1328             if (maxSuspendResource.get() > 0 && resources.size() >= maxSuspendResource.get()) {
1329                 // Resume the first in.
1330                 if (policy == POLICY.FIFO) {
1331                     // TODO handle null return from poll()
1332                     AtmosphereResource resource = resources.poll();
1333                     try {
1334                         logger.warn("Too many resource. Forcing resume of {} ", resource.uuid());
1335                         resource.resume();
1336                     } catch (Throwable t) {
1337                         logger.warn("failed to resume resource {} ", resource, t);
1338                     }
1339                 } else if (policy == POLICY.REJECT) {
1340                     throw new RejectedExecutionException(String.format("Maximum suspended AtmosphereResources %s", maxSuspendResource));
1341                 }
1342             }
1343 
1344             if (!r.isSuspended()) {
1345                 logger.warn("AtmosphereResource {} is not suspended. If cached messages exists, this may cause unexpected situation. Suspend first", r.uuid());
1346             }
1347 
1348             if (!backwardCompatible && resources.contains(r)) {
1349                 boolean duplicate = r.transport() != AtmosphereResource.TRANSPORT.WEBSOCKET
1350                         || AtmosphereResourceImpl.class.cast(r).getRequest(false).getAttribute(INJECTED_ATMOSPHERE_RESOURCE) != null;
1351 
1352                 if (duplicate) {
1353                     AtmosphereResourceImpl dup = (AtmosphereResourceImpl) config.resourcesFactory().find(r.uuid());
1354                     if (dup != null && dup != r ) {
1355                         if ( ! dup.isPendingClose() ) {
1356                             logger.warn("Duplicate resource {}. Could be caused by a dead connection not detected by your server. Replacing the old one with the fresh one", r.uuid());
1357                         } else {
1358                             logger.debug("Not yet closed resource still active {}", r.uuid());
1359                         }
1360                         AtmosphereResourceImpl.class.cast(dup).dirtyClose();
1361                     } else {
1362                         logger.debug("Duplicate resource {}", r.uuid());
1363                         return this;
1364                     }
1365                 } else {
1366                     logger.debug("Duplicate resource {}", r.uuid());
1367                     return this;
1368                 }
1369             }
1370 
1371             // Only synchronize if we have a valid BroadcasterCache
1372             if (!bc.getBroadcasterCache().getClass().equals(BroadcasterCache.DEFAULT.getClass().getName())) {
1373                 // In case we are adding messages to the cache, we need to make sure the operation is done before.
1374                 synchronized (resources) {
1375                     cacheAndSuspend(r);
1376                 }
1377             } else {
1378                 cacheAndSuspend(r);
1379             }
1380         } finally {
1381             // OK reset
1382             if (!resources.isEmpty()) {
1383                 synchronized (awaitBarrier) {
1384                     awaitBarrier.notifyAll();
1385                 }
1386             }
1387         }
1388         return this;
1389     }

1391     /**
1392      * Look in the cache to see if there are messages available, and take the appropriate actions.
1393      *
1394      * @param r AtmosphereResource
1395      */
1396     protected void cacheAndSuspend(AtmosphereResource r) {
1397         // In case the connection is closed, for whatever reason
1398         if (!isAtmosphereResourceValid(r)) {
1399             logger.debug("Unable to add AtmosphereResource {}", r.uuid());
1400             return;
1401         }
1402 
1403         boolean wasResumed = checkCachedAndPush(r, r.getAtmosphereResourceEvent());
1404         if (!wasResumed && isAtmosphereResourceValid(r)) {
1405             logger.trace("Associating AtmosphereResource {} with Broadcaster {}", r.uuid(), getID());
1406 
1407             String parentUUID = r.transport().equals(AtmosphereResource.TRANSPORT.WEBSOCKET) ?
1408                     (String) AtmosphereResourceImpl.class.cast(r).getRequest(false).getAttribute(SUSPENDED_ATMOSPHERE_RESOURCE_UUID) :
1409                     null;
1410             if (!backwardCompatible && parentUUID != null) {
1411                 AtmosphereResource p = config.resourcesFactory().find(parentUUID);
1412                 if (p != null && !resources.contains(p)) {
1413                     notifyAndAdd(p);
1414                 } else if (p == null) {
1415                     notifyAndAdd(r);
1416                 } else {
1417                     logger.trace("AtmosphereResource {} was already mapped to {}", r.uuid(), parentUUID);
1418                 }
1419             } else {
1420                 notifyAndAdd(r);
1421             }
1422         } else if (!wasResumed) {
1423             logger.debug("Unable to add AtmosphereResource {} to {}", r.uuid(), name);
1424         }
1425     }

1489     protected Broadcaster removeAtmosphereResource(AtmosphereResource r, boolean executeDone) {
1490         if (destroyed.get()) {
1491             logger.debug(DESTROYED, getID(), "removeAtmosphereResource(AtmosphereResource r)");
1492             return this;
1493         }
1494 
1495         boolean removed = resources.remove(r);
1496         if (removed) {
1497             if (r.isSuspended()) {
1498                 logger.trace("Excluded from {} : {}", getID(), r.uuid());
1499                 bc.getBroadcasterCache().excludeFromCache(getID(), r);
1500             }
1501             notifyOnRemoveAtmosphereResourceListener(r);
1502         } else {
1503             logger.trace("Unable to remove {} from {}", r.uuid(), getID());
1504         }
1505         r.removeBroadcaster(this);
1506 
1507         if (!removed) return this;
1508 
1509         logger.trace("Removing AtmosphereResource {} for Broadcaster {}", r.uuid(), name);
1510         writeQueues.remove(r.uuid());
1511 
1512         // Here we need to make sure we aren't in the process of broadcasting and unlock the Future.
1513         if (executeDone) {
1514             AtmosphereResourceImpl aImpl = AtmosphereResourceImpl.class.cast(r);
1515             BroadcasterFuture f = (BroadcasterFuture) aImpl.getRequest(false).getAttribute(getID());
1516             if (f != null && !f.isDone() && !f.isCancelled()) {
1517                 aImpl.getRequest(false).removeAttribute(getID());
1518                 entryDone(f);
1519             }
1520         }
1521 
1522         return this;
1523     }
