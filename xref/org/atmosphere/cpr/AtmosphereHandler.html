53  package org.atmosphere.cpr;
54  
55  import java.io.IOException;
56  
57  /**
58   * Implementation of {@link AtmosphereHandler} allows creation of event-driven web applications which are hosted in
59   * the browser. An {@link AtmosphereHandler} allows a web application to suspend and resume an HTTP response. An HTTP
60   * response can be suspended {@link AtmosphereResource#suspend()} and later resume via {@link  AtmosphereResource#resume()}.
61   * Messages can also be shared between suspended responses and their associated {@link AtmosphereHandler} using a
62   * {@link Broadcaster}. Any invocation of {@link Broadcaster#broadcast(java.lang.Object)} will allow a suspended
63   * response to write the content of the message {@link AtmosphereHandler#onStateChange(org.atmosphere.cpr.AtmosphereResourceEvent)}.
64   * <p/>
65   * <strong>A class implementing {@link AtmosphereHandler} must be thread safe.</strong>
66   * <p/>
67   * For example, a simple pubsub based AtmosphereHandler will take the form of
68   * <p/>
69  <blockquote><pre>
70   public class AtmosphereHandlerPubSub extends AbstractReflectorAtmosphereHandler {
71  
72      public void onRequest(AtmosphereResource r) throws IOException {
73  
74          AtmosphereRequest req = r.getRequest();
75          AtmosphereResponse res = r.getResponse();
76          String method = req.getMethod();
77  
78          // Suspend the response.
79          if ("GET".equalsIgnoreCase(method)) {
80              // Log all events on the console, including WebSocket events.
81              r.addEventListener(new WebSocketEventListenerAdapter());
82  
83              res.setContentType("text/html;charset=ISO-8859-1");
84  
85              Broadcaster b = lookupBroadcaster(req.getPathInfo());
86              r.setBroadcaster(b).suspend();
87          } else if ("POST".equalsIgnoreCase(method)) {
88              Broadcaster b = lookupBroadcaster(req.getPathInfo());
89  
90              String message = req.getReader().readLine();
91  
92              if (message != null && message.indexOf("message") != -1) {
93                  b.broadcast(message.substring("message=".length()));
94              }
95          }
96      }
97  
98      public void destroy() {
99      }
100 
101     Broadcaster lookupBroadcaster(String pathInfo) {
102         String[] decodedPath = pathInfo.split("/");
103         Broadcaster b = BroadcasterFactory.getDefault().lookup(decodedPath[decodedPath.length - 1], true);
104         return b;
105     }
106 
107 }</pre></blockquote>
108  *
109  * It is recommended to use the {@link org.atmosphere.config.service.AtmosphereHandlerService}
110  * or {@link org.atmosphere.config.service.ManagedService} annotation to reduce the number
111  * of lines of code and take advantage of {@link AtmosphereInterceptor} such as
112  * {@link org.atmosphere.interceptor.AtmosphereResourceStateRecovery} and
113  * {@link org.atmosphere.interceptor.AtmosphereResourceLifecycleInterceptor}.
114  *
115  * @author Jeanfrancois Arcand
116  */
117 public interface AtmosphereHandler {
118 
119     /**
120      * When a client sends a request to its associated {@link AtmosphereHandler}, it can decide if the underlying
121      * connection can be suspended (creating a Continuation) or handle the connection synchronously.
122      * <p/>
123      * It is recommended to only suspend requests for which HTTP method is a GET and use the POST method to send data
124      * to the server, without marking the connection as asynchronous.
125      *
126      * @param resource an {@link AtmosphereResource}
127      * @throws java.io.IOException
128      */
129     void onRequest(AtmosphereResource resource) throws IOException;
130 
131     /**
132      * This method is invoked when the {@link Broadcaster} executes a broadcast operation. When this method is invoked
133      * its associated {@link Broadcaster}, any suspended connection will be allowed to write the data back to its
134      * associated clients.
135      * <p/>
136      * This method will also be invoked when a response get resumed, e.g. when {@link AtmosphereResource#resume} gets
137      * invoked. In that case, {@link AtmosphereResourceEvent#isResuming} will return true.
138      * <p/>
139      * This method will also be invoked when the {@link AtmosphereResource#suspend(long)} expires. In that case,
140      * {@link AtmosphereResourceEvent#isResumedOnTimeout} will return <tt>true</tt>.
141      *
142      * @param event an {@link AtmosphereResourceEvent}
143      * @throws java.io.IOException
144      */
145     void onStateChange(AtmosphereResourceEvent event) throws IOException;
146 
147     /**
148      * Destroy this handler
149      */
150     void destroy();
151 }
