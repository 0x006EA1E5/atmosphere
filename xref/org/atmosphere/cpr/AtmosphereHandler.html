16  package org.atmosphere.cpr;
17  
18  import java.io.IOException;
19  
20  /**
21   * Implementation of {@link AtmosphereHandler} allows creation of event-driven web applications which are hosted in
22   * the browser. An {@link AtmosphereHandler} allows a web application to suspend and resume an HTTP response. An HTTP
23   * response can be suspended {@link AtmosphereResource#suspend()} and later resume via {@link  AtmosphereResource#resume()}.
24   * Messages can also be shared between suspended responses and their associated {@link AtmosphereHandler} using a
25   * {@link Broadcaster}. Any invocation of {@link Broadcaster#broadcast(java.lang.Object)} will allow a suspended
26   * response to write the content of the message {@link AtmosphereHandler#onStateChange(org.atmosphere.cpr.AtmosphereResourceEvent)}.
27   * <p/>
28   * <strong>A class implementing {@link AtmosphereHandler} must be thread safe.</strong>
29   * <p/>
30   * For example, a simple pubsub based AtmosphereHandler will take the form of
31   * <p/>
32  <blockquote><pre>
33   public class AtmosphereHandlerPubSub extends AbstractReflectorAtmosphereHandler {
34  
35      public void onRequest(AtmosphereResource r) throws IOException {
36  
37          AtmosphereRequest req = r.getRequest();
38          AtmosphereResponse res = r.getResponse();
39          String method = req.getMethod();
40  
41          // Suspend the response.
42          if ("GET".equalsIgnoreCase(method)) {
43              // Log all events on the console, including WebSocket events.
44              r.addEventListener(new WebSocketEventListenerAdapter());
45  
46              res.setContentType("text/html;charset=ISO-8859-1");
47  
48              Broadcaster b = lookupBroadcaster(req.getPathInfo());
49              r.setBroadcaster(b).suspend();
50          } else if ("POST".equalsIgnoreCase(method)) {
51              Broadcaster b = lookupBroadcaster(req.getPathInfo());
52  
53              String message = req.getReader().readLine();
54  
55              if (message != null && message.indexOf("message") != -1) {
56                  b.broadcast(message.substring("message=".length()));
57              }
58          }
59      }
60  
61      public void destroy() {
62      }
63  
64      Broadcaster lookupBroadcaster(String pathInfo) {
65          String[] decodedPath = pathInfo.split("/");
66          Broadcaster b = BroadcasterFactory.getDefault().lookup(decodedPath[decodedPath.length - 1], true);
67          return b;
68      }
69  
70  }</pre></blockquote>
71   *
72   * It is recommended to use the {@link org.atmosphere.config.service.AtmosphereHandlerService}
73   * or {@link org.atmosphere.config.service.ManagedService} annotation to reduce the number
74   * of lines of code and take advantage of {@link AtmosphereInterceptor} such as
75   * {@link org.atmosphere.interceptor.AtmosphereResourceStateRecovery} and
76   * {@link org.atmosphere.interceptor.AtmosphereResourceLifecycleInterceptor}.
77   *
78   * @author Jeanfrancois Arcand
79   */
80  public interface AtmosphereHandler {
81  
82      /**
83       * When a client sends a request to its associated {@link AtmosphereHandler}, it can decide if the underlying
84       * connection can be suspended (creating a Continuation) or handle the connection synchronously.
85       * <p/>
86       * It is recommended to only suspend requests for which HTTP method is a GET and use the POST method to send data
87       * to the server, without marking the connection as asynchronous.
88       *
89       * @param resource an {@link AtmosphereResource}
90       * @throws java.io.IOException
91       */
92      void onRequest(AtmosphereResource resource) throws IOException;
93  
94      /**
95       * This method is invoked when the {@link Broadcaster} executes a broadcast operation. When this method is invoked
96       * its associated {@link Broadcaster}, any suspended connection will be allowed to write the data back to its
97       * associated clients.
98       * <p/>
99       * This method will also be invoked when a response get resumed, e.g. when {@link AtmosphereResource#resume} gets
100      * invoked. In that case, {@link AtmosphereResourceEvent#isResuming} will return true.
101      * <p/>
102      * This method will also be invoked when the {@link AtmosphereResource#suspend(long)} expires. In that case,
103      * {@link AtmosphereResourceEvent#isResumedOnTimeout} will return <tt>true</tt>.
104      *
105      * @param event an {@link AtmosphereResourceEvent}
106      * @throws java.io.IOException
107      */
108     void onStateChange(AtmosphereResourceEvent event) throws IOException;
109 
110     /**
111      * Destroy this handler
112      */
113     void destroy();
114 }
