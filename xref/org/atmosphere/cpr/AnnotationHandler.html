16  package org.atmosphere.cpr;
17  
18  import org.atmosphere.annotation.Processor;
19  import org.atmosphere.config.AtmosphereAnnotation;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  import java.lang.annotation.Annotation;
24  import java.util.Collection;
25  import java.util.HashMap;
26  import java.util.Map;
27  
28  /**
29   * A class that handles the results of an annotation scan. This class contains the logic that maps
30   * an annotation type to the corresponding framework setup.
31   *
32   * @author Stuart Douglas
33   * @author Jeanfrancois Arcand
34   */
35  public class AnnotationHandler {
36  
37      private static final Logger logger = LoggerFactory.getLogger(AnnotationHandler.class);
38  
39      private final Map<Class<? extends Annotation>, Class<? extends Processor>> annotations = new HashMap<Class<? extends Annotation>, Class<? extends Processor>>();
40      private final Map<Class<? extends Processor>, Processor> processors = new HashMap<Class<? extends Processor>, Processor>();
41  
42  
43      public AnnotationHandler() {
44      }
45  
46      public Class<? extends Processor> handleProcessor(Class<?> clazz) {
47          if (Processor.class.isAssignableFrom(clazz)) {
48              Class<Processor> p = (Class<Processor>) clazz;
49              logger.trace("Processor {} associated with {}", p, p.getAnnotation(AtmosphereAnnotation.class).value() );
50              annotations.put(p.getAnnotation(AtmosphereAnnotation.class).value(), p);
51              return p;
52          }
53          return null;
54      }
55  
56      public Class<? extends Annotation>[] handledClass(){
57          Collection<Class<? extends Annotation>> c = annotations.keySet();
58          return c.toArray(new Class[0]);
59      }
60  
61      public AnnotationHandler handleAnnotation(final AtmosphereFramework framework, final Class<? extends Annotation> annotation, final Class<?> discoveredClass) {
62          framework.annotationScanned(true);
63          Class<? extends Processor> a = annotations.get(annotation);
64  
65          if (a != null) {
66              Processor p = processors.get(a);
67              if (p == null) {
68                  try {
69                      p = a.newInstance();
70                  } catch (Exception e) {
71                      logger.warn("Unable to create Processor {}", p);
72                  }
73                  processors.put(a, p);
74              }
75              p.handle(framework, discoveredClass);
76              logger.trace("Annotation {} handled by {}", annotation, p.getClass().getName());
77          } else {
78              logger.trace("Annotation {} unhandled", annotation);
79          }
80          return this;
81      }
82  
83      public void destroy() {
84          annotations.clear();
85          processors.clear();
86      }
87  
88  }
