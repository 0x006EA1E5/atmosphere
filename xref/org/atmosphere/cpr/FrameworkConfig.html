18  import org.atmosphere.cpr.AtmosphereFramework.AtmosphereHandlerWrapper;
19  import org.atmosphere.cpr.AtmosphereRequest.NoOpsRequest;
20  import org.atmosphere.interceptor.JavaScriptProtocol;
21  import org.atmosphere.websocket.WebSocketProcessor;

23  /**
24   * Request attribute a framework integrator can use to lookup Atmosphere internal objects.
25   *
26   * @author Jeanfrancois Arcand
27   */
28  public interface FrameworkConfig {
29      /**
30       * The default Hazelcast Broadcaster class.
31       */
32      String HAZELCAST_BROADCASTER = "org.atmosphere.plugin.hazelcast.HazelcastBroadcaster";
33      /**
34       * The default Jersey Broadcaster class.
35       */
36      String JERSEY_BROADCASTER = "org.atmosphere.jersey.JerseyBroadcaster";
37      /**
38       * The default Redis Broadcaster class.
39       */
40      String REDIS_BROADCASTER = "org.atmosphere.plugin.redis.RedisBroadcaster";
41      /**
42       * The default JMS Broadcaster class.
43       */
44      String JMS_BROADCASTER = "org.atmosphere.plugin.jms.JMSBroadcaster";
45      /**
46       * The default JGroups Broadcaster class.
47       */
48      String JGROUPS_BROADCASTER = "org.atmosphere.plugin.jgroups.JGroupsBroadcaster";
49      /**
50       * The default RMI Broadcaster class.
51       */
52      String RMI_BROADCASTER = "org.atmosphere.plugin.rmi.RMIBroadcaster";
53      /**
54       * The default RabbitMQ Broadcaster class.
55       */
56      String RABBITMQ_BROADCASTER = "org.atmosphere.plugin.rabbitmq.RabbitMQBroadcaster";
57      /**
58       * The default XMPP Broadcaster class.
59       */
60      String XMPP_BROADCASTER = "org.atmosphere.plugin.xmpp.XMPPBroadcaster";
61      /**
62       * The default Jersey container class.
63       */
64      String JERSEY_CONTAINER = "com.sun.jersey.spi.container.servlet.ServletContainer";
65      /**
66       * A request attribute used to lookup the {@link AtmosphereNativeCometServlet}. This attribute is for framework integrators and not recommend for normal applications.
67       */
68      String ATMOSPHERE_SERVLET = "org.atmosphere.cpr.AtmosphereServlet";
69      /**
70       * A request attribute used to lookup the {@link AtmosphereResource}. This attribute is for framework integrators and not recommend for normal applications.
71       */
72      String ATMOSPHERE_RESOURCE = AtmosphereResource.class.getName();
73      /**
74       * A request attribute used to lookup the {@link AtmosphereResource} created by an external component and injected inside the {@link AsynchronousProcessor}.
75       */
76      String INJECTED_ATMOSPHERE_RESOURCE = AtmosphereResource.class.getName() + ".injected";
77      /**
78       * Tell a {@link AsyncSupport} it can support session or not.
79       */
80      String SUPPORT_SESSION = AsynchronousProcessor.class.getName() + ".supportSession";
81      /**
82       * A request attribute used to lookup the {@link AtmosphereHandler}. This attribute is for framework integrator and not recommend for normal application.
83       */
84      String ATMOSPHERE_HANDLER_WRAPPER = AtmosphereHandlerWrapper.class.getName();
85      /**
86       * True if the {@link AtmosphereHandlerWrapper} has been modified by an {@link AtmosphereInterceptor}.
87       */
88      String NEW_MAPPING = AtmosphereHandlerWrapper.class.getName() + ".newMapping";
89      /**
90       * A reference to the Jersey's {@link ContainerResponse}.
91       */
92      String CONTAINER_RESPONSE = "org.atmosphere.jersey.containerResponse";
93      /**
94       * Decide to write extra header.
95       */
96      String WRITE_HEADERS = AtmosphereResource.class.getName() + ".writeHeader";
97      /**
98       * Used by a container to tell Atmosphere Runtime what is the expected content type.
99       */
100     String EXPECTED_CONTENT_TYPE = FrameworkConfig.class.getName() + ".expectedContentType";
101     /**
102      * The name of the sub-protocol used.
103      */
104     String WEBSOCKET_SUBPROTOCOL = "websocket-subprotocol";
105     /**
106      * The SimpleHttpProtocol.
107      */
108     String SIMPLE_HTTP_OVER_WEBSOCKET = "polling-websocket-message";
109     /**
110      * Cancel suspending a connection.
111      */
112     String CANCEL_SUSPEND_OPERATION = "doNotSuspend";
113     /**
114      * AtmosphereConfig instance.
115      */
116     String ATMOSPHERE_CONFIG = AtmosphereConfig.class.getName();
117     /**
118      * Instance of Jersey's ContainerResponseWriter that can be configured by a Framework running on top of Atmosphere.
119      */
120     String JERSEY_CONTAINER_RESPONSE_WRITER_INSTANCE = "org.atmosphere.jersey.containerResponseWriterInstance";
121     /**
122      * Current transport used.
123      */
124     String TRANSPORT_IN_USE = FrameworkConfig.class.getName() + ".transportUsed";
125     /**
126      * The callback for handshaking the {@link org.atmosphere.interceptor.JavaScriptProtocol}.
127      */
128     String CALLBACK_JAVASCRIPT_PROTOCOL = JavaScriptProtocol.class.getName() + ".callback";
129     /**
130      * The Jersey package used for scanning annotation.
131      */
132     String JERSEY_SCANNING_PACKAGE = "com.sun.jersey.config.property.packages";
133     /**
134      * The Jersey package used for scanning annotation.
135      */
136     String JERSEY2_SCANNING_PACKAGE = "jersey.config.server.provider.packages";
137     /**
138      * Throw Exception from cloned request.
139      */
140     String THROW_EXCEPTION_ON_CLONED_REQUEST = NoOpsRequest.class.getName() + ".throwExceptionOnClonedRequest";
141     /**
142      * The subject for the current request.
143      */
144     String SECURITY_SUBJECT = AtmosphereRequest.class.getName() + ".subject";
145     /**
146      * The {@link javax.servlet.AsyncContext}.
147      */
148     String ASYNC_CONTEXT = "org.atmosphere.container.asyncContext";
149     /**
150      * A flag indicating a message has been written on the resource. This is useful to know if a resource must be resumed for transport like
151      * long-polling.
152      */
153     String MESSAGE_WRITTEN = Broadcaster.class.getName() + ".messageWritten";
154     /**
155      * Guice Injector
156      */
157     String GUICE_INJECTOR = "org.atmosphere.guice.GuiceObjectFactory";
158     /**
159      * Spring Injector
160      */
161     String SPRING_INJECTOR = "org.atmosphere.spring.SpringWebObjectFactory";
162     /**
163      * CDI Injector
164      */
165     String CDI_INJECTOR = "org.atmosphere.cdi.CDIObjectFactory";
166     /**
167      * The path that mapped the {@link AtmosphereHandler}
168      */
169     String MAPPED_PATH = AtmosphereHandler.class.getName() + ".mappedBy";
170     /**
171      * Tag for WebSocket's Message
172      */
173     String WEBSOCKET_MESSAGE = WebSocketProcessor.class.getName() + ".websocket.message";
174 
175 
176 }
