99      @Override
100     public void init(ServletConfig sc) throws ServletException {
101 
102         String maxInactive = sc.getInitParameter(MAX_INACTIVE) != null ? sc.getInitParameter(MAX_INACTIVE) :
103                 config.getInitParameter(MAX_INACTIVE);
104 
105         if (maxInactive != null) {
106             trackActiveRequest = true;
107             final long maxInactiveTime = Long.parseLong(maxInactive);
108             if (maxInactiveTime <= 0) return;
109 
110             closedDetector.scheduleAtFixedRate(new Runnable() {
111                 public void run() {
112                     for (AtmosphereRequest req : aliveRequests.keySet()) {
113                         long l = (Long) req.getAttribute(MAX_INACTIVE);
114                         if (l > 0 && System.currentTimeMillis() - l > maxInactiveTime) {
115                             try {
116                                 if (logger.isTraceEnabled()) {
117                                     logger.trace("Close detector disconnecting {}. Current size {}", req.resource(), aliveRequests.size());
118                                 }
119                                 AtmosphereResourceImpl r = (AtmosphereResourceImpl) aliveRequests.remove(req);
120                                 cancelled(req, r.getResponse(false));
121                             } catch (Throwable e) {
122                                 logger.warn("closedDetector", e);
123                             } finally {
124                                 try {
125                                     req.setAttribute(MAX_INACTIVE, (long) -1);
126                                 } catch (Throwable t) {
127                                     logger.trace("closedDetector", t);
128                                 }
129                             }
130                         }
131                     }
132                 }
133             }, 0, 1, TimeUnit.SECONDS);
134         }
135     }

168     /**
169      * Invoke the {@link AtmosphereHandler#onRequest} method.
170      *
171      * @param req the {@link AtmosphereRequest}
172      * @param res the {@link AtmosphereResponse}
173      * @return action the Action operation.
174      * @throws java.io.IOException
175      * @throws javax.servlet.ServletException
176      */
177     Action action(AtmosphereRequest req, AtmosphereResponse res) throws IOException, ServletException {
178 
179         if (Utils.webSocketEnabled(req) && !supportWebSocket()) {
180             res.setStatus(501);
181             res.addHeader(X_ATMOSPHERE_ERROR, "Websocket protocol not supported");
182             res.flushBuffer();
183             return new Action();
184         }
185 
186         if (config.handlers().isEmpty()) {
187             logger.error("No AtmosphereHandler found. Make sure you define it inside WEB-INF/atmosphere.xml or annotate using @___Service");
188             throw new AtmosphereMappingException("No AtmosphereHandler found. Make sure you define it inside WEB-INF/atmosphere.xml or annotate using @___Service");
189         }
190 
191         if (res.request() == null) {
192             res.request(req);
193         }
194 
195         if (supportSession()) {
196             // Create the session needed to support the Resume
197             // operation from disparate requests.
198             req.getSession(true);
199         }
200 
201         req.setAttribute(FrameworkConfig.SUPPORT_SESSION, supportSession());
202         
203         AtmosphereHandlerWrapper handlerWrapper = map(req);
204         if (config.getBroadcasterFactory() == null) {
205             logger.error("Atmosphere is misconfigured and will not work. BroadcasterFactory is null");
206             return Action.CANCELLED;
207         }
208         AtmosphereResourceImpl resource = configureWorkflow(null, handlerWrapper, req, res);
209 
210         String v = req.getHeader(HeaderConfig.X_ATMO_BINARY);
211         if (v != null) {
212             resource.forceBinaryWrite(Boolean.valueOf(v));
213         }
214 
215         // Globally defined
216         Action a = invokeInterceptors(config.framework().interceptors(), resource);
217         if (a.type() != Action.TYPE.CONTINUE) {
218             return a;
219         }
220 
221         // Per AtmosphereHandler
222         a = invokeInterceptors(handlerWrapper.interceptors, resource);
223         if (a.type() != Action.TYPE.CONTINUE) {
224             return a;
225         }
226 
227         // Remap occured.
228         if (req.getAttribute(FrameworkConfig.NEW_MAPPING) != null) {
229             req.removeAttribute(FrameworkConfig.NEW_MAPPING);
230             handlerWrapper = map(req);
231             if (handlerWrapper == null) {
232                 logger.debug("Remap {}", resource.uuid());
233                 throw new AtmosphereMappingException("Invalid state. No AtmosphereHandler maps request for " + req.getRequestURI());
234             }
235             resource = configureWorkflow(resource, handlerWrapper, req, res);
236             resource.setBroadcaster(handlerWrapper.broadcaster);
237         }
238 
239         //Unit test mock the request and will throw NPE.
240         boolean skipAtmosphereHandler = req.getAttribute(SKIP_ATMOSPHEREHANDLER.name()) != null
241                 ? (Boolean) req.getAttribute(SKIP_ATMOSPHEREHANDLER.name()) : Boolean.FALSE;
242         if (!skipAtmosphereHandler) {
243             try {
244                 handlerWrapper.atmosphereHandler.onRequest(resource);
245             } catch (IOException t) {
246                 resource.onThrowable(t);
247                 throw t;
248             }
249         }
250 
251         postInterceptors(handlerWrapper.interceptors, resource);
252         postInterceptors(config.framework().interceptors(), resource);
253 
254         if (trackActiveRequest && resource.isSuspended() && req.getAttribute(FrameworkConfig.CANCEL_SUSPEND_OPERATION) == null) {
255             req.setAttribute(MAX_INACTIVE, System.currentTimeMillis());
256             aliveRequests.put(req, resource);
257         }
258 
259         Action action = skipAtmosphereHandler ? Action.CANCELLED : resource.action();
260         if (supportSession() && action.type().equals(Action.TYPE.SUSPEND)) {
261             // Do not allow times out.
262             SessionTimeoutSupport.setupTimeout(req.getSession());
263         }
264         logger.trace("Action for {} was {} with transport " + req.getHeader(X_ATMOSPHERE_TRANSPORT), req.resource() != null ? req.resource().uuid() : "null", action);
265         return action;
266     }

268     private AtmosphereResourceImpl configureWorkflow(AtmosphereResourceImpl resource,
269                                                      AtmosphereHandlerWrapper handlerWrapper,
270                                                      AtmosphereRequest req, AtmosphereResponse res) {
271 
272         config.getBroadcasterFactory().add(handlerWrapper.broadcaster, handlerWrapper.broadcaster.getID());
273 
274         // Check Broadcaster state. If destroyed, replace it.
275         Broadcaster b = handlerWrapper.broadcaster;
276         if (b.isDestroyed()) {
277             BroadcasterFactory f = config.getBroadcasterFactory();
278             synchronized (f) {
279                 f.remove(b, b.getID());
280                 try {
281                     handlerWrapper.broadcaster = f.get(b.getID());
282                 } catch (IllegalStateException ex) {
283                     // Something wrong occurred, let's not fail and loookup the value
284                     logger.trace("", ex);
285                     // fallback to lookup
286                     handlerWrapper.broadcaster = f.lookup(b.getID(), true);
287                 }
288             }
289         }
290 
291         if (resource == null) {
292             resource = (AtmosphereResourceImpl) req.getAttribute(FrameworkConfig.INJECTED_ATMOSPHERE_RESOURCE);
293         }
294 
295         if (resource == null) {
296             // TODO: cast is dangerous
297             resource = (AtmosphereResourceImpl)
298                     AtmosphereResourceFactory.getDefault().create(config, handlerWrapper.broadcaster, res, this, handlerWrapper.atmosphereHandler);
299         } else {
300             // TODO: REDESIGN, UGLY.
301             try {
302                 // Make sure it wasn't set before
303                 resource.getBroadcaster();
304             } catch (IllegalStateException ex) {
305                 resource.setBroadcaster(handlerWrapper.broadcaster);
306             }
307             resource.atmosphereHandler(handlerWrapper.atmosphereHandler);
308         }
309 
310         req.setAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE, resource);
311         req.setAttribute(FrameworkConfig.ATMOSPHERE_HANDLER_WRAPPER, handlerWrapper);
312         req.setAttribute(SKIP_ATMOSPHEREHANDLER.name(), Boolean.FALSE);
313         return resource;
314     }

316     private String path(AtmosphereRequest request) {
317         String path;
318         String pathInfo = null;
319         try {
320             pathInfo = request.getPathInfo();
321         } catch (IllegalStateException ex) {
322             // http://java.net/jira/browse/GRIZZLY-1301
323         }
324 
325         if (pathInfo != null) {
326             path = request.getServletPath() + pathInfo;
327         } else {
328             path = request.getServletPath();
329         }
330 
331         if (path == null || path.isEmpty()) {
332             path = "/";
333         }
334         return path;
335     }

337     private Action invokeInterceptors(List<AtmosphereInterceptor> c, AtmosphereResource r) {
338         Action a = Action.CONTINUE;
339         for (AtmosphereInterceptor arc : c) {
340             a = arc.inspect(r);
341             if (a == null) {
342                 logger.trace("Action was null for {}", arc);
343                 a = Action.CANCELLED;
344             }
345 
346             boolean skip = a.type() == SKIP_ATMOSPHEREHANDLER;
347             if (skip) {
348                 logger.trace("AtmosphereInterceptor {} asked to skip the AtmosphereHandler for {}", arc, r.uuid());
349                 r.getRequest().setAttribute(SKIP_ATMOSPHEREHANDLER.name(), Boolean.TRUE);
350             }
351 
352             if (a.type() != Action.TYPE.CONTINUE) {
353                 logger.trace("Interceptor {} interrupted the dispatch for {} with " + a, arc, r.uuid());
354                 return a;
355             }
356         }
357         return a;
358     }

360     private void postInterceptors(List<AtmosphereInterceptor> c, AtmosphereResource r) {
361         for (int i = c.size() - 1; i > -1; i--) {
362             c.get(i).postInspect(r);
363         }
364     }

381         AtmosphereHandlerWrapper atmosphereHandlerWrapper = mapper.map(req, config.handlers());
382         if (atmosphereHandlerWrapper == null) {

391     /**
392      * All proprietary Comet based {@link Servlet} must invoke the resume method when the Atmosphere's application
393      * decide to resume the {@link AtmosphereResponse}. The returned value, of type {@link Action}, tells the
394      * proprietary Comet {@link Servlet} to resume (again), suspended or do nothing with the current {@link AtmosphereResponse}.
395      *
396      * @param request the {@link AtmosphereRequest}
397      * @param response the {@link AtmosphereResponse}
398      * @return action the Action operation.
399      * @throws java.io.IOException
400      * @throws javax.servlet.ServletException
401      */
402     public Action resumed(AtmosphereRequest request, AtmosphereResponse response)
403             throws IOException, ServletException {
404 
405         AtmosphereResourceImpl r =
406                 (AtmosphereResourceImpl) request.getAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
407 
408         if (r == null) return Action.CANCELLED; // We are cancelled already
409 
410         AtmosphereHandler atmosphereHandler = r.getAtmosphereHandler();
411 
412         AtmosphereResourceEvent event = r.getAtmosphereResourceEvent();
413         if (event != null && event.isResuming() && !event.isCancelled()) {
414             synchronized (r) {
415                 atmosphereHandler.onStateChange(event);
416             }
417         }
418         return Action.RESUME;
419     }

422      * All proprietary Comet based {@link Servlet} must invoke the timedout method when the underlying WebServer time
423      * out the {@link AtmosphereResponse}. The returned value, of type {@link Action}, tells the proprietary
424      * Comet {@link Servlet} to resume (again), suspended or do nothing with the current {@link AtmosphereResponse}.
425      *
426      * @param req the {@link AtmosphereRequest}
427      * @param res the {@link AtmosphereResponse}
428      * @return action the Action operation.
429      * @throws java.io.IOException
430      * @throws javax.servlet.ServletException
431      */
432     public Action timedout(AtmosphereRequest req, AtmosphereResponse res)
433             throws IOException, ServletException {
434 
435         logger.trace("Timing out {}", req);
436         if (trackActiveRequest(req) && completeLifecycle(req.resource(), false)) {
437             config.framework().notify(Action.TYPE.TIMEOUT, req, res);
438         }
439         return timedoutAction;
440     }

442     protected boolean trackActiveRequest(AtmosphereRequest req) {
443         if (trackActiveRequest) {
444             try {
445                 long l = (Long) req.getAttribute(MAX_INACTIVE);
446                 if (l == -1) {
447                     // The closedDetector closed the connection.
448                     return false;
449                 }
450                 req.setAttribute(MAX_INACTIVE, (long) -1);
451                 // GlassFish
452             } catch (Throwable ex) {
453                 logger.trace("Request already recycled", req);
454                 // Request is no longer active, return
455                 return false;
456 
457             }
458         }
459         return true;
460     }

462     /**
463      * Cancel or times out an {@link AtmosphereResource} by invoking it's associated {@link AtmosphereHandler#onStateChange(AtmosphereResourceEvent)}
464      *
465      * @param r an {@link AtmosphereResource}
466      * @param cancelled true if cancelled, false if timedout
467      * @return true if the operation was executed.
468      */
469     public boolean completeLifecycle(final AtmosphereResource r, boolean cancelled) {
470         if (r != null && !r.isCancelled()) {
471             logger.trace("Finishing lifecycle for AtmosphereResource {}", r.uuid());
472             final AtmosphereResourceImpl impl = AtmosphereResourceImpl.class.cast(r);
473             synchronized (impl) {
474                 try {
475                     if (impl.isCancelled()) {
476                         logger.trace("{} is already cancelled", impl.uuid());
477                         return false;
478                     }
479 
480                     AtmosphereResourceEventImpl e = impl.getAtmosphereResourceEvent();
481                     if (!e.isClosedByClient()) {
482                         if (cancelled) {
483                             e.setCancelled(true);
484                         } else {
485                             e.setIsResumedOnTimeout(true);
486                             Broadcaster b = r.getBroadcaster();
487                             if (b instanceof DefaultBroadcaster) {
488                                 ((DefaultBroadcaster) b).broadcastOnResume(r);
489                             }
490 
491                             // TODO: Was it there for legacy reason?
492                             // impl.getAtmosphereResourceEvent().setIsResumedOnTimeout(impl.resumeOnBroadcast());
493                         }
494                     }
495                     invokeAtmosphereHandler(impl);
496                 } catch (Throwable ex) {
497                     // Something wrong happened, ignore the exception
498                     logger.trace("Failed to cancel resource: {}", impl.uuid(), ex);
499                 } finally {
500                     try {
501                         impl.notifyListeners();
502                         impl.setIsInScope(false);
503                         try {
504                             impl.getResponse().getOutputStream().close();
505                         } catch (Throwable t) {
506                             try {
507                                 impl.getResponse().getWriter().close();
508                             } catch (Throwable t2) {
509                             }
510                         }
511                         impl.cancel();
512                     } catch (Throwable t) {
513                         logger.trace("completeLifecycle", t);
514                     } finally {
515                         impl._destroy();
516                     }
517                 }
518             }
519             return true;
520         } else {
521             logger.trace("AtmosphereResource {} was already cancelled or gc", r != null ? r.uuid() : "null");
522             return false;
523         }
524     }

526     /**
527      * Invoke the associated {@link AtmosphereHandler}. This method must be synchronized on an AtmosphereResource.
528      *
529      * @param r a {@link AtmosphereResourceImpl}
530      * @throws IOException
531      */
532     protected void invokeAtmosphereHandler(AtmosphereResourceImpl r) throws IOException {
533         if (!r.isInScope()) {
534             logger.trace("AtmosphereResource out of scope {}", r.uuid());
535             return;
536         }
537 
538         AtmosphereRequest req = r.getRequest(false);
539         String disableOnEvent = r.getAtmosphereConfig().getInitParameter(ApplicationConfig.DISABLE_ONSTATE_EVENT);
540         r.getAtmosphereResourceEvent().setMessage(r.writeOnTimeout());
541         try {
542             if (disableOnEvent == null || !disableOnEvent.equals(String.valueOf(true))) {
543                 AtmosphereHandler atmosphereHandler = r.getAtmosphereHandler();
544 
545                 if (atmosphereHandler != null) {
546                     try {
547                         atmosphereHandler.onStateChange(r.getAtmosphereResourceEvent());
548                     } finally {
549                         Meteor m = (Meteor) req.getAttribute(AtmosphereResourceImpl.METEOR);
550                         if (m != null) {
551                             m.destroy();
552                         }
553                     }
554                 }
555             }
556         } catch (IOException ex) {
557             try {
558                 r.onThrowable(ex);
559             } catch (Throwable t) {
560                 logger.warn("failed calling onThrowable()", ex);
561             }
562         }
563     }

567      * detect that the client closed the connection.
568      *
569      * @param req the {@link AtmosphereRequest}

572      * @throws java.io.IOException

575     public Action cancelled(AtmosphereRequest req, AtmosphereResponse res)
576             throws IOException, ServletException {
577 
578         logger.trace("Cancelling {}", req);
579         if (trackActiveRequest(req) && completeLifecycle(req.resource(), true)) {
580             config.framework().notify(Action.TYPE.CANCELLED, req, res);
581         }
582         return cancelledAction;

585     protected void shutdown() {
586         closedDetector.shutdownNow();
587         for (AtmosphereResource resource : aliveRequests.values()) {
588             try {
589                 resource.resume();
590             } catch (Throwable t) {
591                 // Something wrong happenned, ignore the exception
592                 logger.debug("failed on resume: " + resource, t);
593             }
594         }
595     }

602     /**
603      * A callback class that can be used by Framework integrator to handle the close/timedout/resume life cycle
604      * of an {@link AtmosphereResource}. This class only supports {@link AsyncSupport} implementation that
605      * extends {@link AsynchronousProcessor}.
606      */
607     public final static class AsynchronousProcessorHook {
608 
609         private final AtmosphereResourceImpl r;
610 
611         public AsynchronousProcessorHook(AtmosphereResourceImpl r) {
612             this.r = r;
613             if (!AsynchronousProcessor.class.isAssignableFrom(r.asyncSupport.getClass())) {
614                 throw new IllegalStateException("AsyncSupport must extends AsynchronousProcessor");
615             }
616         }
617 
618         public void closed() {
619             try {
620                 ((AsynchronousProcessor) r.asyncSupport).cancelled(r.getRequest(false), r.getResponse(false));
621             } catch (IOException e) {
622                 logger.debug("", e);
623             } catch (ServletException e) {
624                 logger.debug("", e);
625             }
626         }
627 
628         public void timedOut() {
629             try {
630                 ((AsynchronousProcessor) r.asyncSupport).timedout(r.getRequest(false), r.getResponse(false));
631             } catch (IOException e) {
632                 logger.debug("", e);
633             } catch (ServletException e) {
634                 logger.debug("", e);
635             }
636         }
637 
638         public void resume() {
639             ((AsynchronousProcessor) r.asyncSupport).action(r);
640         }
