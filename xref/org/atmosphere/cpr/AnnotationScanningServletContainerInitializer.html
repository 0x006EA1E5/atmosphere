62      @Override
63      public void onStartup(final Set<Class<?>> classes, final ServletContext servletContext) throws ServletException {
64          final Map<Class<? extends Annotation>, Set<Class<?>>> classesByAnnotation = new HashMap<Class<? extends Annotation>, Set<Class<?>>>();
65          if (classes != null) {
66              for(final Class<?> clazz : classes) {
67                  for(Annotation annotation : clazz.getAnnotations()) {
68                      Set<Class<?>> classSet = classesByAnnotation.get(annotation.annotationType());
69                      if(classSet == null) {
70                          classesByAnnotation.put(annotation.annotationType(), classSet = new HashSet<Class<?>>());
71                      }
72                      classSet.add(clazz);
73                  }
74              }
75          }
76          servletContext.setAttribute(DefaultAnnotationProcessor.ANNOTATION_ATTRIBUTE, classesByAnnotation);
77      }
