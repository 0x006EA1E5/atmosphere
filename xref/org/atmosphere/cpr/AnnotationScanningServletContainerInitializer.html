53  /**
54   * A ServletContainerInitializer that scans for annotations, and places them in a map keyed by annotation type in the
55   * servlet context.
56   *
57   * NOTE: Any new Atmosphere's annotation must be hardcoded here.
58   *
59   * @author Stuart Douglas
60   */
61  @HandlesTypes({
62          AtmosphereHandlerService.class,
63          BroadcasterCacheService.class,
64          BroadcasterFilterService.class,
65          BroadcasterFactoryService.class,
66          BroadcasterService.class,
67          MeteorService.class,
68          WebSocketFactoryService.class,
69          WebSocketHandlerService.class,
70          WebSocketProtocolService.class,
71          AtmosphereInterceptorService.class,
72          BroadcasterListenerService.class,
73          AsyncSupportService.class,
74          AsyncSupportListenerService.class,
75          WebSocketProcessorService.class,
76          BroadcasterCacheInspectorService.class,
77          ManagedService.class,
78          AtmosphereService.class,
79          EndpointMapperService.class,
80          BroadcasterCacheListenerService.class,
81          AtmosphereAnnotation.class,
82          AtmosphereResourceFactoryService.class,
83          AtmosphereFrameworkListenerService.class,
84          AtmosphereResourceListenerService.class,
85          UUIDProviderService.class
86  })
87  public class AnnotationScanningServletContainerInitializer implements ServletContainerInitializer {
88  
89      @Override
90      public void onStartup(final Set<Class<?>> classes, final ServletContext servletContext) throws ServletException {
91          final Map<Class<? extends Annotation>, Set<Class<?>>> classesByAnnotation = new HashMap<Class<? extends Annotation>, Set<Class<?>>>();
92          if (classes != null) {
93              for(final Class<?> clazz : classes) {
94                  for(Annotation annotation : clazz.getAnnotations()) {
95                      Set<Class<?>> classSet = classesByAnnotation.get(annotation.annotationType());
96                      if(classSet == null) {
97                          classesByAnnotation.put(annotation.annotationType(), classSet = new HashSet<Class<?>>());
98                      }
99                      classSet.add(clazz);
100                 }
101             }
102         }
103         servletContext.setAttribute(DefaultAnnotationProcessor.ANNOTATION_ATTRIBUTE, classesByAnnotation);
104     }
