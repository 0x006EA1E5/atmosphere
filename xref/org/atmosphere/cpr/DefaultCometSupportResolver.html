89      /**
90       * Convenience method that tests if a class with the given FQN is present on the classpath
91       *
92       * @param testClass
93       * @return true if the class is present
94       */
95      protected boolean testClassExists(final String testClass) {
96          try {
97              return testClass != null && testClass.length() > 0 &&
98                      Thread.currentThread().getContextClassLoader().loadClass(testClass) != null;
99          } catch (ClassNotFoundException ex) {
100             return false;
101         } catch (NoClassDefFoundError ex) {
102             return false;
103         }
104     }

106     /**
107      * Returns a list of comet support by containers available on the classpath
108      *
109      * @return
110      */
111     public List<Class<? extends CometSupport>> detectContainersPresent() {
112         return new LinkedList<Class<? extends CometSupport>>() {
113             {
114                 if (testClassExists(GLASSFISH_V2))
115                     add(GlassFishv2CometSupport.class);
116 
117                 if (testClassExists(JETTY))
118                     add(JettyCometSupport.class);
119 
120                 if (testClassExists(JETTY_7))
121                     add(Jetty7CometSupport.class);
122 
123                 if (testClassExists(JETTY_8))
124                     add(Jetty7CometSupport.class);
125 
126                 if (testClassExists(JBOSSWEB))
127                     add(JBossWebCometSupport.class);
128 
129                 if (testClassExists(TOMCAT_7))
130                     add(Tomcat7CometSupport.class);
131 
132                 if (testClassExists(TOMCAT) || testClassExists(JBOSS_5))
133                     add(TomcatCometSupport.class);
134 
135                 if (testClassExists(GRIZZLY))
136                     add(GrizzlyCometSupport.class);
137 
138                 if (testClassExists(WEBLOGIC))
139                     add(WebLogicCometSupport.class);
140             }
141         };
142     }

144     public List<Class<? extends CometSupport>> detectWebSocketPresent() {
145         List l = new LinkedList<Class<? extends CometSupport>>() {
146             {
147                 if (testClassExists(JETTY_8))
148                     add(JettyCometSupportWithWebSocket.class);
149 
150                 if (testClassExists(GRIZZLY_WEBSOCKET))
151                     add(GlassFishWebSocketSupport.class);
152 
153             }
154         };
155 
156         if (l.isEmpty()) {
157             return detectContainersPresent();
158         }
159         return l;
160     }

162     /**
163      * This method is used to determine the default CometSupport if all else fails
164      *
165      * @param preferBlocking
166      * @return
167      */
168     public CometSupport defaultCometSupport(final boolean preferBlocking) {
169         if (!preferBlocking && testClassExists(SERVLET_30)) {
170             if (detectWebSocketPresent().size() > 0) {
171                 return new Servlet30CometSupportWithWebSocket(config);
172             }
173             return new Servlet30CometSupport(config);
174         } else {
175             return new BlockingIOCometSupport(config);
176         }
177     }

179     /**
180      * Given a Class of something that extends CometSupport, it tries to return an instance of that class
181      * <p/>
182      * The class has to have a visible constructor with the signature (@link {AtmosphereConfig})
183      *
184      * @param targetClass
185      * @return an instance of the specified class
186      */
187     public CometSupport newCometSupport(final Class<? extends CometSupport> targetClass) {
188         try {
189             return targetClass.getDeclaredConstructor(new Class[]{AtmosphereConfig.class})
190                     .newInstance(config);
191         } catch (final Exception e) {
192             logger.error("failed to create comet support class: {}, error: {}", targetClass, e.getMessage());
193             throw new IllegalArgumentException(
194                     "Comet support class " + targetClass.getCanonicalName() + " has bad signature.", e);
195         }
196     }
197 
198     public CometSupport newCometSupport(final String targetClassFQN) {
199         try {
200             ClassLoader cl = Thread.currentThread().getContextClassLoader();
201             return (CometSupport) cl.loadClass(targetClassFQN)
202                     .getDeclaredConstructor(new Class[]{AtmosphereConfig.class}).newInstance(config);
203         } catch (final Exception e) {
204             logger.error("failed to create comet support class: {}, error: {}", targetClassFQN, e.getMessage());
205             throw new IllegalArgumentException("Comet support class " + targetClassFQN + " has bad signature.", e);
206         }
207     }

209     /**
210      * This method is the general interface to the outside world
211      *
212      * @param useNativeIfPossible - should the resolver try to use a native container comet support if present?
213      * @param defaultToBlocking   - should the resolver default to blocking IO comet support?
214      * @return an instance of CometSupport
215      */
216     public CometSupport resolve(final boolean useNativeIfPossible, final boolean defaultToBlocking) {
217         final CometSupport servletAsyncSupport = defaultCometSupport(defaultToBlocking);
218 
219         final CometSupport nativeSupport;
220         if (!defaultToBlocking && (useNativeIfPossible ||
221                 servletAsyncSupport.getClass().getName().equals(BlockingIOCometSupport.class.getName()))) {
222             nativeSupport = resolveNativeCometSupport(detectContainersPresent());
223             return nativeSupport == null ? servletAsyncSupport : nativeSupport;
224         }
225         return servletAsyncSupport;
226     }
227 
228     public CometSupport resolve(boolean useNativeIfPossible, boolean defaultToBlocking, boolean useWebsocketIfPossible) {
229         CometSupport cs;
230         if (!useWebsocketIfPossible) {
231             cs = resolve(useNativeIfPossible, defaultToBlocking);
232         } else {
233             cs = resolveWebSocket(detectWebSocketPresent());
234         }
235 
236         if (cs == null) {
237             return new BlockingIOCometSupport(config);
238         } else {
239             return cs;
240         }
241     }
242 
243     public CometSupport resolveWebSocket(final java.util.List<Class<? extends CometSupport>> available) {
244         if (available == null || available.isEmpty()) return null;
245         else return newCometSupport(available.get(0));
246     }
247 
248     /**
249      * This method is called to determine which native comet support to the used
250      *
251      * @param available
252      * @return the result of @link {resolveMultipleNativeSupportConflict} if there are more than 1 item in the list of available ontainers
253      */
254     protected CometSupport resolveNativeCometSupport(final java.util.List<Class<? extends CometSupport>> available) {
255         if (available == null || available.isEmpty()) return null;
256         else if (available.size() == 1) return newCometSupport(available.get(0));
257         else return resolveMultipleNativeSupportConflict(available);
258     }
259 
260     /**
261      * This method is called if there are more than one potential native container in scope
262      *
263      * @return a CometSupport instance
264      */
265     protected CometSupport resolveMultipleNativeSupportConflict(final List<Class<? extends CometSupport>> available) {
266         final StringBuilder b = new StringBuilder("Found multiple containers, please specify which one to use: ");
267         for (Class<? extends CometSupport> cs : available) {
268             b.append((cs != null) ? cs.getCanonicalName() : "null").append(", ");
269         }
270 
271         b.append(" until you do, Atmosphere will use:" + available.get(0));
272         logger.warn("{}", b.toString());
273         return newCometSupport(available.get(0));
274     }
