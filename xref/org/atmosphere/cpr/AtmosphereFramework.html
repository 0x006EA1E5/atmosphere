354     /**
355      * Add an {@link AtmosphereHandler} serviced by the {@link Servlet}
356      * This API is exposed to allow embedding an Atmosphere application.
357      *
358      * @param mapping The servlet mapping (servlet path)
359      * @param h       implementation of an {@link AtmosphereHandler}
360      * @param l       An array of {@link AtmosphereInterceptor}.
361      */
362     public AtmosphereFramework addAtmosphereHandler(String mapping, AtmosphereHandler h, List<AtmosphereInterceptor> l) {
363         if (!mapping.startsWith("/")) {
364             mapping = "/" + mapping;
365         }
366         AtmosphereHandlerWrapper w = new AtmosphereHandlerWrapper(broadcasterFactory, h, mapping);
367         w.interceptors = l;
368         addMapping(mapping, w);
369 
370         if (isInit) {
371             try {
372                 if (h instanceof AtmosphereServletProcessor) {
373                     ((AtmosphereServletProcessor) h).init(config);
374                 }
375             } catch (ServletException e) {
376                 throw new RuntimeException(e);
377             }
378             initHandlerInterceptors(w);
379         } else {
380             logger.info("Installed AtmosphereHandler {} mapped to context-path: {}", h.getClass().getName(), mapping);
381             logger.info("Installed the following AtmosphereInterceptor mapped to AtmosphereHandler {}", h.getClass().getName());
382             if (l.size() > 0) {
383                 for (AtmosphereInterceptor s : l) {
384                     logger.info("\t{} : {}", s.getClass().getName(), s);
385                 }
386             }
387         }
388         return this;
389     }

403     private AtmosphereFramework addMapping(String path, AtmosphereHandlerWrapper w) {
404         // We are using JAXRS mapping algorithm.
405         if (path.contains("*")) {
406             path = path.replace("*", mappingRegex);
407         }
408 
409         if (path.endsWith("/")) {
410             path = path + mappingRegex;
411         }
412 
413         atmosphereHandlers.put(path, w);
414         return this;
415     }

1710         req.setAttribute(BROADCASTER_FACTORY, BroadcasterFactory.getDefault());
1711         req.setAttribute(PROPERTY_USE_STREAM, useStreamForFlushingComments);
1712         req.setAttribute(BROADCASTER_CLASS, broadcasterClassName);
1713         req.setAttribute(ATMOSPHERE_CONFIG, config);
1714         req.setAttribute(FrameworkConfig.THROW_EXCEPTION_ON_CLONED_REQUEST, "" + config.isThrowExceptionOnCloned());

1716         String s = config.getInitParameter(ALLOW_QUERYSTRING_AS_REQUEST);
1717         if (s != null) {
1718             skip = Boolean.valueOf(s);
1719         }
1720 
1721         if (!skip || req.getAttribute(WEBSOCKET_SUSPEND) == null) {
1722             Map<String, String> headers = configureQueryStringAsRequest(req);
1723             String body = headers.remove(ATMOSPHERE_POST_BODY);
1724             if (body != null && body.isEmpty()) {
1725                 body = null;
1726             }
1727 
1728             // Reconfigure the request. Clear the Atmosphere queryString
1729             req.headers(headers)
1730                     .method(body != null && req.getMethod().equalsIgnoreCase("GET") ? "POST" : req.getMethod());
1731 
1732             if (body != null) {
1733                 req.body(URLDecoder.decode(body, req.getCharacterEncoding() == null ? "UTF-8" : req.getCharacterEncoding()));
1734             }
1735         }
1736 
1737         s = req.getHeader(X_ATMOSPHERE_TRACKING_ID);
1738 
1739         // Lookup for websocket
1740         if (s == null || s.equals("0")) {
1741             String unique = config.getInitParameter(ApplicationConfig.UNIQUE_UUID_WEBSOCKET);
1742             if (unique != null && Boolean.valueOf(unique)) {

1744             }
1745         }
1746

1748             s = UUID.randomUUID().toString();
1749             res.setHeader(X_ATMOSPHERE_TRACKING_ID, s);
1750         } else {
1751             // This may breaks 1.0.0 application because the WebSocket's associated AtmosphereResource will
1752             // all have the same UUID, and retrieving the original one for WebSocket, so we don't set it at all.
1753             // Null means it is not an HTTP request.
1754             if (req.resource() == null) {
1755                 res.setHeader(X_ATMOSPHERE_TRACKING_ID, s);

1759                 res.setHeader(X_ATMOSPHERE_TRACKING_ID, s);
1760             }

1762

1764             req.setAttribute(SUSPENDED_ATMOSPHERE_RESOURCE_UUID, s);
1765         }

1779         Action a = null;

1785                 if (!isFilter) {

1787                             ex.getMessage());

1794                 if (current instanceof AsynchronousProcessor) {
1795                     ((AsynchronousProcessor) current).shutdown();
1796                 }
1797 
1798                 asyncSupport.init(config.getServletConfig());
1799                 logger.warn("Using " + asyncSupport.getClass().getName());
1800

1802             } else {
1803                 logger.error("AtmosphereFramework exception", ex);
1804                 throw ex;
1805             }
1806         } finally {
1807             if (a != null) {
1808                 notify(a.type(), req, res);
1809             }

1812                 if (req != null && a != null && a.type() != Action.TYPE.SUSPEND) {
1813                     req.destroy();

1818         }

1943     protected Map<String, String> configureQueryStringAsRequest(AtmosphereRequest request) {
1944         Map<String, String> headers = new HashMap<String, String>();
1945 
1946         StringBuilder q = new StringBuilder();
1947         try {
1948             String qs = request.getQueryString();
1949             if (qs != null && !qs.isEmpty()) {
1950                 String[] params = qs.split("&");
1951                 String[] s;
1952                 for (String p : params) {
1953                     s = p.split("=", 2);;
1954                     if (s[0].equalsIgnoreCase("Content-Type")) {
1955                         // Use the one set by the user first.
1956                         if (request.getContentType() == null ||
1957                                 !request.getContentType().equalsIgnoreCase(s.length > 1 ? s[1] : "")) {
1958                             request.contentType(s.length > 1 ? URLDecoder.decode(s[1], "UTF-8") : "");
1959                         }
1960                     }
1961                     if (!s[0].isEmpty()
1962                             && !s[0].toLowerCase().startsWith("x-atmo")
1963                             && !s[0].equalsIgnoreCase("x-cache-date")
1964                             && !s[0].equalsIgnoreCase("Content-Type")
1965                             && !s[0].equalsIgnoreCase("_")) {
1966                         q.append(s[0]).append("=").append(s.length > 1 ? s[1] : "").append("&");
1967                     }
1968                     headers.put(s[0], s.length > 1 ? s[1] : "");
1969                 }
1970             }
1971         } catch (Exception ex) {
1972             logger.error("Unable to parse query string", ex);
1973         }
1974         String disallowModifyQueryString = config.getInitParameter(ApplicationConfig.DISALLOW_MODIFY_QUERYSTRING);
1975         if (disallowModifyQueryString == null ||
1976                 disallowModifyQueryString.length() == 0 || "false".equalsIgnoreCase(disallowModifyQueryString)) {
1977             if (q.length() > 0) {
1978                 q.deleteCharAt(q.length() - 1);
1979             }
1980             request.queryString(q.toString());
1981         }
1982 
1983         logger.trace("Query String translated to headers {}", headers);
1984         return headers;
1985     }

2100     protected void positionInterceptor(InvokationOrder.PRIORITY p, AtmosphereInterceptor c) {
2101         switch (p) {
2102             case AFTER_DEFAULT:
2103                 interceptors.addLast(c);
2104                 break;
2105             case BEFORE_DEFAULT:
2106                 int pos = executeFirstSet ? 1 : 0;
2107                 interceptors.add(pos, c);
2108                 break;
2109             case FIRST_BEFORE_DEFAULT:
2110                 if (executeFirstSet)
2111                     throw new IllegalStateException("Cannot set more than one AtmosphereInterceptor to be executed first");
2112                 logger.info("AtmosphereInterceptor {} will always be executed first", c);
2113                 interceptors.addFirst(c);
2114                 executeFirstSet = true;
2115                 break;
2116         }
2117     }
