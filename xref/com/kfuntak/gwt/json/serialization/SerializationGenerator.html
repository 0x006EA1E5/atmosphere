37      public String generate(TreeLogger logger, GeneratorContext ctx,
38              String requestedClass) throws UnableToCompleteException {
39          //get the type oracle
40          typeOracle = ctx.getTypeOracle();
41          assert (typeOracle != null);
42          serializeInterface = typeOracle.findType(JsonSerializable.class.getName());
43          assert (serializeInterface != null);
44          stringClass = typeOracle.findType(String.class.getName());
45          assert (stringClass != null);
46  
47          //get class from type oracle
48          JClassType serializeClass = typeOracle.findType(requestedClass);
49  
50          if (serializeClass == null) {
51              logger.log(TreeLogger.ERROR, "Unable to find metadata for type '"
52                      + requestedClass + "'", null);
53              throw new UnableToCompleteException();
54          }
55  
56          //create source writer
57          String packageName = serializeClass.getPackage().getName();
58          className = serializeClass.getSimpleSourceName() + "_TypeSerializer";
59          PrintWriter printWriter = ctx.tryCreate(logger, packageName, className);
60          if (printWriter == null) {
61              return packageName + "." + className;
62          }
63          ClassSourceFileComposerFactory composerFactory =
64                  new ClassSourceFileComposerFactory(packageName, className);
65          composerFactory.setSuperclass("com.kfuntak.gwt.json.serialization.client.Serializer");
66  
67  //		// Java imports
68          composerFactory.addImport(java.util.Collection.class.getName());
69          composerFactory.addImport(java.util.List.class.getName());
70          composerFactory.addImport(java.util.ArrayList.class.getName());
71          composerFactory.addImport(java.util.LinkedList.class.getName());
72          composerFactory.addImport(java.util.Stack.class.getName());
73          composerFactory.addImport(java.util.Vector.class.getName());
74          composerFactory.addImport(java.util.Set.class.getName());
75          composerFactory.addImport(java.util.TreeSet.class.getName());
76          composerFactory.addImport(java.util.HashSet.class.getName());
77          composerFactory.addImport(java.util.LinkedHashSet.class.getName());
78          composerFactory.addImport(java.util.SortedSet.class.getName());
79          composerFactory.addImport(java.util.Date.class.getName());
80  //		// GWT imports
81          composerFactory.addImport(com.google.gwt.core.client.GWT.class.getName());
82          composerFactory.addImport(com.google.gwt.json.client.JSONNull.class.getName());
83          composerFactory.addImport(com.google.gwt.json.client.JSONNumber.class.getName());
84          composerFactory.addImport(com.google.gwt.json.client.JSONString.class.getName());
85          composerFactory.addImport(com.google.gwt.json.client.JSONValue.class.getName());
86          composerFactory.addImport(com.google.gwt.json.client.JSONObject.class.getName());
87          composerFactory.addImport(com.google.gwt.json.client.JSONArray.class.getName());
88          composerFactory.addImport(com.google.gwt.json.client.JSONBoolean.class.getName());
89          composerFactory.addImport(com.google.gwt.json.client.JSONParser.class.getName());
90          composerFactory.addImport(com.google.gwt.json.client.JSONException.class.getName());
91  //		// Module imports
92          composerFactory.addImport(ObjectSerializer.class.getName());
93          composerFactory.addImport(JsonSerializable.class.getName());
94          composerFactory.addImport(IncompatibleObjectException.class.getName());
95          composerFactory.addImport(SerializerHelper.class.getName());
96          composerFactory.addImport(DeserializerHelper.class.getName());
97  
98          JClassType[] subTypes = serializeInterface.getSubtypes();
99          for (int i = 0; i < subTypes.length; ++i) {
100             composerFactory.addImport(subTypes[i].getQualifiedSourceName());
101         }
102 
103         srcWriter = composerFactory.createSourceWriter(ctx, printWriter);
104         if (srcWriter == null) {
105             return packageName + "." + className;
106         }
107 
108         //create a serializer for each interface that supports Serializable
109         for (int i = 0; i < subTypes.length; ++i) {
110             if (subTypes[i].isAbstract()) {
111                 continue;
112             }
113             srcWriter.println("public class " + subTypes[i].getName() + "_SerializableImpl implements ObjectSerializer{");
114 //					System.out.println("public class "+subTypes[i].getName()+"_SerializableImpl implements ObjectSerializer{");
115             srcWriter.indent();
116             srcWriter.println("public " + subTypes[i].getName() + "_SerializableImpl(){}");
117 //					System.out.println("public "+subTypes[i].getName()+"_SerializableImpl(){}");
118 
119             StringBuffer buffer = new StringBuffer();
120             try {
121                 String defaultSerializationString = generateDefaultSerialization();
122                 String typeSerializationString = generateTypeSerialization(subTypes[i].getQualifiedSourceName());
123                 String defaultDeserializationString = generateDefaultDeserialization(subTypes[i].getQualifiedSourceName());
124                 String tyepDeserializationString = generateTypeDeserialization(subTypes[i].getQualifiedSourceName());
125 
126                 buffer.append(defaultSerializationString);
127                 buffer.append("\n");
128                 buffer.append(typeSerializationString);
129                 buffer.append("\n");
130                 buffer.append(defaultDeserializationString);
131                 buffer.append("\n");
132                 buffer.append(tyepDeserializationString);
133                 buffer.append("\n");
134                 buffer.append("}");
135                 buffer.append("\n");
136                 //System.out.println(buffer.toString());
137             } catch (NotFoundException e) {
138                 e.printStackTrace();
139             }
140 
141             srcWriter.println(buffer.toString());
142 //					System.out.println(buffer.toString());
143         }
144 
145 
146         //in the class constructor, add each serializer
147         srcWriter.println("public " + className + "(){");
148 //		System.out.println("public "+className+"(){");
149         srcWriter.indent();
150         for (int i = 0; i < subTypes.length; ++i) {
151             if (subTypes[i].isAbstract()) {
152                 continue;
153             }
154             srcWriter.println("addObjectSerializer(\"" + subTypes[i].getQualifiedSourceName() + "\", new " + subTypes[i].getName() + "_SerializableImpl() );");
155 //					System.out.println("addObjectSerializer(\""+subTypes[i].getQualifiedSourceName()+"\", new "+subTypes[i].getName()+"_SerializableImpl() );");
156         }
157         srcWriter.outdent();
158         srcWriter.println("}");
159 //		System.out.println("}");
160 
161         srcWriter.commit(logger);
162         return packageName + "." + className;
163     }

166 
167         JClassType baseType = typeOracle.getType(typeName);
168         String packageName = baseType.getPackage().getName();
169 
170         StringBuffer buffer = new StringBuffer();

175         buffer.append("if(jsonValue instanceof JSONNull){");
176         buffer.append("\n");
177         buffer.append("return null;");
178         buffer.append("\n");
179         buffer.append("}");

181 
182         // Throw Incompatible exception is JsonValue is not an instance of
183         // JsonObject
184         buffer.append("if(!(jsonValue instanceof JSONObject)){");

187         buffer.append("\n");
188         buffer.append("}");

190 
191         // Initialise JsonObject then
192         String baseTypeName = baseType.getSimpleSourceName();
193         buffer.append("JSONObject jsonObject=(JSONObject)jsonValue;");
194         buffer.append("\n");
195         buffer.append(baseTypeName + " mainResult=new " + baseTypeName + "();");

202         buffer.append("\n");
203         buffer.append("JSONValue fieldJsonValue=null;");
204         buffer.append("\n");
205 
206         // Start deSerialisation
207         List<JField> allFields = new ArrayList<JField>();
208         JField[] fields = baseType.getFields();
209         for (JField field : fields) {
210             if (!field.isStatic() && !field.isTransient()) {
211                 allFields.add(field);

217             while (flag) {

220                     JField[] subClassFields = superClassType.getFields();
221                     for (JField subClassField : subClassFields) {
222                         if (!subClassField.isStatic() && !subClassField.isTransient()) {
223                             allFields.add(subClassField);
224                         }
225                     }
226                 } else {
227                     flag = false;
228                 }
229             }
230         }
231         fields = new JField[allFields.size()];
232         allFields.toArray(fields);
233 
234         for (JField field : fields) {
235             JType fieldType = field.getType();
236             String fieldName = field.getName();
237             String fieldNameForGS = getNameForGS(fieldName);
238             buffer.append("fieldJsonValue=jsonObject.get(\"" + fieldName + "\");");
239             buffer.append("\n");

319     private void deserializeCollection(StringBuffer buffer, JClassType fieldClassType, String fieldNameForGS, String fieldName) throws NotFoundException {
320         // Return null if JSON object is null
321         buffer.append("if(fieldJsonValue==null || fieldJsonValue instanceof JSONNull){");
322         buffer.append("\n");
323         buffer.append("mainResult.set" + fieldNameForGS + "(null);");
324         buffer.append("\n");
325         buffer.append("return mainResult;");
326         buffer.append("\n");
327         buffer.append("}");
328         buffer.append("\n");
329 
330         // Throw Incompatible exception if the JSON object is not a collection
331         buffer.append("if(!(fieldJsonValue instanceof JSONArray)){");
332         buffer.append("\n");
333         buffer.append("throw new IncompatibleObjectException();");
334         buffer.append("\n");
335         buffer.append("}");
336         buffer.append("\n");
337 
338         // Start deSerilisation
339         buffer.append("inputJsonArray=(JSONArray)fieldJsonValue;");
340         buffer.append("\n");
341         buffer.append("inpJsonArSize=inputJsonArray.size();");
342         buffer.append("\n");
343 
344         String fieldTypeQualifiedName = fieldClassType.getQualifiedSourceName();
345         JParameterizedType parameterizedType = (JParameterizedType) fieldClassType;
346         fieldClassType = parameterizedType.getTypeArgs()[0];
347         String parameterSimpleName = fieldClassType.getSimpleSourceName();
348         String fieldColName = fieldName + "Col";// Field Collection Result
349         // Object Name
350         importsList.add(fieldClassType.getQualifiedSourceName());
351         if (fieldTypeQualifiedName.equals("java.util.List") || fieldTypeQualifiedName.equals("java.util.ArrayList")) {
352             buffer.append("ArrayList<" + parameterSimpleName + "> " + fieldColName + " = new ArrayList<" + parameterSimpleName + ">();");
353             buffer.append("\n");
354         } else if (fieldTypeQualifiedName.equals("java.util.Set") || fieldTypeQualifiedName.equals("java.util.HashSet")) {
355             buffer.append("HashSet<" + parameterSimpleName + "> " + fieldColName + " = new HashSet<" + parameterSimpleName + ">();");
356             buffer.append("\n");
357         } else if (fieldTypeQualifiedName.equals("java.util.SortedSet") || fieldTypeQualifiedName.equals("java.util.TreeSet")) {
358             buffer.append("TreeSet<" + parameterSimpleName + "> " + fieldColName + " = new TreeSet<" + parameterSimpleName + ">();");
359             buffer.append("\n");
360         } else if (fieldTypeQualifiedName.equals("java.util.LinkedList")) {
361             buffer.append("LinkedList<" + parameterSimpleName + "> " + fieldColName + " = new LinkedList<" + parameterSimpleName + ">();");
362             buffer.append("\n");
363             buffer.append("mainResult.set" + fieldNameForGS + "(" + fieldColName + ");");
364             buffer.append("\n");
365         } else if (fieldTypeQualifiedName.equals("java.util.Stack")) {
366             buffer.append("Stack<" + parameterSimpleName + "> " + fieldColName + " = new Stack<" + parameterSimpleName + ">();");
367             buffer.append("\n");
368         } else if (fieldTypeQualifiedName.equals("java.util.Vector")) {
369             buffer.append("Vector<" + parameterSimpleName + "> " + fieldColName + " = new Vector<" + parameterSimpleName + ">();");
370             buffer.append("\n");
371         } else if (fieldTypeQualifiedName.equals("java.util.LinkedHashSet")) {
372             buffer.append("LinkedHashSet<" + parameterSimpleName + "> " + fieldColName + "=new LinkedHashSet<" + parameterSimpleName + ">();");
373             buffer.append("\n");
374         }
375         buffer.append("for(int ij=0;ij<inpJsonArSize;ij++){");
376         // DeSerialise individual elements
377         buffer.append("fieldJsonValue=inputJsonArray.get(ij);");
378         if (fieldClassType.getQualifiedSourceName().equals("java.lang.Short")) {
379             buffer.append(fieldColName + ".add(DeserializerHelper.getShort(fieldJsonValue));");
380             buffer.append("\n");
381         } else if (fieldClassType.getQualifiedSourceName().equals("java.lang.Byte")) {
382             buffer.append(fieldColName + ".add(DeserializerHelper.getByte(fieldJsonValue));");
383             buffer.append("\n");
384         } else if (fieldClassType.getQualifiedSourceName().equals("java.lang.Long")) {
385             buffer.append(fieldColName + ".add(DeserializerHelper.getLong(fieldJsonValue));");
386             buffer.append("\n");
387         } else if (fieldClassType.getQualifiedSourceName().equals("java.lang.Integer")) {
388             buffer.append(fieldColName + ".add(DeserializerHelper.getInt(fieldJsonValue));");
389             buffer.append("\n");
390         } else if (fieldClassType.getQualifiedSourceName().equals("java.lang.Float")) {
391             buffer.append(fieldColName + ".add(DeserializerHelper.getFloat(fieldJsonValue));");
392             buffer.append("\n");
393         } else if (fieldClassType.getQualifiedSourceName().equals("java.lang.Double")) {
394             buffer.append(fieldColName + ".add(DeserializerHelper.getDouble(fieldJsonValue));");
395             buffer.append("\n");
396         } else if (fieldClassType.getQualifiedSourceName().equals("java.lang.Boolean")) {
397             buffer.append(fieldColName + ".add(DeserializerHelper.getBoolean(fieldJsonValue));");
398             buffer.append("\n");
399         } else if (fieldClassType.getQualifiedSourceName().equals("java.lang.Character")) {
400             buffer.append(fieldColName + ".add(DeserializerHelper.getShort(fieldJsonValue));");
401             buffer.append("\n");
402         } else if (fieldClassType.getQualifiedSourceName().equals("java.util.Date")) {
403             buffer.append(fieldColName + ".add(DeserializerHelper.getDate(fieldJsonValue));");
404             buffer.append("\n");
405         } else if (fieldClassType.isAssignableTo(typeOracle.getType("com.kfuntak.gwt.json.serialization.client.JsonSerializable"))) {
406             importsList.add(fieldClassType.getQualifiedSourceName());
407             buffer.append("serializer = GWT.create(Serializer.class);");
408             buffer.append("\n");
409             buffer.append("JSONValue _class = ((JSONObject)fieldJsonValue).get(\"class\");");
410             buffer.append("\n");
411             buffer.append("if (_class != null && _class instanceof JSONString) {");
412             buffer.append("\n");
413             buffer.append(fieldColName + ".add((" + fieldClassType.getSimpleSourceName() + ")serializer.deSerialize(fieldJsonValue, ((JSONString)_class).stringValue()));");
414             buffer.append("\n");
415             buffer.append("} else {");
416             buffer.append(fieldColName + ".add((" + fieldClassType.getSimpleSourceName() + ")serializer.deSerialize(fieldJsonValue, \"" + fieldClassType.getQualifiedSourceName() + "\"));");
417             buffer.append("}");
418             buffer.append("\n");
419         } else if (fieldClassType.getQualifiedSourceName().equals("java.lang.String")) {
420             buffer.append(fieldColName + ".add(DeserializerHelper.getString(fieldJsonValue));");
421             buffer.append("\n");
422         }
423         buffer.append("}");
424         buffer.append("\n");
425         buffer.append("mainResult.set" + fieldNameForGS + "(" + fieldColName + ");");
426         buffer.append("\n");
427     }

440     private String generateTypeSerialization(String typeName) throws NotFoundException {
441 
442         JClassType baseType = typeOracle.getType(typeName);
443         String packageName = baseType.getPackage().getName();
444 
445         StringBuffer buffer = new StringBuffer();
446         buffer.append("public JSONValue serializeToJson(Object object){");
447         buffer.append("\n");
448 
449         // Return JSONNull instance if object is null
450         buffer.append("if(object==null){");
451         buffer.append("\n");
452         buffer.append("return JSONNull.getInstance();");
453         buffer.append("\n");
454         buffer.append("}");
455         buffer.append("\n");
456 
457         // Throw Incompatible Exception if object is not of the type it claims
458         // to be
459         buffer.append("if(!(object instanceof " + baseType.getSimpleSourceName() + ")){");
460         buffer.append("\n");
461         buffer.append("throw new IncompatibleObjectException();");
462         buffer.append("\n");
463         buffer.append("}");
464         buffer.append("\n");
465 
466         // Initialise result object
467         buffer.append("JSONObject mainResult=new JSONObject();");
468         buffer.append("\n");
469         buffer.append("JSONValue jsonValue=null;");
470         buffer.append("\n");
471         buffer.append("JSONArray jsonResultArray=null;");
472         buffer.append("\n");
473         buffer.append("int index=0;");
474         buffer.append("\n");
475         buffer.append("Serializer serializer=null;");
476         buffer.append("\n");
477         buffer.append("Object fieldValue=null;");
478         buffer.append("\n");
479         buffer.append(baseType.getSimpleSourceName() + " mainVariable=(" + baseType.getSimpleSourceName() + ")object;");
480         buffer.append("\n");
481 
482         // Serialise fields
483         List<JField> allFields = new ArrayList<JField>();
484         JField[] fields = baseType.getFields();
485         for (JField field : fields) {
486             if (!field.isStatic() && !field.isTransient()) {
487                 allFields.add(field);
488             }
489         }
490         if (baseType.isAssignableTo(typeOracle.getType("com.kfuntak.gwt.json.serialization.client.JsonSerializable"))) {
491             boolean flag = true;
